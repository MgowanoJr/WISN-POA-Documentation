import { NgxDhis2HttpClientService } from '@hisptz/ngx-dhis2-http-client';
import {forkJoin, from, interval, Observable, of, throwError} from 'rxjs';
import * as _ from 'lodash';
import { RequestOptions, Headers, Http } from '@angular/http';
import {delay, flatMap, mergeMap, retryWhen, share, switchMap} from 'rxjs/operators';
import {WisnPoaPageService} from './wisn-poa-page.service';
import * as fromUtilHelpers from '../../shared/helpers/util-functions';

@Injectable({
  providedIn: 'root'
})
export class WisnAnalyticsService {
  options: any;

  constructor(
    private httpClient: NgxDhis2HttpClientService,
    private wisnPoaService: WisnPoaPageService,
    private httpProvider: Http
  ) {
    const reqstHeadDon = new Headers({
      Accept: '*/*',
      'Access-Control-Allow-Origin': '*',
      'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8'
    });
    this.options = new RequestOptions({ headers: reqstHeadDon });
  }

  getWISNdataNoaggregations(orgunitId, period, dx, dataEntryLevel): Observable<any> {
    const orgunitArray = [orgunitId];
    return from(orgunitArray).pipe(
      mergeMap((orgunit, index) => {
        const url =
          'analytics?dimension=dx:' +
          dx.substring(1) +
          '' +
          '&dimension=ou:' +
          orgunit + ';LEVEL-' + dataEntryLevel +
          '&filter=pe:' +
          period +
          '&displayProperty=NAME&skipMeta=true';
        return this.httpClient.get(url).pipe();
      }, 10)
    );
  }

  getWISNHMISdataNoaggregations(childOrgunits: any[], period, dx): Observable<any> {
    return from(childOrgunits).pipe(
      mergeMap((orgunit, index) => {
        const url =
          'analytics?dimension=dx:' +
          dx.substring(1) +
          '' +
          '&dimension=ou:' +
          orgunit.id +
          '&filter=pe:' +
          period +
          '&displayProperty=NAME&skipMeta=false';
        return this.httpClient.get(url).pipe();
      }, 10)
    );
  }

  getDataSetDataValues(dataSetId, orgUnitId, period): Observable<any> {
    const url =
      '../../../dhis-web-dataentry/getDataValues.action?dataSetId=' +
      dataSetId +
      '&periodId=' +
      period +
      '' +
      '&organisationUnitId=' +
      orgUnitId +
      '&multiOrganisationUnit=true';
    return this.httpProvider.get(url).pipe();
  }

  getDataSetDesign(dataSetId): Observable<any> {
    const url = 'dataSets/' + dataSetId + '/metadata.json';
    return this.httpClient.get(url).pipe();
  }

  getDataValues(dataSetId, orgUnitId, period): Observable<any> {
    const url =
      'dataValueSets.json?dataSet=' +
      dataSetId +
      '&period=' +
      period +
      '&orgUnit=' +
      orgUnitId +
      '';
    return this.httpClient.get(url).pipe();
  }

  getMultipleOrgunitDatavalues(dataSetId, period, orgUnits: any[]): Observable<any> {
    return from(orgUnits).pipe(
      mergeMap((orgUnit, index) => {
        const url = `dataValueSets.json?dataSet=${dataSetId}&period=${period}&orgUnit=${orgUnit.id}`;
          return <Observable<any>>
            this.httpClient.get(url)
              .pipe(retryWhen(() => {
              return interval(5000).pipe(
                flatMap(count => count === 3 ? throwError('Giving up') : of(count))
              );
            }));
      }, 10)
    );
  }

  getMultipleWisnReportAnalytics(childrenOrgunits: any[], dataEntryLevel, indicators, period): Observable<any> {
    return from(childrenOrgunits).pipe(
      mergeMap((orgunit, index) => {
        // tslint:disable-next-line:max-line-length
        const url = `analytics?dimension=dx:${indicators}&dimension=ou:${orgunit.id};LEVEL-${dataEntryLevel}&filter=pe:${period}&displayProperty=NAME&hierarchyMeta=true`;
          return <Observable<any>>
            this.httpClient.get(url)
              .pipe();
      }, 10)
    );
  }

  getSingleIndicatorWisnReportAnalytics(orgunitIndicatorArray: any[], dataEntryLevel, selectedOrgunit, period): Observable<any> {
    return from(orgunitIndicatorArray).pipe(
      mergeMap((orgunit) => {
         // tslint:disable-next-line:max-line-length
        const url = `analytics?dimension=dx:${orgunit.dataElementId}&dimension=ou:${orgunit.orgunitId};LEVEL-${dataEntryLevel}&filter=pe:${period}&displayProperty=NAME&hierarchyMeta=true`;
        return <Observable<any>>
          this.httpClient.get(url)
          .pipe();
      }, 10)
    );
  }

  getLatestMultipleEventDataValues(childrenOrgunits: any[], program): Observable<any> {
    return from(childrenOrgunits).pipe(
      mergeMap((orgunit, index) => {
        const url = `events.json?orgUnit=${orgunit.id}&program=${program}&order=eventDate:desc&paging=false`;
          return <Observable<any>>
            this.httpClient.get(url)
              .pipe();
      }, 10)
    );
  }

  getLatestMultipleChildrenEventDataValues(selectedOrgunitId, program): Observable<any> {
    const url = `events.json?orgUnit=${selectedOrgunitId}&program=${program}&order=eventDate:desc&paging=false&ouMode=CHILDREN`;
    return this.httpClient.get(url).pipe();
  }

  getMultipleHRHanalyticsData(childrenOrgunits: any[], periodId, indicators): Observable<any> {
    return from(childrenOrgunits).pipe(
      mergeMap((orgunit, index) => {
         // tslint:disable-next-line:max-line-length
        const url = `analytics?dimension=dx:${indicators}&dimension=ou:${orgunit.id}&filter=pe:${periodId}&displayProperty=NAME&skipMeta=false`;
          return <Observable<any>>
            this.httpClient.get(url).pipe();
      }, 10)
    );
  }

  getWisnHRHeventData(program, orgunits, period, dx): Observable<any> {
    const url =
      'analytics/events/query/' +
      program.id +
      '.json?dimension=pe:' +
      period.id +
      '&dimension=ou:' +
      orgunits.substring(1) +
      '&' +
      '' +
      dx +
      '&displayProperty=NAME&outputType=EVENT&pageSize=500&page=1';
    return this.httpClient.get(url).pipe();
  }

  getDatasetSectionDataElements(form): Observable<any> {
    const url =
      form.type +
      '/' +
      form.id +
      '/sections.json?&fields=id,name,dataElements[id,name,code,valueType,shortName]';
    return this.httpClient.get(url).pipe();
  }

  getProgramSectionDataElements(form): Observable<any> {
    const url =
      form.type +
      '/' +
      form.id +
      '/programStages.json?fields=programStageSections[id,name,dataElements[id,name,code]]';
    return this.httpClient.get(url).pipe();
  }

  importSingleDataValue(payload): Observable<any> {
    const url = '../../../api/dataValues';
    return this.httpProvider.post(url, payload, this.options).pipe();
  }

  uploadMultipleOrgunitDataValues(hmisDataToUpload: any[]): Observable<any> {
    return from(hmisDataToUpload).pipe(
      mergeMap((data: any, index) => {
        const payload = 'de=' + data.dx + '&co=' + data.dataSet.co + '&ds=' + data.dataSet.id
          + '&ou=' + data.orgunitId + '&pe=' + data.period + '&value=' + data.value;
          return <Observable<any>>
          this.httpProvider.post('../../../api/dataValues', payload, this.options);
      }, 10)
    );
  }

  uploadWisnDataFromCSV(wisnReportDataValues: any[]): Observable<any> {
    return from(wisnReportDataValues).pipe(
      mergeMap((dataValueSet: any, index) => {
          return <Observable<any>>
          this.httpClient.post('dataValueSets', dataValueSet, this.options);
      }, 10)
    );
  }

  updateSingleProgramDataValue(payload): Observable<any> {
    // const url = 'events/' + payload.event;
    const url = 'events.json';
    return this.httpClient.post(url, payload).pipe();
  }

  uploadBulkEvents(eventsPayload: any[]): Observable<any> {
    const url = 'events.json';
    return from(eventsPayload).pipe(
      mergeMap((event: any) => {
        return <Observable<any>>
        this.httpClient.post(url, event);
      })
    );
  }

  uploadSingleProgramDataValue(payload): Observable<any> {
    const url = 'events/' + payload.event;
    return this.httpClient.post(url, payload).pipe();
  }

  getSingleDataValues(facilities, dataSetId, period) {
    _.map(facilities, facility => {
      return this.httpClient.get(
        'dataValueSets.json?dataSet=' +
          dataSetId +
          '&period=' +
          period +
          '&orgUnit=' +
          facility +
          ''
      );
    });
  }

  getAllOrganisationUnitsBasedOnType(orgunitTypegroupUid): Observable<any> {
    const url = `organisationUnitGroups/${orgunitTypegroupUid}.json?fields=name,id,organisationUnits`;
    return this.httpClient.get(url);
  }

  importBulkDatavalues(dataValusets: any[]): Observable<any> {
    const url = 'dataValueSets';
    return from(dataValusets).pipe(
      mergeMap((dataValueSet) => {
        return <Observable<any>>
        this.httpClient.post(url, dataValueSet, this.options);
      }, 15)
    );
  }

}
