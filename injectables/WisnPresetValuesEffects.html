import {Injectable} from '@angular/core';
import {Store} from '@ngrx/store';
import {WisnAnalyticsService} from '../../shared/services/wisn-analytics.service';
import {Actions, Effect, ofType} from '@ngrx/effects';
import {AppState} from '../reducers';
import {catchError, map, switchMap, tap, withLatestFrom} from 'rxjs/operators';
import * as fromWisnPresetValues from '../actions/wisn-preset-value.actions';
import * as fromWisnReportActions from '../actions/wisn-report.actions';
import * as fromWisnHMISactions from '../actions/wisn-hmis-data.actions';
import {
  AddWisnPresetValue,
  AddWisnPresetValues, LoadWisnPresetValues, PresetValuesDataProgressLoader,
  UpdateWisnFormDesign,
  WisnPresetValueActionTypes
} from '../actions/wisn-preset-value.actions';
import {UpdateWisnPresetValue} from '../actions/wisn-preset-value.actions';
import * as fromUtilHelpers from '../../shared/helpers/util-functions';
import * as fromPrestValuesHelper from '../../shared/helpers/wisn-preset-value.helpers';
import { getCurrentGlobalFilter } from '../selectors';
import { getPageData } from '../selectors/wisn-poa-pages.selectors';
import { getAllWisnPresetValuesState, getCurrentFacilityType, getPresetFacilityTypes } from '../selectors/preset-values.selectors';
import { WisnPoaPageService } from 'src/app/shared/services';

@Injectable()
export class WisnPresetValuesEffects {

  facility: any;
  period: any;
  globalFilterOrgunit: any;
  dataSet: any;
  dataSetDesign: any;
  dataValues: any;
  singleDataPayload: any;
  param: any;
  elementId: any;
  facilitiesDataValues: any;
  sections: any;
  activePresetFacility: any;
  presetDataValuesToUse: any;
  constructor(
    private actions$: Actions,
    private wisnAnalyticsService: WisnAnalyticsService,
    private wisnPoaService: WisnPoaPageService,
    private store: Store<AppState>
  ) {}

  @Effect({dispatch: false})
  loadWisnPresetValues$ = this.actions$.pipe(
    ofType( WisnPresetValueActionTypes.LoadWisnPresetValues ),
    withLatestFrom(
      this.store.select(getCurrentGlobalFilter),
      this.store.select(getPageData('presetvalues'))),
    map(([action, currentGlobalFilters, wisnPresetValuePageInfo]: [fromWisnPresetValues.LoadWisnPresetValues, any, any]) => {
      const wisnPresetValuePage = wisnPresetValuePageInfo[0] ? wisnPresetValuePageInfo[0] : {};
      this.globalFilterOrgunit = currentGlobalFilters.ou ? currentGlobalFilters.ou : {};
      const presetValueOrgunt = wisnPresetValuePage ?
      wisnPresetValuePage.presetValueOrgunt ?
      wisnPresetValuePage.presetValueOrgunt : {id: '', name: ''} : {id: '', name: ''};
      this.period = currentGlobalFilters.pe ? currentGlobalFilters.pe : {};
      this.dataSet = wisnPresetValuePage.dataSource ? wisnPresetValuePage.dataSource[0] : {};
      const dataValues = wisnPresetValuePage.dataValues ?
      Object.assign({}, wisnPresetValuePage.dataValues) : {dispensary: {}, healthcenter: {}};
      const presetFacilityTypes = wisnPresetValuePage.presetFacilityTypes ? wisnPresetValuePage.presetFacilityTypes : [];
      this.store.dispatch(new AddWisnPresetValues({
        dataValues: fromPrestValuesHelper.reFormatPresetDataValues(dataValues, this.globalFilterOrgunit, this.period),
        presetFacilityTypes: presetFacilityTypes,
        presetValueOrgunt: presetValueOrgunt
      }));
    }),
    switchMap(() => this.wisnAnalyticsService.getDatasetSectionDataElements(this.dataSet)),
    tap((dataSetSection: any) => {
      const sections = dataSetSection.sections;

      // this.hideCadres();  // this function for auto-hide cadres based on the selected facility
      this.store.dispatch(new UpdateWisnPresetValue({
        progressLoader: 50,
        sections: sections
      }));
    }),
    switchMap(() => this.wisnAnalyticsService.getDataSetDesign(this.dataSet.id)),
    map((dataSetMetaData: any) => {
      const dataSetDesign = dataSetMetaData.dataEntryForms ? dataSetMetaData.dataEntryForms[0] ?
      dataSetMetaData.dataEntryForms[0].htmlCode : [] : '';
      this.store.dispatch(new UpdateWisnFormDesign({
        formDesign: dataSetDesign,
        progressLoader: 100 }));
    })
  );

  @Effect({dispatch: false})
  uploadingPresetValue$ = this.actions$.pipe(
    ofType(WisnPresetValueActionTypes.UploadWisnPresetValue),
    withLatestFrom(this.store.select(getAllWisnPresetValuesState),
    this.store.select(getCurrentFacilityType),
    this.store.select(getPresetFacilityTypes),
    this.store.select(getPageData('presetvalues'))),
    map(([action, presetDataValues, currentFacilityType, presetFacilityTypes, wisnPresetValuePageInfo]:
      [fromWisnPresetValues.UploadWisnPresetValue, any, any, any, any]) => {
        const wisnPresetValuePage = wisnPresetValuePageInfo[0] ? wisnPresetValuePageInfo[0] : {};
        const activePresetFacility = presetFacilityTypes.filter(facility => facility.id === currentFacilityType);
        this.activePresetFacility = activePresetFacility[0] ? activePresetFacility[0] : {};
        this.presetDataValuesToUse = presetDataValues[currentFacilityType];
        wisnPresetValuePage.dataValues = presetDataValues;
        const pagePayload = {'wisn-poa-pages': wisnPresetValuePage };
        this.wisnPoaService.updateDataStorePages(Object.keys(pagePayload), wisnPresetValuePage).subscribe(response => {
          console.log('[WISN+POA] Successful updated preset dataVales');
        });
    }),
    switchMap(() => this.wisnAnalyticsService.getAllOrganisationUnitsBasedOnType(this.activePresetFacility.orgunitGroupId)),
    map((orgunitsOfSameGroup) => {
      const orgunits = orgunitsOfSameGroup.organisationUnits ? orgunitsOfSameGroup.organisationUnits : [];
      const compressedDataValues = fromPrestValuesHelper.prepareDatavalueSetForUpload(this.presetDataValuesToUse.dataValues, orgunits);
      this.uploadPrestValues(compressedDataValues);
    })
  );

  uploadPrestValues(dataValuePayload) {
    this.store.dispatch(new PresetValuesDataProgressLoader(0));
      let progressNumerator = 0;
      const progressDenominator = dataValuePayload.length;
      this.wisnAnalyticsService.importBulkDatavalues(dataValuePayload).subscribe(response => {
        progressNumerator += 1;
        const progresPercent: any = ((progressNumerator / progressDenominator) * 100).toFixed(1);
        this.store.dispatch(new PresetValuesDataProgressLoader(progresPercent));
      });
  }

  // function for hiding cadres based on the selected facility type
}
