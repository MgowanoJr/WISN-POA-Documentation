import { Injectable } from '@angular/core';
import { Actions, Effect, ofType } from '@ngrx/effects';
import { Store } from '@ngrx/store';
import { AppState } from '../reducers';
import { WisnAnalyticsService } from '../../shared/services/wisn-analytics.service';
import {
  AddWisnHmisData, CompleteDownLaodWisnHmisData, NoChangesOnWisnHmisState, ResetHmiState, UpsertWisnHmisData,
  WisnHmisDataActionTypes, WisnHMISErrorHandler
} from '../actions/wisn-hmis-data.actions';
import {concatMap, exhaustMap, map, switchMap, tap, withLatestFrom} from 'rxjs/operators';
import * as fromWisnHMISactions from '../actions/wisn-hmis-data.actions';
import {LoadWisnReports, WisnReportActionTypes} from '../actions/wisn-report.actions';
import {LoadWisnHmisDatas} from '../actions';
import * as fromWisnHmisSelectors from '../selectors/wisn-hmis.selectors';
import {forkJoin, Observable} from 'rxjs';
import {mergeMap} from 'rxjs-compat/operator/mergeMap';
import 'rxjs/add/observable/from';
import {AddWisnPresetValues, UpsertWisnPresetValues} from '../actions/wisn-preset-value.actions';
import * as fromWisnPresetValuesSelectors from '../selectors/preset-values.selectors';
import {WisnPoaPageService} from '../../shared/services/wisn-poa-page.service';
import * as fromUtilHelpers from '../../shared/helpers/util-functions';
import { getCurrentGlobalFilter, getCurrentUser } from '../selectors';
import { getWisnHmisDataState } from '../selectors/wisn-hmis.selectors';
import { getPageData } from '../selectors/wisn-poa-pages.selectors';
import { getDataEntryLevelState } from '../selectors/general-configuration.selector';
import * as fromWisnHMIShelpers from '../../shared/helpers/wisn-hmis.helpers';

@Injectable()
export class WisnHmisDataEffects {
  orgunitCollections: string;
  analyticsDxCollection: string;
  period: any;
  dataSet: any;
  dataElements: any;
  selectedOrgunit: any;
  hmisPageData: any;
  stateEntityIds: any;
  hmisDataCollection: any;
  dataEntryLevel: any;
  currentUser: any;
  constructor(
    private actions$: Actions,
    private wisnAnalyticsService: WisnAnalyticsService,
    private wisnPoaService: WisnPoaPageService,
    private store: Store<AppState>
  ) {}

  @Effect()
  checkingWisnHMISState = this.actions$.pipe(
    ofType(WisnHmisDataActionTypes.CheckingWisnHmisState),
    withLatestFrom(this.store.select(getWisnHmisDataState),
      this.store.select(getCurrentGlobalFilter)),
    map(([action, wisnHmisData, currentGlobalFilters]:
           [fromWisnHMISactions.CheckingWisnHmisState, any, any]) => {
      const wisnHmisState = wisnHmisData ? wisnHmisData : {};
      const currentOrgunit = currentGlobalFilters.ou ? currentGlobalFilters.ou : {};
      const curentPeriod = currentGlobalFilters.pe ? currentGlobalFilters.pe : {};

      if (wisnHmisState.entities['hmisData'].length > 1 &&
        wisnHmisState.lastDataSelection.orgunit === currentOrgunit.id &&
        wisnHmisState.lastDataSelection.period === curentPeriod.id) {
        // do nothing because state info is already there
        return new NoChangesOnWisnHmisState();
      }
        // dispatch action for loading the wisnReport
        return new LoadWisnHmisDatas(); // fire action for loading HMIS data
    })
  );

  @Effect({dispatch: false})
  loadHMISdata$ = this.actions$.pipe(
    ofType(WisnHmisDataActionTypes.LoadWisnHmisDatas),
    withLatestFrom(this.store.select(getPageData('hmis')),
      this.store.select(getCurrentGlobalFilter),
      this.store.select(getDataEntryLevelState),
      this.store.select(getCurrentUser)),
    map(([action, wisnHMISPageData, currentGlobalFilters, dataEntryOrgUnitLevel, currentUserData]:
           [fromWisnHMISactions.LoadWisnHmisDatas, any, any, any, any]) => {
      this.selectedOrgunit = currentGlobalFilters.ou ? currentGlobalFilters.ou : {};
      this.period = currentGlobalFilters.pe ? currentGlobalFilters.pe : {};
      const hmisPageData = wisnHMISPageData ? wisnHMISPageData[0] ? wisnHMISPageData[0] : {} : {};
      this.dataSet = hmisPageData.dataSource ? hmisPageData.dataSource[0] ? hmisPageData.dataSource[0] : {} : {};
      this.dataEntryLevel = dataEntryOrgUnitLevel ? dataEntryOrgUnitLevel : 4;
      this.currentUser = currentUserData ? currentUserData : {};
    }),
    switchMap(() =>
      this.wisnAnalyticsService.getDatasetSectionDataElements(this.dataSet)
    ),
    map((dataSetSection: any) => {
        const sections = dataSetSection.sections;
        sections.forEach((section: any) => {
          if (section.id === this.dataSet.sections[0]) {
            this.dataElements = section.dataElements;
          }
        });
        this.analyticsDxCollection = fromWisnHMIShelpers.getAnalticsDe(this.dataElements ? this.dataElements : []);
        this.stateEntityIds = fromUtilHelpers.generateStateUids(this.selectedOrgunit);
        const lastDataSelection = {
          orgunit: this.selectedOrgunit.id,
          period: this.period.id
        };

        this.store.dispatch(new AddWisnHmisData({
          lastDataSelection: lastDataSelection,
          hmisDataElements: [{shortName: 'Facility'}, ...this.dataElements],
          wisnHmisData: [],
          stateEntityIds: this.stateEntityIds
        }));
      }
    ),
    tap(() => {
      this.store.dispatch(new ResetHmiState());
      if ((this.dataEntryLevel - this.selectedOrgunit.level) > 1) { // this to aggrigate the dataValues from Analytics
        this.getHMISanalyticsData(this.selectedOrgunit, this.period, this.dataSet,
          this.analyticsDxCollection, this.currentUser);
      } else { // esle pic data values from individual facility level
        let progressNumerator = 0;
        const progressDenominator = this.selectedOrgunit.children ? this.selectedOrgunit.children.length : 1;
        this.wisnAnalyticsService.getMultipleOrgunitDatavalues(
          this.dataSet.id, this.period.id, this.selectedOrgunit.children ? this.selectedOrgunit.children : [this.selectedOrgunit]
        ).subscribe(singleOrgunitDataValue => {
          const hmisDataCollection = fromWisnHMIShelpers.processHMISdataValues(singleOrgunitDataValue, this.dataElements,
            this.currentUser.userCredentials ? this.currentUser.userCredentials : {}, this.selectedOrgunit, this.period, this.dataSet);
          progressNumerator += 1;
          const progresPercent = ((progressNumerator / progressDenominator) * 100).toFixed(1) ;
          this.store.dispatch(new UpsertWisnHmisData({data: hmisDataCollection, progressCount: progresPercent}));
        }, error1 => {});
      }
    }
    )
  );

  @Effect({dispatch: false})
  updatingSingHMISdata$ = this.actions$.pipe(
    ofType(WisnHmisDataActionTypes.UpdateWisnHmisData),
    withLatestFrom( this.store.select(
      fromWisnHmisSelectors.getWisnHmisDataEntities
    )),
    tap(([action, hmisState]:
           [fromWisnHMISactions.UpdateWisnHmisData, any]) => {
      const objectData = {  };
      objectData[action.payload.changedData.dataElementId] = action.payload.changedData.value;
      const dataPayload = { ...action.payload.rowData, ...objectData}; // re-create the row payload
      hmisState.hmisData[action.payload.index] = dataPayload;
    })
  );

  @Effect()
  downloadHMISdata$ = this.actions$.pipe(
    ofType(WisnHmisDataActionTypes.DownLaodWisnHmisData),
    withLatestFrom(this.store.select(
      fromWisnHmisSelectors.getWisnHmisDataElements
    ), this.store.select(
      fromWisnHmisSelectors.getWisnHmisDataEntities
    )),
    map(([action, hmisDataElements, hmisDataCollected]:
           [fromWisnHMISactions.DownLaodWisnHmisData, any, any]) => {
      const hmisDataCollection = hmisDataCollected.hmisData;
      const tableHeader = [];
      hmisDataElements.forEach((dx: any) => {
        tableHeader.push(dx.shortName);
      });
      tableHeader[0] = 'Facility';
      const tablerows = fromWisnHMIShelpers.generateCSVrows(hmisDataCollection, hmisDataElements);
      const row = [tableHeader, ...tablerows];
      let csvContent = 'data:text/csv;charset=utf-8,';
      row.forEach(function(rowArray) {
        const rowEntry = rowArray.join(',');
        csvContent += rowEntry + '\r\n';
      });
      const encodedUri = encodeURI(csvContent);
      const link = document.createElement('a');
      link.setAttribute('href', encodedUri);
      link.setAttribute('download',  'HMIS Data.csv');
      link.click();

      return new CompleteDownLaodWisnHmisData();
    }),
  );

  @Effect()
  uploadingHmisData$ = this.actions$.pipe(
    ofType(WisnHmisDataActionTypes.UploadWisnHmisDatas),
    withLatestFrom(this.store.select(
      fromWisnHmisSelectors.getWisnHmisDataElements
    ), this.store.select(
      fromWisnHmisSelectors.getWisnHmisDataEntities
    )),
    map(([action, hmisDataElements, hmisDateBeforeUpload]:
           [fromWisnHMISactions.UploadWisnHmisDatas, any, any]) => {
      const csvData = action.payload;
      const processedData = fromWisnHMIShelpers.processWisnHMISdataforUpload(csvData,
        hmisDataElements ? hmisDataElements : [], this.selectedOrgunit,
        this.currentUser.userCredentials ? this.currentUser.userCredentials : {}, this.period, this.dataSet);
      const changedHMISdata = fromWisnHMIShelpers
      .getChangedHMISData(processedData.hmisDataToUpload, hmisDateBeforeUpload.hmisData, this.period);
      const lastDataSelection = {
        orgunit: this.selectedOrgunit.id,
        period: this.period.id
      };
      // this.updateDataValues(facilitiesDataValues, processedData.uploadedRestructured);
      this.uploadHMISdataValuesFromCSV(changedHMISdata);
      this.store.dispatch(new ResetHmiState()); // this to reset hmisData
      return new AddWisnHmisData({
        lastDataSelection: lastDataSelection,
        hmisDataElements: [{ shortName: 'Facility' }, ...this.dataElements],
        wisnHmisData: processedData.uploadedRestructured,
        stateEntityIds: fromUtilHelpers.generateStateUids(this.selectedOrgunit)
      });
    })
  );

  getHMISanalyticsData(selectedOrgunit, period, dataSet, analyticsDxCollection, currentUser) {
    let progressNumerator = 0;
    const progressDenominator = selectedOrgunit.children.length;
      this.wisnAnalyticsService.getWISNHMISdataNoaggregations(
        selectedOrgunit.children,
        period.id,
        analyticsDxCollection
      ).subscribe((hmisDataValues: any) => {
        const hmisData = hmisDataValues.rows;
        let hmisDataCollection = [];
          const analyticsValueIndex = hmisDataValues.headers.findIndex(head => head.name === 'value');
          const analyticsDxIndex = hmisDataValues.headers.findIndex(head => head.name === 'dx');
          const analyticsOUIndex = hmisDataValues.headers.findIndex(head => head.name === 'ou');
        if (selectedOrgunit.children && hmisData.length > 0) {
          hmisDataCollection = fromWisnHMIShelpers.generateHMISdataCollected(selectedOrgunit, hmisData, analyticsOUIndex, dataSet, period);
            } else {
              hmisDataCollection = fromWisnHMIShelpers.generateHMISdataCollectedByAnalytics(selectedOrgunit,
                hmisDataValues, dataSet, period);
            }

            if (hmisData.length > 0) {
              hmisDataCollection = fromWisnHMIShelpers.regenerateHMISdataCollection(hmisData, hmisDataCollection,
                currentUser.userCredentials ? currentUser.userCredentials : {}, analyticsOUIndex, analyticsDxIndex, analyticsValueIndex);
            } else {
              this.dataElements.forEach((hmisDx: any) => {
                hmisDataCollection.forEach((hmisDataObject: any) => {
                  hmisDataObject[hmisDx.id] = {value: 0,
                    storedBy: currentUser.userCredentials.username ?
                    currentUser.userCredentials.username : ''};
                });
              });
            }
        const lastDataSelection = {
          orgunit: selectedOrgunit.id,
          period: period.id
        };
          progressNumerator += 1;
          const progresPercent = ((progressNumerator / progressDenominator) * 100).toFixed(1);
        this.hmisDataCollection = hmisDataCollection;
        this.store.dispatch(new AddWisnHmisData({
            lastDataSelection: lastDataSelection,
            hmisDataElements: [{shortName: 'Facility'}, ...this.dataElements],
            wisnHmisData: hmisDataCollection,
            progressLoader: progresPercent,
            stateEntityIds: fromUtilHelpers.generateStateUids(selectedOrgunit)
          })
        );
      },
          error => this.store.dispatch(new WisnHMISErrorHandler(error))
          );
  }

  uploadHMISdataValuesFromCSV(changedHMISdata) {
    const importData = changedHMISdata.hmisDataToImport;
    const inputBoxChanges = changedHMISdata.inputBoxChanges;

    this.wisnAnalyticsService.uploadWisnDataFromCSV([importData]).subscribe(response => {
      (inputBoxChanges || []).forEach(inputBox => {
        try {
              document.getElementById(inputBox).style.backgroundColor = '#b9ffb9';
              document.getElementById(inputBox).style.borderColor = '#dc3545';
            } catch (e) {}
      });
    });
  }
}
