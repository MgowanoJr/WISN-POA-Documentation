import { Injectable } from '@angular/core';
import { Store } from '@ngrx/store';
import { AppState } from '../reducers/index';
import { Router } from '@angular/router';
import { Actions, Effect, ofType } from '@ngrx/effects';
import * as _ from 'lodash';

import {
  AddGlobalFilterAction,
  GlobalFilterActionTypes,
  GlobalFilterUpdateAction,
  UpdateGlobalFilterAction
} from '../actions/global-filter.actions';
import {
  filter,
  map,
  tap,
  withLatestFrom,
  switchMap,
  first,
  take
} from 'rxjs/operators';
import { PeriodService } from '../../shared/modules/period-filter/period.service';
import { ROUTER_NAVIGATION } from '@ngrx/router-store';
import { getCurrentUser } from '../selectors/current-user.selectors';
import { OrgUnitService } from '../../shared/modules/org-unit-filter/org-unit.service';

import * as fromGlobalFilterSelectors from '../selectors/global-filter.selectors';
import {
  getCurrentPage,
  getWisnPoaPages
} from '../selectors/wisn-poa-pages.selectors';
import { getCurrentPageState } from '../selectors';
import { Observable } from 'rxjs';
import { Go } from '../actions';
@Injectable()
export class GlobalFilterEffects {
  constructor(
    private store: Store<AppState>,
    private router: Router,
    private actions$: Actions,
    private periodService: PeriodService,
    private orgUnitService: OrgUnitService
  ) {}

  @Effect({ dispatch: false })
  routerNavigation$ = this.actions$.pipe(
    ofType(ROUTER_NAVIGATION),
    withLatestFrom(
      this.store.select(fromGlobalFilterSelectors.getCurrentGlobalFilter)
    ),
    tap(([action, globalFilter]: [any, any]) => {
      this.store
        .select(getCurrentUser)
        .pipe(filter((userInfo: any) => userInfo))
        .subscribe(currentUser => {
          // Get route query parameters
          const queryParams = action.payload.routerState.queryParams;
          const currentPage = action.payload.routerState.url;

          // Find currenct globalFilter based on current page

          // Only set this when there is no global filter to avoid multiple settings of global filter
          if (!globalFilter) {
            // Get orgUnit ID either from route parameters or from user information
            const orgUnitId =
              queryParams && queryParams.orgUnit
                ? queryParams.orgUnit
                : currentUser &&
                  currentUser.organisationUnits &&
                  currentUser.organisationUnits[0]
                ? currentUser.organisationUnits[0].id
                : '';

            // Get period ID either from route parameters or from current date information

            // TODO MAKE PERIOD SELECTION GENERIC TO ACCOMODATE CHANGING PERIOD TYPE, CURRENTLY ONLY YEARLY PERIOD TYPE IS SUPPORTED
            const currentDate = new Date();
            const currentYear = currentDate.getFullYear() - 1;
            const periodId =
              queryParams && queryParams.period
                ? queryParams.period
                : currentYear.toString();

            const periodObject = _.find(
              this.periodService.getPeriodsBasedOnType('Yearly', currentYear),
              ['id', periodId]
            ) || { id: currentYear.toString(), name: currentYear.toString() };

            let orgUnitAncestors;
            this.orgUnitService
              .getCurrentOrgUnitAncestors(orgUnitId)
              .subscribe(
                data => (orgUnitAncestors = data ? data.ancestors : [])
              );

            this.orgUnitService
              .getOrgunitById(orgUnitId)
              .subscribe((orgUnitResult: any) => {
                const orgUnit =
                  orgUnitResult || currentUser.organisationUnits[0];
                this.store.dispatch(
                  new AddGlobalFilterAction(
                    {
                      id: 'wisn_poa_page',
                      pe: {
                        name: periodObject.name,
                        id: periodObject.id,
                        type: 'Yearly',
                        year: currentYear
                      },
                      ou: {
                        name: orgUnit.name,
                        id: orgUnit.id,
                        children: orgUnit.children,
                        ancestors: orgUnitAncestors,
                        level: orgUnit.level
                      },
                      visitedPages: []
                    },
                    currentPage
                  )
                );
              });
          }
        });
    })
  );

  @Effect()
  globalFilterUpdate$ = this.actions$.pipe(
    ofType(GlobalFilterActionTypes.GLOBAL_FILTER_UPDATE),
    withLatestFrom(this.store.select(getCurrentPageState)),
    map(([action, currentPage]: [GlobalFilterUpdateAction, string]) => {
      return new UpdateGlobalFilterAction(
        'wisn_poa_page',
        {
          [action.globalFilter.name]: action.globalFilter
            ? action.globalFilter.globalItem
              ? action.globalFilter.globalItem
              : action.globalFilter.items
              ? action.globalFilter.items[0]
              : null
            : null
        },
        currentPage
      );
    })
  );

  @Effect({ dispatch: false })
  updateGlobalFilter$: Observable<any> = this.actions$.pipe(
    ofType(GlobalFilterActionTypes.UPDATE_GLOBAL_FILTER),
    withLatestFrom(
      this.store.select(fromGlobalFilterSelectors.getCurrentGlobalFilter)
    ),
    tap(([action, currentGlobalFilter]: [UpdateGlobalFilterAction, any]) => {
      if (
        currentGlobalFilter &&
        currentGlobalFilter.ou &&
        currentGlobalFilter.pe
      ) {
        this.store.dispatch(
          new Go({
            path: [action.route],
            query: {
              orgUnit: currentGlobalFilter.ou.id,
              period: currentGlobalFilter.pe.id
            }
          })
        );
      }
    })
  );
}
