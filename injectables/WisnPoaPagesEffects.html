import { Injectable } from '@angular/core';
import { Actions, Effect, ofType } from '@ngrx/effects';
import * as _ from 'lodash';
import { WisnPoaPageService } from '../../shared/services/wisn-poa-page.service';
import {
  AddWisnPoaPages,
  FailLoadWisnPoaPages,
  WisnPoaPagesActionTypes
} from '../actions/wisn-poa-pages.actions';
import {
  catchError,
  map,
  switchMap,
  tap,
  first,
  withLatestFrom
} from 'rxjs/operators';
import { of } from 'rxjs/observable/of';
import { Observable } from 'rxjs';
import { ROUTER_NAVIGATION, RouterNavigationAction } from '@ngrx/router-store';
import { AppState } from '../reducers';
import { Store } from '@ngrx/store';
import { getCurrentGlobalFilter } from '../selectors';
import { getWisnPoaPages } from '../selectors/wisn-poa-pages.selectors';
import {
  GlobalFilterActionTypes,
  Go,
  LoadWisnHmisDatas,
  LoadWisnReports,
  UpdateGlobalFilterAction
} from '../actions';
import {
  CheckingWisnHRHState,
  LoadWisnHrhDatas
} from '../actions/wisn-hrh-data.actions';
import {
  LoadWisnPresetValues,
  UpsertWisnPresetValue
} from '../actions/wisn-preset-value.actions';
import {
  CheckingPoaPriState,
  LoadPoaPriLists
} from '../actions/poa-pri-list.actions';
import { getWisnReportData } from '../selectors/wisn-report.selectors';
import * as fromRouterActions from '../actions/router.actions';
import {
  CheckingPoaCostAssumptionsState,
  LoadPoaCostAssumptions
} from '../actions/poa-cost-assumption.actions';
import {
  CheckingPoaDecidedSurplusState,
  LoadPoaDecidedSurpluss
} from '../actions/poa-decided-surplus.actions';
import * as fromLocalStorageService from '../../shared/services/local-storage.service';

import { CheckWisnReportState } from '../actions/wisn-report.actions';
import * as fromGeneralConfigurationSelectors from '../selectors/general-configuration.selector';
import { CheckingWisnHmisState } from '../actions/wisn-hmis-data.actions';

@Injectable()
export class WisnPoaPagesEffects {
  pageLoadInitiated: boolean;
  constructor(
    private actions$: Actions,
    private store: Store<AppState>,
    private pagesService: WisnPoaPageService
  ) {}

  @Effect()
  loadPages$ = this.actions$.pipe(
    ofType(WisnPoaPagesActionTypes.LoadWisnPoaPages),
    switchMap(() => this.pagesService.loadWisnPoaPages()),
    map((loadedPage: any) => {
      _.map(loadedPage, wisnPoaPage => {
        this.pagesService
          .addDataToLocalDatabase(
            fromLocalStorageService.WISN_POA_PAGES_KEY,
            wisnPoaPage
          )
          .subscribe();
      });
      return new AddWisnPoaPages(loadedPage);
    }),
    catchError(error => of(new FailLoadWisnPoaPages(error)))
  );

  @Effect({ dispatch: false })
  routerNavigationOrUpdateGlobalFilter$: Observable<any> = this.actions$.pipe(
    ofType(ROUTER_NAVIGATION),
    tap((action: any) => {
      const currentRoute =
        action.route || action.payload.routerState.url.split('?')[0];
      const queryParams = action.payload.routerState.queryParams;
      this.store
        .select(getCurrentGlobalFilter)
        .pipe(
          first((globalFilter: any) => globalFilter),
          switchMap((globalFilter: any) =>
            this.store.select(getWisnPoaPages).pipe(
              first((pages: any[]) => pages.length > 0),
              map((pages: any) => {
                return { globalFilter, pages };
              })
            )
          )
        )
        .subscribe((pageDetails: any) => {
          const splitedRoute = currentRoute.split('/');

          const currentPageId = splitedRoute.length >= 3 ? splitedRoute[2] : '';
          const currentSubPageId =
            splitedRoute.length > 3 ? splitedRoute[3] : '';

          const pageObject = _.find(pageDetails.pages, ['id', currentPageId]);

          const currentPage =
            currentSubPageId !== '' && pageObject
              ? _.find(pageObject.subPages, ['id', currentSubPageId]) ||
                pageObject
              : pageObject;

          if (currentPage) {
            // console.log('checking current page :' + JSON.stringify(currentPage))
            // THIS IS WHERE YOU CAN START USING DATA SOURCE AND GLOBAL FILTER TO POPULATE CURRENT PAGE
            if (currentPage.id === 'wisnreport') {
              if (currentPage['subPages'] && currentSubPageId === '') {
                // Preseting first subpage
                const initCadreId = _.toLower(currentPage['subPages'][0].id);
                this.store.dispatch(
                  new fromRouterActions.Go({
                    path: [splitedRoute[1], splitedRoute[2], 'all'],
                    query: {
                      orgUnit: pageDetails.globalFilter.ou.id,
                      period: pageDetails.globalFilter.pe.id
                    }
                  })
                );
              }

              this.store.dispatch(new CheckWisnReportState());
            } else if (currentPage.id === 'hmis') {
              this.store.dispatch(new CheckingWisnHmisState());
            } else if (currentPage.id === 'hrh') {
              this.store.dispatch(new CheckingWisnHRHState()); // fire action for loading wisn hrh data
            } else if (currentPage.id === 'presetvalues') {
              this.store
                .select(
                  fromGeneralConfigurationSelectors.getPresetValueEntryLevel
                )
                .subscribe(wisnPresetEntryLevel => {
                  this.store.dispatch(
                    new UpsertWisnPresetValue({
                      id: wisnPresetEntryLevel
                    })
                  );
                  this.store.dispatch(new LoadWisnPresetValues());

                  // this is hardcoded temporarly until proper approach is discussed better approach.
                  // fire action for loading wisn preset values dataValues
                });
            } else if (currentPage.id === 'prireport') {
              this.store.dispatch(new CheckingPoaPriState());
            } else if (currentPage.id === 'standardpackage') {
              this.store.dispatch(new CheckingPoaPriState());
            } else if (currentPage.id === 'costassumptions') {
              this.store.dispatch(new CheckingPoaCostAssumptionsState()); // fire action for loading poa cost assumptions
            } else if (currentPage.id === 'decidehrhsurplus') {
              this.store.dispatch(new CheckingPoaDecidedSurplusState()); // fire action for loading poa cost assumptions
            } else if (
              currentPage.id === 'distributehrh' ||
              currentPage.id === 'redistributehrh'
            ) {
              // implemented this temporalry for now
              this.store.dispatch(new CheckingPoaPriState());
            }
          }
        });
    })
  );
}
