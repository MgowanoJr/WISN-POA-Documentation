import { Injectable } from "@angular/core";
import * as _ from "lodash";
declare var unescape: any;
@Injectable({ providedIn: "root" })
export class VisualizationExportService {
  exportXLS(filename: string, htmlTable: any) {
    if (this._getMsieVersion() || this._isFirefox()) {
      console.error("Not supported browser");
    }
    // Other Browser can download xls
    if (htmlTable) {
      const uri = "data:application/vnd.ms-excel;base64,",
        template =
          '<html xmlns:o="urn:schemas-microsoft-com:office:office" xmlns:x="urn:schemas-microsoft-com:' +
          'office:excel" xmlns="http://www.w3.org/TR/REC-html40"><head><!--[if gte mso 9]><xml><x:ExcelWorkbook>' +
          "<x:ExcelWorksheets><x:ExcelWorksheet><x:Name>{worksheet}</x:Name><x:WorksheetOptions><x:DisplayGridlines/>" +
          "</x:WorksheetOptions></x:ExcelWorksheet></x:ExcelWorksheets></x:ExcelWorkbook></xml><![endif]-->" +
          '</head><body><table border="1">{table}</table><br /><table border="1">{table}</table></body></html>',
        base64 = s => window.btoa(unescape(encodeURIComponent(s))),
        format = (s, c) => s.replace(/{(\w+)}/g, (m, p) => c[p]);

      const ctx = { worksheet: "Sheet 1", filename: filename };
      let str =
        '<html xmlns:o="urn:schemas-microsoft-com:office:office" xmlns:x="urn:schemas-microsoft-com:office' +
        ':excel" xmlns="http://www.w3.org/TR/REC-html40"><head><!--[if gte mso 9]><xml><x:ExcelWorkbook>' +
        "<x:ExcelWorksheets><x:ExcelWorksheet><x:Name>{worksheet}</x:Name><x:WorksheetOptions><x:DisplayGridlines/>" +
        "</x:WorksheetOptions></x:ExcelWorksheet></x:ExcelWorksheets></x:ExcelWorkbook></xml><![endif]--></head><body>";

      const matchedTableContent = htmlTable.match(
        /<table[^>]*>([\w|\W]*)<\/table>/im
      );
      ctx["table1"] =
        matchedTableContent && matchedTableContent.length > 1
          ? matchedTableContent[1]
          : "";

      str +=
        '<b>{filename}</b><br/><table border="1">{table1}</table></body></html>';

      setTimeout(() => {
        const link = document.createElement("a");
        link.download = filename + ".xls";
        link.href = uri + base64(format(str, ctx));
        link.click();
      }, 100);
    }
  }

  generateTableOne(tableData) {
    const shortageSummaryRows = [];

    const shortageSummaryHeader = [
      "Cadre",
      "Current Staff",
      "Needed Staff",
      "Shortage",
      "Shortage Percentage"
    ];
    _.each(tableData, tableRow => {
      const rowValues = [
        tableRow.cadreName,
        tableRow.currentStaff,
        tableRow.neededStaff,
        tableRow.shortage,
        tableRow.shortagePercentage + "%"
      ];
      shortageSummaryRows.push(rowValues);
    });

    shortageSummaryRows.push([
      "Total",
      _.sumBy(shortageSummaryRows, "1"),
      _.sumBy(shortageSummaryRows, "2"),
      _.sumBy(shortageSummaryRows, "2") - _.sumBy(shortageSummaryRows, "1"),
      _.round(
        ((_.sumBy(shortageSummaryRows, "2") -
          _.sumBy(shortageSummaryRows, "1")) *
          100) /
          _.sumBy(shortageSummaryRows, "2"),
        2
      ) + "%"
    ]);
    const shortageSummaryCSVStructure = [
      shortageSummaryHeader,
      ...shortageSummaryRows
    ];

    return shortageSummaryCSVStructure;
  }

  generateTableTwo(tableData, summaryData?) {
    const priData = [];
    if (summaryData) {
      console.log(summaryData);
      const priDataHeader = [
        "Order",
        "PRI",
        "Cadre Type",
        "Facility",
        "Assignment Designation",
        "Total Salary Cost",
        "Allowance Cost"
      ];
      _.each(tableData, (tableRow, index) => {
        const rowValues = [
          index + 1,
          tableRow.pri,
          tableRow.cadre,
          tableRow.facilityName,
          tableRow.assinmentDesignation,
          tableRow.newAssignmentCost,
          tableRow.redistributionCost
        ];
        priData.push(rowValues);
      });
      const priDataStructure = [priDataHeader, ...priData];
      return priDataStructure;
    } else if (!summaryData) {
      const priDataHeader = [
        "SN",
        "PRI",
        "Cadre Type",
        "Organization Unit",
        "Organization Unit Hierarchy",
        "Assignment Designation",
        "Total Salary Cost",
        "Allowance Cost",
        "Redistributed from"
      ];
      _.each(tableData, (tableRow, index) => {
        const rowValues = [
          index + 1,
          tableRow.pri,
          tableRow.cadre,
          tableRow.facilityName,
          tableRow.districtName,
          tableRow.assinmentDesignation,
          tableRow.newAssignmentCost,
          tableRow.redistributionCost,
          tableRow.redistributedFrom
            ? tableRow.redistributedFrom.orgunitName
            : ""
        ];
        priData.push(rowValues);
      });
      const priDataStructure = [priDataHeader, ...priData];
      return priDataStructure;
    }
  }
  exportPaginatedCSV(priData, filename?, surplusData?) {
    const surplusDataCSVStructure = this.generateTableOne(surplusData);
    const priDataStructure = surplusData
      ? _.chunk(this.generateTableTwo(priData, surplusData), 30000)
      : _.chunk(this.generateTableTwo(priData, surplusData), 25000);
    const mergedFileURIs = [];
    if (surplusData) {
      let surplusDataCSVContent =
        "data:application/vnd.openxmlformats-officedocument.spreadsheetml.sheet;charset=utf-8,";

      surplusDataCSVStructure.forEach(function(rowArray) {
        const rowEntry = rowArray.join(",");
        surplusDataCSVContent += rowEntry + "\r\n";
      });
      const surplusDataEncodedUri = encodeURI(surplusDataCSVContent);
      mergedFileURIs.push(surplusDataEncodedUri);
    }
    _.map(priDataStructure, priDataStructureItem => {
      let priDataCSVContent =
        "data:application/vnd.openxmlformats-officedocument.spreadsheetml.sheet;charset=utf-8,";

      priDataStructureItem.forEach(function(rowArray) {
        const rowEntry = rowArray.join(",");
        priDataCSVContent += rowEntry + "\r\n";
      });
      const poaDataEncodedUri = encodeURI(priDataCSVContent);
      mergedFileURIs.push(poaDataEncodedUri);
    });

    const link = document.createElement("a");
    _.map(mergedFileURIs, fileURI => {
      link.setAttribute("href", fileURI);
      link.setAttribute("download", filename);
      link.click();
    });
  }
  exportCSV(filename: string, htmlTable: any, csv?: any) {
    // Generate our CSV string from out HTML Table
    const csvString = csv ? csv : this._tableToCSV(htmlTable);
    // Create a CSV Blob
    const blob = new Blob([csvString], { type: "text/csv" });

    // Determine which approach to take for the download
    if (navigator.msSaveOrOpenBlob) {
      // Works for Internet Explorer and Microsoft Edge
      navigator.msSaveOrOpenBlob(blob, filename + ".csv");
    } else {
      this._downloadAnchor(URL.createObjectURL(blob), "csv", filename);
    }
  }
  private _getMsieVersion() {
    const userAgent = window.navigator.userAgent;

    const msie = userAgent.indexOf("MSIE ");

    if (msie > 0) {
      // IE 10 or older => return version number
      return parseInt(
        userAgent.substring(msie + 5, userAgent.indexOf(".", msie)),
        10
      );
    }

    const trident = userAgent.indexOf("Trident/");
    if (trident > 0) {
      // IE 11 => return version number
      const rv = userAgent.indexOf("rv:");
      return parseInt(
        userAgent.substring(rv + 3, userAgent.indexOf(".", rv)),
        10
      );
    }

    const edge = userAgent.indexOf("Edge/");
    if (edge > 0) {
      // Edge (IE 12+) => return version number
      return parseInt(
        userAgent.substring(edge + 5, userAgent.indexOf(".", edge)),
        10
      );
    }

    // other browser
    return false;
  }

  private _isFirefox() {
    if (navigator.userAgent.indexOf("Firefox") > 0) {
      return 1;
    }

    return 0;
  }

  private _downloadAnchor(content, ext, filename) {
    const anchor = document.createElement("a");
    anchor.style.display = "!important";
    anchor.id = "downloadanchor";
    document.body.appendChild(anchor);

    // If the [download] attribute is supported, try to use it

    if ("download" in anchor) {
      anchor.download = filename + "." + ext;
    }
    anchor.href = content;
    anchor.click();
    anchor.remove();
  }

  private _tableToCSV(table) {
    // We'll be co-opting `slice` to create arrays
    const slice = Array.prototype.slice;

    return slice
      .call(table.rows)
      .map(function(row) {
        console.log(row);
        return slice
          .call(row.cells)
          .map(function(cell) {
            return '"t"'.replace("t", cell.textContent);
          })
          .join(",");
      })
      .join("\r\n");
  }
}
