import { Injectable } from '@angular/core';
import {NgxDhis2HttpClientService} from '@hisptz/ngx-dhis2-http-client';
import {from, interval, Observable, of, throwError} from 'rxjs';
import {catchError, flatMap, mergeMap, retryWhen, share} from 'rxjs/operators';

@Injectable({
  providedIn: 'root'
})
export class PoaAnalyticsService {

  constructor(private httpClient: NgxDhis2HttpClientService) { }

  loadCostAssumptionsDataElements(program): Observable<any> {
    const sectionUrl = 'programs/' + program.id + '/programStages.json?' +
      'fields=programStageSections[id,name,dataElements[id,name,displayFormName]]';
    return this.httpClient.get(sectionUrl).pipe();
  }

  getLastCostAssumptionModificationDate(): Observable<any> {
    const url = 'dataStore/wisn-poa-data-memory/cost-assumptions.json';
    return this.httpClient.get(url).pipe();
  }

  getCostAssumptionsAnalyticsData(programId, orgunits, dataElements, period): Observable<any> {
    const url = 'analytics/events/query/' + programId + '.json?' +
      'dimension=pe:' + period + '&dimension=ou:' + orgunits + '&' + dataElements + '&' +
      'stage=PljrUu0N2wc&displayProperty=NAME&outputType=EVENT&pageSize=200&page=1';
    return this.httpClient.get(url).pipe();
  }

  uploadPRIstateDraft(orgunitId, payload): Observable<any> {
    const url = `dataStore/poa-pri-state/${orgunitId}`;
    return this.httpClient.put(url, payload).pipe();
  }

  fetchPRIstateDraft(orgunitId): Observable<any> {
    const orgunitIds = [orgunitId];
    let url = '';
    return from(orgunitIds).pipe(
      mergeMap((orgunit, index) => {
        url = `dataStore/poa-pri-state/${orgunit}`;
        return <Observable<any>> this.httpClient.get(url)
          .pipe(retryWhen(() => {
            return interval(5000).pipe(
              flatMap(count => count === 3 ?  throwError('Giving up') : of(count))
            );
          }));
      }, 10)
    );
  }

  makeLocalUid() {
    let text = '';
    const possible =
      'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
    for (let i = 0; i < 11; i++) {
      text += possible.charAt(Math.floor(Math.random() * possible.length));
    }
    return text;
  }
}
