import { Injectable } from '@angular/core';
import { NgxDhis2HttpClientService } from '@hisptz/ngx-dhis2-http-client';

import { WisnPoaPages } from '../../store/models/wisn-poa-pages.model';
import { Observable } from 'rxjs/Observable';
import {
  catchError,
  flatMap,
  map,
  mergeMap,
  retryWhen,
  switchMap,
  share
} from 'rxjs/operators';
import * as _ from 'lodash';
import { throwError, forkJoin, of, from, interval } from 'rxjs';
import { pagesReducer } from '../../store/reducers/wisn-poa-pages.reducer';

import {
  LocalStorageService,
  WISN_POA_PAGES_KEY
} from './local-storage.service';
import { AppState } from 'src/app/store';
import { Store } from '@ngrx/store';
import { currentUserRolePermissions } from 'src/app/store/selectors/general-configuration.selector';

@Injectable({
  providedIn: 'root'
})
export class WisnPoaPageService {
  dataStoreUrl: string;
  wisnPoaPagesConfigurationUrl: string;

  constructor(
    private httpClient: NgxDhis2HttpClientService,
    private localDbService: LocalStorageService,
    private store: Store<AppState>
  ) {
    this.dataStoreUrl = 'dataStore/wisn-poa-pages';
    this.wisnPoaPagesConfigurationUrl = 'wisn-poa-pages-configurations.json';
  }

  getUserPermissionAction(componentId: string) {
    let userRoleAction = '';
   const roleService =  this.store.select(currentUserRolePermissions).subscribe((userRole: any) => {
        if (userRole.permissions) {
          userRole.permissions.forEach((permissionAction: any) => {
            if (permissionAction.id === componentId) {
              userRoleAction = permissionAction.action;
            }
          });
        }
      });
      roleService.unsubscribe();
      return userRoleAction;
}

  checkWisnPoaPages(wisnPoaNameSpace): Observable<any> {
    const url = `dataStore/${wisnPoaNameSpace}`;
    return this.httpClient.get(url).pipe();
  }

  loadWisnPoaPages(): Observable<WisnPoaPages[]> {
    return this.httpClient.get(this.dataStoreUrl).pipe(
      switchMap((pageIds: string[]) =>
        forkJoin(
          _.map(pageIds, (pageId: string) =>
            this.httpClient.get(`${this.dataStoreUrl}/${pageId}`)
          )
        )
      ),
      catchError((error: any) => {
        if (error.status !== 404) {
          // checks if the error is page not foound then execute a procedure for installing the pages
          return throwError(error);
        } else {
          // procedure for installing wisn-poa-pages
          this.httpClient
            .get(this.wisnPoaPagesConfigurationUrl)
            .pipe(
              switchMap((pages: any) =>
                forkJoin(
                  _.map(pages.wisnPages, (page: any) =>
                    this.httpClient.post(
                      `${this.dataStoreUrl}/${page.id}`,
                      page
                    )
                  )
                )
              )
            );
        }
      })
    );
  }

  loadDataStorePages(pages: any[], pagesData: any): Observable<any> {
    return from(pages).pipe(
      mergeMap((pageId, index) =>
        forkJoin(
          _.map(pagesData[pageId], (keyItem: any) =>
            this.httpClient

  getWisnPoaPages(appPageId: string) {
    return this.httpClient.get(this.dataStoreUrl).pipe(
      switchMap((pageIds: string[]) =>
        forkJoin(
          _.map(pageIds, (pageId: string) =>
            this.httpClient.get(`${this.dataStoreUrl}/${pageId}`)
          )
        )
      ),
      catchError((error: any) => {
        if (error.status !== 404) {
          return throwError(error);
        } else {
          const pagePayload = { id: '' }; // to hold wisnPoaPages data to iteratee for posting
          return this.httpClient
            .post(`${this.dataStoreUrl}/${pagePayload.id}`, pagePayload)
            .pipe(map(() => [pagePayload]));
        }
      })
    );
  }

  initiateData() {
    return forkJoin(this.getDataFromLocalDatabase(WISN_POA_PAGES_KEY));
  }

  getDataFromLocalDatabase(key: string): Observable<any> {
    return Observable.create(observer => {
      this.localDbService.getAll(key).subscribe(
        items => {
          if (items.length !== 0) {
            observer.next(items);
            observer.complete();
          } else {
            let dataStream$ = null;
            switch (key) {
              case WISN_POA_PAGES_KEY:
                dataStream$ = this.loadWisnPoaPages();
                break;
              default:
                console.error('The key passed is not recognized');
                break;
            }
            dataStream$.subscribe(
              data => {
                data.forEach(val => {
                  this.localDbService.add(key, val).subscribe(v => null);
                });
                observer.next(data);
                observer.complete();
              },
              error => observer.error(error)
            );
          }
        },
        error => observer.error(error)
      );
    });
  }

  addDataToLocalDatabase(key: string, data: any): Observable<any> {
    return Observable.create(observer => {
      this.localDbService.getAll(key).subscribe(
        items => {
          this.localDbService.add(key, data).subscribe(v => null, error => {});
          observer.next(data);
          observer.complete();
        },
        error => observer.error(error)
      );
    });
  }
