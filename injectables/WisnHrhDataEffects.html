  uploaoHRHdataFromCSV(hrhDataToUpload, hrhimportData, hrhdataBeforeUpload) {
    const importFormatData = hrhimportData ? hrhimportData : this.hrhEventsImportData;
    console.log(JSON.stringify(hrhDataToUpload));
    importFormatData.forEach((importPayload: any) => {
      importPayload.dataValues.forEach(dataValue => {
        hrhDataToUpload.forEach((hrhData: any) => {
          if (hrhData.dx === dataValue.dataElement && hrhData.orgunitId === importPayload.orgUnit) {
            dataValue.value = hrhData.value;
          }
        });
      });
    });

    importFormatData.forEach((importPayload: any) => {
      this.wisnAnalyticsService.updateSingleProgramDataValue(importPayload).subscribe((response) => {
        hrhdataBeforeUpload.forEach(oldData => {
          hrhDataToUpload.forEach(hrhUpload => {
            if (oldData.orgunitId === hrhUpload.orgunitId &&
              oldData[hrhUpload.dx].value !== hrhUpload.value) {
                const elementUid = oldData.orgunitId + '-' + hrhUpload.dx;
                try {
                  document.getElementById(elementUid).style.backgroundColor = '#b9ffb9';
                  document.getElementById(elementUid).style.borderColor = '#dc3545';
                } catch (e) {}
              }
          });
        });
      },
        error => this.store.dispatch(new WisnHRHErrorHandler(error))
      );
    });
  }

  fetchMultipleHRHeventData(selectedOrgunit, childOrgunits, period, program, hrhProgramDataElements, currentUser) {
    let progressNumerator = 0;
    const progressDenominator = childOrgunits.length;
    this.wisnAnalyticsService.getLatestMultipleEventDataValues(childOrgunits, program.id)
      .subscribe((eventsData: any) => {
          let currentHRHDataHolder = {};
        if (eventsData.events.length > 0) {
          currentHRHDataHolder = fromWisnHRHhelpers.generatedHRHData(eventsData, hrhProgramDataElements, period.id, program.id);
        } else {
          currentHRHDataHolder = fromWisnHRHhelpers.generateHRHDataWhenNoEvent(hrhProgramDataElements,
            currentUser, childOrgunits, period, program, progressNumerator);
        }
        progressNumerator += 1;
        const progresPercent = ((progressNumerator / progressDenominator) * 100).toFixed(1);
        const lastDataSelection = {
          orgunit: selectedOrgunit.id,
          period: period.id
        };
        this.store.dispatch(new AddWisnHrhData({
          lastDataSelection: lastDataSelection,
          wisnHrhData: [currentHRHDataHolder],
          wisnHrhDataElements: [
            { name: 'Facility'},
            ...hrhProgramDataElements
          ],
          progressLoader: progresPercent,
          stateEntityIds: fromUtilHelpers.generateStateUids(selectedOrgunit)
        }));
      },
        error => this.store.dispatch(new WisnHRHErrorHandler(error))
      );
  }

  fetchMultipleHRHAnalyticsData(selectedOrgunit, childOrgunits, period,
    hrhProgramDataEelements, hrhIndicators, program) {
    let progressNumerator = 0;
    const progressDenominator = childOrgunits ? childOrgunits.length : 1;
    this.wisnAnalyticsService.getMultipleHRHanalyticsData(childOrgunits ? childOrgunits : [selectedOrgunit], period.id, hrhIndicators )
      .subscribe((analyticsData: any) => {
        const analyticsValueIndex = analyticsData.headers.findIndex(head => head.name === 'value');
        const analyticsDxIndex = analyticsData.headers.findIndex(head => head.name === 'dx');
        const orgunitId = analyticsData.metaData ? analyticsData.metaData.dimensions.ou[0] : '';
        const currentOrgunit = {
          id: orgunitId,
          name: analyticsData.metaData.items[orgunitId].name};
        const currentHRHDataHolder = fromWisnHRHhelpers.generateHRHDataFromAnalytics(analyticsData, hrhProgramDataEelements,
          analyticsDxIndex, analyticsValueIndex, currentOrgunit, period, program);
        progressNumerator += 1;
        const progresPercent = ((progressNumerator / progressDenominator) * 100).toFixed(1);
        const lastDataSelection = {
          orgunit: selectedOrgunit.id,
          period: period.id
        };
        this.store.dispatch(new AddWisnHrhData({
          lastDataSelection: lastDataSelection,
          wisnHrhData: [currentHRHDataHolder],
          wisnHrhDataElements: [
            { name: 'Facility'},
            ...hrhProgramDataEelements
          ],
          progressLoader: progresPercent,
          stateEntityIds: fromUtilHelpers.generateStateUids(selectedOrgunit)
        }));
        },
        error => this.store.dispatch(new WisnHRHErrorHandler(error))
      );
  }
