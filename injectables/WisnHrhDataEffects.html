import { Injectable } from '@angular/core';
import { Store } from '@ngrx/store';
import { AppState } from '../reducers';
import { Actions, Effect, ofType } from '@ngrx/effects';
import { WisnAnalyticsService } from '../../shared/services/wisn-analytics.service';
import {catchError, map, switchMap, tap, withLatestFrom} from 'rxjs/operators';
import {
  AddHRHprogramDataValues,
  AddWisnHrhData,
  AddWisnHrhDataElements, CompleteDownloadingWisnHrhDatas, CompleteUploadingWisnHrhDatas,
  LoadHRHprogramData,
  LoadWisnHrhDatas,
  NoChangesOnWisnHRHState,
  UpdateWisnHrhData,
  UpdatingWisnHRHData,
  UpsertWisnHrhData,
  WisnHrhDataActionTypes, WisnHRHErrorHandler
} from '../actions/wisn-hrh-data.actions';
import * as fromWisnHRHactions from '../actions/wisn-hrh-data.actions';
import {WisnHmisDataActionTypes} from '../actions/wisn-hmis-data.actions';
import {LoadWisnHmisDatas} from '../actions';
import {of} from 'rxjs';
import * as fromWisnHrhSelectors from '../selectors/wisn-hrh.selectors';
import {WisnPoaPageService} from '../../shared/services/wisn-poa-page.service';
import * as fromUtilHelpers from '../../shared/helpers/util-functions';
import * as fromWisnHRHhelpers from '../../shared/helpers/wisn-hrh.helpers';
import { getCurrentGlobalFilter, getCurrentUser } from '../selectors';
import { getPageData } from '../selectors/wisn-poa-pages.selectors';

@Injectable()
export class WisnHrhDataEffects {
  period: any;
  program: any;
  dataElements: any;
  selectedOrgunit: any;
  hrhPageData: any;
  stateEntityIds: any;
  dataEntryLevel: any;
  wisnReportPage: any;
  hrhProgramDataElements: any[];
  hrhEventsImportData: any[];
  singleEventChanged: {};
  currentDomElementFieldUpdated: string;
  currentUser: any;
  todayTimeDate: string =  '' + new Date().getFullYear() + '-' + (1 + new Date().getMonth()) + '-' + new Date().getDate();

  constructor(
    private actions$: Actions,
    private wisnAnalyticsService: WisnAnalyticsService,
    private wisnPoaService: WisnPoaPageService,
    private store: Store<AppState>
  ) {}


  @Effect()
  checkingWisnHMISState = this.actions$.pipe(
    ofType(WisnHrhDataActionTypes.CheckingWisnHRHState),
    withLatestFrom(this.store),
    map(([action, appState]: [fromWisnHRHactions.CheckingWisnHRHState, any]) => {
      const wisnHRHState = appState.wisnHrhData;
      const currentOrgunit = appState.globalFilter.entities['wisn_poa_page'].ou;
      const curentPeriod = appState.globalFilter.entities['wisn_poa_page'].pe;

      if (wisnHRHState.entities.length > 1 &&
        wisnHRHState.lastDataSelection.orgunit === currentOrgunit.id &&
        wisnHRHState.lastDataSelection.period === curentPeriod.id) {
        // do nothing because state info is already there
        return new NoChangesOnWisnHRHState();
      }
        // dispatch action for loading the wisnReport
      return new LoadWisnHrhDatas(); // fire action for loading HRH data
    })
  );

  @Effect({dispatch: false})
  loadHRHdata$ = this.actions$.pipe(
    ofType(WisnHrhDataActionTypes.LoadWisnHrhDatas),
    withLatestFrom(this.store),
    map(([action, appState]: [fromWisnHRHactions.LoadWisnHrhDatas, any]) => {
      // console.log(appState);
      this.selectedOrgunit = appState.globalFilter.entities['wisn_poa_page'].ou;
      this.period = appState.globalFilter.entities['wisn_poa_page'].pe;
      this.hrhPageData = appState.wisnPoaPages.entities['hrh'];
      this.program = this.hrhPageData['dataSource'][0];
      this.dataEntryLevel = appState.generalConfigurations.dataEntryOrgUnitLevel ?
        appState.generalConfigurations.dataEntryOrgUnitLevel : 4;
      this.currentUser = appState.currentUser.user;
      this.wisnReportPage = appState.wisnPoaPages.entities['wisnreport'];
    }),
    switchMap(() =>
      this.wisnAnalyticsService.getProgramSectionDataElements(this.program)
    ),
    tap((programSection: any) => {
      const stateEntityUids = fromUtilHelpers.generateStateUids(this.selectedOrgunit);
      const programStageSections = programSection.programStages;
      const sections = programStageSections[0].programStageSections;
      this.stateEntityIds = stateEntityUids;
      sections.forEach((section: any) => {
        if (section.id === this.program['sections'][0]) {
          this.dataElements = section.dataElements;
        }
      });
      const lastDataSelection = {
        orgunit: this.selectedOrgunit.id,
        period: this.period.id
      };
      this.store.dispatch(new AddWisnHrhData({
        lastDataSelection: lastDataSelection,
        wisnHrhData: [],
        wisnHrhDataElements: [
          { name: 'Facility'},
          ...this.dataElements
        ],
        progressLoader: 0,
        stateEntityIds: stateEntityUids
      }));
      if ((this.dataEntryLevel - this.selectedOrgunit.level) === 1) {
        this.fetchMultipleHRHeventData(this.selectedOrgunit, this.selectedOrgunit.children,
          this.period, this.program, this.dataElements, this.currentUser.userCredentials);
      } else {
        let hrhIndicators = '';
        this.wisnReportPage.subPages.forEach((page: any) => {
          hrhIndicators += page.dataSource[1].id ? ';' + page.dataSource[1].id : '';
        });
        this.fetchMultipleHRHAnalyticsData(this.selectedOrgunit, this.selectedOrgunit.children,
          this.period, this.dataElements, hrhIndicators, this.program);
      }
      this.store.dispatch( new LoadHRHprogramData()); // loads the HRH program import formats
    })
  );

  @Effect()
  loadHRHProgram$ = this.actions$.pipe(
    ofType(WisnHrhDataActionTypes.LoadHRHprogramData),
    withLatestFrom(this.store),
    map(([action, appState]: [fromWisnHRHactions.LoadHRHprogramData, any]) => {
      this.selectedOrgunit = appState.globalFilter.entities['wisn_poa_page'].ou;
      this.period = appState.globalFilter.entities['wisn_poa_page'].pe;
      this.hrhPageData = appState.wisnPoaPages.entities['hrh'];
      this.program = this.hrhPageData['dataSource'][0];
    }),
    switchMap(() => this.wisnAnalyticsService.getProgramSectionDataElements(this.program)),
    map((programSection: any) => {
      const programStageSections = programSection.programStages;
      const sections = programStageSections[0].programStageSections;
      this.hrhProgramDataElements = fromWisnHRHhelpers.extractHRHdataElementsFromSections(sections);
    }),
    switchMap(() =>
      this.wisnAnalyticsService
      .getLatestMultipleChildrenEventDataValues(this.selectedOrgunit.id, this.program.id)
    ),
    map((eventsDataValues: any) => {
      const hrhProgramDataElements = this.hrhProgramDataElements.slice();
      const eventsData = fromWisnHRHhelpers.processFetchedHRHDataValues(hrhProgramDataElements,
        eventsDataValues, this.selectedOrgunit.children, this.program);
      this.hrhEventsImportData = eventsData;
      return new AddHRHprogramDataValues(eventsData);
    })
  );

  @Effect()
  updateHRHdata$ = this.actions$.pipe(
    ofType(WisnHrhDataActionTypes.UpdatingWisnHRHData),
    withLatestFrom(this.store.select(
      fromWisnHrhSelectors.getWisnHrhDataEntities
    )),
    map(([action, hrhDataState]: [fromWisnHRHactions.UpdatingWisnHRHData, any]) => {
      this.currentDomElementFieldUpdated = action.payload.changedData.domElementId;
      const hrhEventsDataCollection = this.hrhEventsImportData ? this.hrhEventsImportData : [];
      const singleEventUpload = fromWisnHRHhelpers.processUploadingHRHData(hrhEventsDataCollection, hrhDataState, action.payload);
      if (hrhDataState.length > 0) {
        if (action.payload.isFromHRHpageView) {
          this.store.dispatch( new UpsertWisnHrhData(action.payload));
        }
        this.hrhEventsImportData = hrhEventsDataCollection;
        this.singleEventChanged = singleEventUpload ;
      } else {
        return new LoadWisnHrhDatas(); // fire action for loading HRH data
      }
    }),
    switchMap(() => this.wisnAnalyticsService.updateSingleProgramDataValue(this.singleEventChanged)),
    map((response: any) => {
      try {
        document.getElementById(this.currentDomElementFieldUpdated)
        .style.backgroundColor = '#b9ffb9';
        document.getElementById(this.currentDomElementFieldUpdated)
        .style.borderColor = '#dc3545';
      } catch (e) {}
      return new AddHRHprogramDataValues(this.hrhEventsImportData);
    }),
    catchError(error => of(
      document.getElementById(this.currentDomElementFieldUpdated).style.backgroundColor = '#ebd1d6'
  ))
  );

  @Effect()
  downLoadingHRHdata$ = this.actions$.pipe(
    ofType(WisnHrhDataActionTypes.DownloadingWisnHrhDatas),
    withLatestFrom(this.store.select(
      fromWisnHrhSelectors.getWisnHrhDataElements
    ), this.store.select(
      fromWisnHrhSelectors.getWisnHrhDataEntities
    )),
    map(([action, hrhDataElementsCollection, hrhData]: [fromWisnHRHactions.DownloadingWisnHrhDatas, any, any]) => {
      const hrhDataElements = hrhDataElementsCollection;
      const tableHeader = [];
      hrhDataElements.forEach((dx: any) => {
          tableHeader.push(dx.name);
      });
      // tableHeader[0].name = 'Facility';
      const csvRows = fromWisnHRHhelpers.drawHRHCSVrows(hrhData, hrhDataElements);
      const row = [tableHeader, ...csvRows];
      let csvContent = 'data:text/csv;charset=utf-8,';
      row.forEach(function(rowArray) {
        const rowEntry = rowArray.join(',');
        csvContent += rowEntry + '\r\n';
      });
      const encodedUri = encodeURI(csvContent);
      const link = document.createElement('a');
      link.setAttribute('href', encodedUri);
      link.setAttribute('download',  'HRH Data.csv');
      link.click();

      return new CompleteDownloadingWisnHrhDatas();
    }),
  );

  @Effect()
  uploadingHRHdata$ = this.actions$.pipe(
    ofType(WisnHrhDataActionTypes.UploadingWisnHrhDatas),
    withLatestFrom(this.store.select(
      fromWisnHrhSelectors.getWisnHrhDataElements
    ), this.store.select(
      fromWisnHrhSelectors.getWisnHrhDataEntities
    ), this.store.select(
    fromWisnHrhSelectors.getWisnHrhImportData
    ), this.store.select(getCurrentGlobalFilter),
    this.store.select(getCurrentUser), this.store.select(getPageData('hrh'))),
    map(([action, hrhDataElements, hrhdataBeforeUpload, hrhimportData, currentGlobalFilters, currentUserData, hrhPageData]:
      [fromWisnHRHactions.UploadingWisnHrhDatas, any, any, any, any, any, any]) => {
      const selectedOrgunit = currentGlobalFilters.ou ? currentGlobalFilters.ou : {};
      const period = currentGlobalFilters.pe ? currentGlobalFilters.pe : {};
      const currentUser = currentUserData ? currentUserData : {};
      const program =  hrhPageData[0].dataSource[0];
      const csvData = action.payload;
      const processedDataFromCSVfile = fromWisnHRHhelpers.processHRHDataFromCSVUpload(csvData,
        hrhDataElements, selectedOrgunit, period, program, currentUser.userCredentials);
      const uploadedRestructured = processedDataFromCSVfile.uploadedRestructured;
      const hrhDataToUpload = processedDataFromCSVfile.hrhDataToUpload;
      const hrhdataBeforeUploadNew = (hrhdataBeforeUpload || []).map(oldData => Object.assign({}, oldData));
      const changesOnHRHdata = fromWisnHRHhelpers.getChangedHRHData(hrhdataBeforeUploadNew, hrhDataToUpload, hrhimportData);

      this.uploaoHRHdataFromCSV(changesOnHRHdata);
      return new CompleteUploadingWisnHrhDatas(uploadedRestructured);
    })
  );


  uploaoHRHdataFromCSV(changesOnHRHdata) {
    const importFormatData = changesOnHRHdata.hrhDataToImport ? changesOnHRHdata.hrhDataToImport : [];
    importFormatData.forEach((importPayload: any) => {
      this.wisnAnalyticsService.updateSingleProgramDataValue(importPayload).subscribe((response) => {
        (changesOnHRHdata.inputBoxChanges || []).forEach(inputBox => {
          if (inputBox.includes(importPayload.orgUnit)) {
            try {
              document.getElementById(inputBox).style.backgroundColor = '#b9ffb9';
              document.getElementById(inputBox).style.borderColor = '#dc3545';
            } catch (e) {}
          }
        });
      },
        error => this.store.dispatch(new WisnHRHErrorHandler(error))
      );
    });
  }

  fetchMultipleHRHeventData(selectedOrgunit, childOrgunits, period, program, hrhProgramDataElements, currentUser) {
    let progressNumerator = 0;
    const progressDenominator = childOrgunits.length;
    this.wisnAnalyticsService.getLatestMultipleEventDataValues(childOrgunits, program.id)
      .subscribe((eventsData: any) => {
          let currentHRHDataHolder = {};
        if (eventsData.events.length > 0) {
          currentHRHDataHolder = fromWisnHRHhelpers.generatedHRHData(eventsData, hrhProgramDataElements, period.id, program.id);
        } else {
          currentHRHDataHolder = fromWisnHRHhelpers.generateHRHDataWhenNoEvent(hrhProgramDataElements,
            currentUser, childOrgunits, period, program, progressNumerator);
        }
        progressNumerator += 1;
        const progresPercent = ((progressNumerator / progressDenominator) * 100).toFixed(1);
        const lastDataSelection = {
          orgunit: selectedOrgunit.id,
          period: period.id
        };
        this.store.dispatch(new AddWisnHrhData({
          lastDataSelection: lastDataSelection,
          wisnHrhData: [currentHRHDataHolder],
          wisnHrhDataElements: [
            { name: 'Facility'},
            ...hrhProgramDataElements
          ],
          progressLoader: progresPercent,
          stateEntityIds: fromUtilHelpers.generateStateUids(selectedOrgunit)
        }));
      },
        error => this.store.dispatch(new WisnHRHErrorHandler(error))
      );
  }

  fetchMultipleHRHAnalyticsData(selectedOrgunit, childOrgunits, period,
    hrhProgramDataEelements, hrhIndicators, program) {
    let progressNumerator = 0;
    const progressDenominator = childOrgunits ? childOrgunits.length : 1;
    this.wisnAnalyticsService.getMultipleHRHanalyticsData(childOrgunits ? childOrgunits : [selectedOrgunit], period.id, hrhIndicators )
      .subscribe((analyticsData: any) => {
        const analyticsValueIndex = analyticsData.headers.findIndex(head => head.name === 'value');
        const analyticsDxIndex = analyticsData.headers.findIndex(head => head.name === 'dx');
        const orgunitId = analyticsData.metaData ? analyticsData.metaData.dimensions.ou[0] : '';
        const currentOrgunit = {
          id: orgunitId,
          name: analyticsData.metaData.items[orgunitId].name};
        const currentHRHDataHolder = fromWisnHRHhelpers.generateHRHDataFromAnalytics(analyticsData, hrhProgramDataEelements,
          analyticsDxIndex, analyticsValueIndex, currentOrgunit, period, program);
        progressNumerator += 1;
        const progresPercent = ((progressNumerator / progressDenominator) * 100).toFixed(1);
        const lastDataSelection = {
          orgunit: selectedOrgunit.id,
          period: period.id
        };
        this.store.dispatch(new AddWisnHrhData({
          lastDataSelection: lastDataSelection,
          wisnHrhData: [currentHRHDataHolder],
          wisnHrhDataElements: [
            { name: 'Facility'},
            ...hrhProgramDataEelements
          ],
          progressLoader: progresPercent,
          stateEntityIds: fromUtilHelpers.generateStateUids(selectedOrgunit)
        }));
        },
        error => this.store.dispatch(new WisnHRHErrorHandler(error))
      );
  }
