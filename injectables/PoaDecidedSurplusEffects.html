import { Injectable } from '@angular/core';
import { Actions, Effect, ofType } from '@ngrx/effects';

  AddPoaDecidedSurplus,
  CollectingSurplus,
  LoadPoaDecidedSurpluss,
  NoChangesOnDecidedSurplusState,
  PoaDecidedSurplusActionTypes,
  UpdatedPoaDecidedSurplusSummary,
  UpdatePoaDecidedSurplus,
  UpdatePoaDecidedSurpluss
} from '../actions/poa-decided-surplus.actions';

  LoadPoaPriLists,

@Injectable()
export class PoaDecidedSurplusEffects {
  wisnReportPageDetails: any;
  selectedOrgunit: any;
  period: any;
  wisnReportEntity: any;
  indicatorsSource: any;
  priPageDetails: any;
  decidedSurplusState: any;
  surplusSummary: any;
  hrhProgram: any;
  decidedSurplusPage: any;
  wisnReportState: any;
  cadreFromGeneralConfig: any;
  decidedSurplusHRHdx: any;

  constructor(
    private action$: Actions,
    private poaAnalyticsService: PoaAnalyticsService,
    private wisnAnalyticsService: WisnAnalyticsService,
    private wisnPoaService: WisnPoaPageService,
    private store: Store<AppState>
  ) {}

  @Effect()
  checkingPoaSurplusState = this.action$.pipe(
    ofType(PoaDecidedSurplusActionTypes.CheckingDecidedSurplusState),
    withLatestFrom(this.store.select(getPoaDecidedSurplusDataState),
    this.store.select(getCurrentGlobalFilter)),
    map(
      ([action, poaDecidedSurplus, currentGlobalFilters]: [
        fromPoaDecidedSurplus.CheckingPoaDecidedSurplusState,
        any, any
      ]) => {
        const poaSurplusState = poaDecidedSurplus;
        const currentOrgunit = currentGlobalFilters.ou ? currentGlobalFilters.ou : {};
        const curentPeriod = currentGlobalFilters.pe ? currentGlobalFilters.pe : {};

        if (
          poaSurplusState.entities &&
          poaSurplusState.lastDataSelection.orgunit === currentOrgunit.id &&
          poaSurplusState.lastDataSelection.period === curentPeriod.id
        ) {
          // do nothing because state info is already there
          return new NoChangesOnDecidedSurplusState();
        }
        // dispatch action for loading the wisnReport
        return new LoadPoaDecidedSurpluss();
      }
    )
  );

  @Effect()
  loadDecidedSurplusInfo$ = this.action$.pipe(
    ofType(PoaDecidedSurplusActionTypes.LoadPoaDecidedSurpluss),
    withLatestFrom(this.store),
    map(
      ([action, appState]:
         [fromPoaDecidedSurplus.LoadPoaDecidedSurpluss, any]) => {
        const wisnReportState = appState.wisnReport;
        this.wisnReportState = wisnReportState;
        this.cadreFromGeneralConfig = appState.generalConfigurations.cadreData ?
          appState.generalConfigurations.cadreData : [];
        this.wisnReportEntity = wisnReportState.entities['report'];
        this.selectedOrgunit =
          appState.globalFilter.entities['wisn_poa_page'].ou;
        this.period = appState.globalFilter.entities['wisn_poa_page'].pe;
        this.wisnReportPageDetails =
          appState.wisnPoaPages.entities['wisnreport'];
        this.priPageDetails = appState.wisnPoaPages.entities['prireport'];
        this.indicatorsSource = this.priPageDetails['dataSource'];
        const hrhPageData = appState.wisnPoaPages.entities['hrh'];
        this.hrhProgram = hrhPageData['dataSource'][0];
        const surplusPage = appState.wisnPoaPages.entities['poaanalysis'];
        surplusPage.subPages.forEach((page: any) => {
          if (page.id === 'decidehrhsurplus') {
            this.decidedSurplusPage = page;
          }
        });
      }
    ),
    switchMap(() => this.wisnAnalyticsService.getProgramSectionDataElements(this.hrhProgram)),
    map((programSection: any) => {
      let decidedSurplusHRHdx = [];
      const programStageSections = programSection.programStages;
      const sections = programStageSections[0].programStageSections;
      sections.forEach((section: any) => {
        this.decidedSurplusPage.hrhDecidedSurplusSections
          .forEach((sectionId: any) => {
            if (section.id === sectionId ) {
              decidedSurplusHRHdx = decidedSurplusHRHdx.concat(section.dataElements);
            }
          });
      });
      decidedSurplusHRHdx.forEach((dataElement: any) => {
          // mutating decidedSurplusHRHdx array objects by adding cadreCode property
        dataElement['cadreCode'] = fromUtilHelpers.getFuzzyMatchCadreCode
        (dataElement.name, this.cadreFromGeneralConfig);
      });
      this.decidedSurplusHRHdx = decidedSurplusHRHdx;
    }),
    switchMap(() => this.wisnAnalyticsService
      .getLatestMultipleChildrenEventDataValues(this.selectedOrgunit.id, this.hrhProgram.id)),
    map((eventsDataValues: any) => {
      if (
        this.wisnReportState.entities['report'].length > 0 &&
        this.wisnReportState.lastDataSelection.orgunit ===
        this.selectedOrgunit.id &&
        this.wisnReportState.lastDataSelection.period === this.period.id
      ) {
        const decidedSuplusValueArray = this.decidedSurplusHRHdx
          .filter(dx => !dx.name.includes('Explanations'));
        const decidedSuplusExplanationArray = this.decidedSurplusHRHdx
          .filter(dx => dx.name.includes('Explanations'));

        this.wisnReportEntity.forEach((reportData: any) => {
          // mutating wisnReportEntity array objects by adding decidedsurplus & reasonForSurplus property
          reportData['decidedsurplus'] = fromDecidedSurplusHelpers.getDecidedsurplus(reportData,
            decidedSuplusValueArray, eventsDataValues.events);
          reportData['reasonForSurplus'] = fromDecidedSurplusHelpers.getDecidedsurplus(reportData,
            decidedSuplusExplanationArray, eventsDataValues.events);
        });
        const fromGeneratedDecidedSurplus =
        fromDecidedSurplusHelpers.generateDecidedSurplus(this.selectedOrgunit, this.period, this.indicatorsSource,
          this.wisnReportEntity, this.cadreFromGeneralConfig);
        this.decidedSurplusState = fromGeneratedDecidedSurplus.decidedSurplusState;
        this.surplusSummary = fromGeneratedDecidedSurplus.decidedSummary;
        this.store.dispatch
        (new AddPoaDecidedSurplus(fromGeneratedDecidedSurplus.toStore));
        return new CheckingWisnHRHState(); // loads the HRH data
      } else {
        return new CheckWisnReportState(); // load wisn report out puts so as the data to be available on store
      }
    }),
    catchError(error => of()) // fire an action to indicate that load data has failed
  );

  @Effect({ dispatch: false })
  backLoadDecidedSurplusInfo$ = this.action$.pipe(
    ofType(WisnReportActionTypes.AddWisnReport),
    tap((action: fromWisnReportActions.AddWisnReport) => {
      this.store.dispatch(new LoadPoaDecidedSurpluss());
    })
  );

  @Effect()
  updatingHRHSurplus$ = this.action$.pipe(
    ofType(PoaDecidedSurplusActionTypes.UpdatePoaDecidedSurplus),
    map((action: fromPoaDecidedSurplus.UpdatePoaDecidedSurplus) => {
      const modifiedChangeData = action.payload.rowData;
      const modifiedDataObject = action.payload.dataObject;
      const surplusState = this.decidedSurplusState;
      const decidedSuplusValueArray = this.decidedSurplusHRHdx
        .filter(dx => !dx.name.includes('Explanations'));
      const decidedSuplusExplanationArray = this.decidedSurplusHRHdx
        .filter(dx => dx.name.includes('Explanations'));
      const payloadToHRH = {changedData: [], isFromHRHpageView: false};

      decidedSuplusValueArray.forEach((dx: any) => {
        if (dx.cadreCode === modifiedDataObject.cadreCode) {
          payloadToHRH.changedData.push({
            dataElementId: dx.id,
            value: modifiedDataObject.decidedRedistribution,
            orgunitId: modifiedChangeData.orgUnit.orgUnitId,
            domElementId: ''
          });
        }
      });
      decidedSuplusExplanationArray.forEach((dx: any) => {
        if (dx.cadreCode === modifiedDataObject.cadreCode) {
          payloadToHRH.changedData.push({
            dataElementId: dx.id,
            value: modifiedDataObject.reasonForRedistribution,
            orgunitId: modifiedChangeData.orgUnit.orgUnitId,
            domElementId: ''
          });
        }
      });
      this.store.dispatch(new UpdatingWisnHRHData(payloadToHRH));


      // updating the state information
      surplusState.forEach((stateSurplus: any) => {
        if (stateSurplus.orgunitId === modifiedChangeData.orgUnit.orgUnitId) {
          stateSurplus.subTable.forEach((decisionTable: any) => {
            if (decisionTable.cadreCode === modifiedDataObject.cadreCode) {
              decisionTable.decidedsurplus =
                modifiedDataObject.decidedRedistribution;
              decisionTable.reasonForSurplus =
                modifiedDataObject.reasonForRedistribution;
            }
          });
        }
      });
      this.surplusSummary.forEach((summary: any) => {
        summary.total = 0;
        summary.items = [];
      });
      surplusState.forEach((stateSurplus: any) => {
        stateSurplus.subTable.forEach((decisionTable: any) => {
         this.surplusSummary = fromDecidedSurplusHelpers.updatingSurplusSummary(this.surplusSummary, decisionTable);
        });
      });
      // console.log(JSON.stringify(this.surplusSummary));
      return new UpdatePoaDecidedSurpluss({
        surplusState: surplusState,
        surplusSummary: _.concat(this.surplusSummary, {
          cadre: 'Total',
          total: _.sumBy(this.surplusSummary, 'total')
        }),
        summaryCollection: fromDecidedSurplusHelpers.updatingSummaries(this.surplusSummary)
      });
    })
  );

  @Effect()
  decidedSurplusSummarySelections$ = this.action$.pipe(
    ofType(PoaDecidedSurplusActionTypes.UpdatePoaDecidedSurplusSummary),
    withLatestFrom(this.store.select(getDecidedSurplusSummaryState)),
    map(
      ([action, summaryState]: [
        fromPoaDecidedSurplus.UpdatePoaDecidedSurplusSummary,
        any
      ]) => {
        const priRowData = action.payload.priRowData;
        summaryState.forEach((summary: any) => {
          if (summary.items.length > 0) {
            summary.items.forEach((summaryItem: any, index) => {
              if (summaryItem.id === action.payload.id) {
                priRowData.redistributedFrom.orgunitId = summaryItem.orgunitId;
                priRowData.redistributedFrom.orgunitName =
                  summaryItem.orgunitName;
                priRowData.redistributedFrom.redistSummaryId = summaryItem.id;
                summary.items.splice(index, 1);
              }
            });
          }
        });
        let surplusAvailableitems = [];
        summaryState.forEach((summary: any) => {
          surplusAvailableitems = surplusAvailableitems.concat(summary.items);
        });
        this.store.dispatch(
          new UpdatingPoaPriListFromDecidedSurplus(priRowData)
        );
        return new UpdatedPoaDecidedSurplusSummary({
          redistributionSurplusAvailable: surplusAvailableitems,
          decidedSummary: summaryState
        });
      }
    )
  );

  fetchDecidedSurplusFromHRH() {
    this.wisnAnalyticsService.getProgramSectionDataElements(this.hrhProgram).subscribe((programSection) => {
      let decidedSurplusHRHdx = [];
      const programStageSections = programSection.programStages;
      const sections = programStageSections[0].programStageSections;
      sections.forEach((section: any) => {
        this.decidedSurplusPage.hrhDecidedSurplusSections
          .forEach((sectionId: any) => {
        if (section.id === sectionId ) {
          decidedSurplusHRHdx = decidedSurplusHRHdx.concat(section.dataElements);
        }
      });
      });
    });
  }
