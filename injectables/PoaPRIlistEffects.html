import { Injectable } from '@angular/core';
import { Actions, Effect, ofType } from '@ngrx/effects';
import { AppState, getWisnReportState } from '../reducers';
import { Store } from '@ngrx/store';
import * as _ from 'lodash';
import { map, switchMap, tap, withLatestFrom } from 'rxjs/operators';
import {
  AddPoaPriList,
  CompletedApplyingAutoConstraintsOnPRIstate,
  CompletedApplyingAutoConstraintsOnPRIstateRedistribute,
  CompleteDownloadPoaPriLists,
  LoadPoaPriLists,
  NoChangesOnPoaPRIState,
  PoaPriListActionTypes,
  UpdatePoaPriList,
  UpdatePoaPriListFromDecidedSurplus,
  FetchingPoaPRIStateDraft,
  CheckingPoaPriState,
  SavingPoaPRIStateDraft
} from '../actions/poa-pri-list.actions';
import * as fromPOAPRIlistActions from '../actions/poa-pri-list.actions';
import * as fromPOAHelpers from '../../pages/poa/helpers/index';
import { WisnAnalyticsService } from '../../shared/services/wisn-analytics.service';
import {
  CheckWisnReportState,
  WisnReportActionTypes
} from '../actions/wisn-report.actions';
import * as fromWisnReportActions from '../actions/wisn-report.actions';
import * as fromPoaCostAssumptionsSelectors from '../selectors/poa-cost-assumptions.selectors';
import * as fromPoaPriSelectors from '../selectors/poa-pri.selectors';
import * as fromDecidedSurplusSelectors from '../selectors/poa-decided-surplus.selectors';
import { PoaAnalyticsService } from '../../shared/services/poa-analytics.service';
import * as fromPRIlistHelpers from '../../shared/helpers/poa-pri-list.helpers';
import { getCurrentGlobalFilter } from '../selectors';
import { getPoaPriState } from '../selectors/poa-pri.selectors';
import { getPageData } from '../selectors/wisn-poa-pages.selectors';
import * as fromUtilHelpers from '../../shared/helpers/util-functions';

@Injectable()
export class PoaPRIlistEffects {
  constructor(
    private action$: Actions,
    private poaAnalyticsService: WisnAnalyticsService,
    private poaService: PoaAnalyticsService,
    private store: Store<AppState>
  ) {}

  @Effect()
  checkingPOAPriState = this.action$.pipe(
    ofType(PoaPriListActionTypes.CheckingPoaPRIState),
    withLatestFrom(this.store.select(getPoaPriState),
      this.store.select(getCurrentGlobalFilter)),
    map(
      ([action, poaPriState, currentGlobalFilters]:
        [fromPOAPRIlistActions.CheckingPoaPriState, any, any]) => {
        const selectedOrgunit = currentGlobalFilters.ou ? currentGlobalFilters.ou : {};
        const period = currentGlobalFilters.pe ? currentGlobalFilters.pe : {};
        if (
          poaPriState.entities.length > 0 &&
          poaPriState.lastDataSelection.orgunit === selectedOrgunit.id &&
          poaPriState.lastDataSelection.period === period.id
        ) {
          // do nothing because state info is already there
          return new NoChangesOnPoaPRIState();
        }
        // dispatch action for loading the wisnReport
        // return new LoadPoaPriLists();
        return new FetchingPoaPRIStateDraft();
      }
    )
  );

  @Effect({ dispatch: false })
  loadPRIlist$ = this.action$.pipe(
    ofType(PoaPriListActionTypes.LoadPoaPriLists),
    withLatestFrom(
      this.store.select(getCurrentGlobalFilter),
      this.store.select(getPageData('prireport')),
      this.store.select(getWisnReportState)),
    map(([action, currentGlobalFilters, priPageDetails, wisnReportState]:
      [fromPOAPRIlistActions.LoadPoaPriLists, any, any, any]) => {
      const selectedOrgunit = currentGlobalFilters.ou ? currentGlobalFilters.ou : {};
      const period = currentGlobalFilters.pe ? currentGlobalFilters.pe : {};
      const indicatorsSource = priPageDetails[0].dataSource ? priPageDetails[0].dataSource : [];
      const wisnReportData = wisnReportState.entities['report'] ? wisnReportState.entities['report'] : [];
      if (
        wisnReportData.length > 0 &&
        wisnReportState.lastDataSelection.orgunit ===
        selectedOrgunit.id &&
        wisnReportState.lastDataSelection.period === period.id
      ) {
        const priCollection = fromPRIlistHelpers.generatePRIData(wisnReportData,
          indicatorsSource, selectedOrgunit);
        const lastDataSelection = {
          orgunit: selectedOrgunit.id,
          period: period.id
        };
        this.store.dispatch(
          new AddPoaPriList({
            lastDataSelection: lastDataSelection,
            poaPriList: priCollection,
            priDraftTime: ''
          })
        );
      } else {
        this.store.dispatch(new CheckWisnReportState()); // fire action for loading wisn report
      }
    })
  );

  @Effect({dispatch: false })
  whenWisnReportAdded$ = this.action$.pipe(
    ofType(WisnReportActionTypes.AddWisnReport),
    tap((action: fromWisnReportActions.AddWisnReport) => {
      this.store.dispatch(new CheckingPoaPriState());
    })
  );

  @Effect({ dispatch: false })
  whenWisnReportUploaded$ = this.action$.pipe(
    ofType(WisnReportActionTypes.CompleteUpLoadingWisnReport),
    map(
      ([action]:
         [fromWisnReportActions.CompleteUpLoadingWisnReport, any]) => {
          this.store.dispatch(new CheckingPoaPriState());
      }
    )
  );

  @Effect()
  updatingPRIlist$ = this.action$.pipe(
    ofType(PoaPriListActionTypes.UpdatingPoaPriList),
    withLatestFrom(
      this.store.select(
        fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsSalaryCostState
      ),
      this.store.select(
        fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsAllowanceCostState
      ),
      this.store.select(fromPoaPriSelectors.getPoaPriEntitieState)
    ),
    map(
      ([action, salaryCosts, allowanceCosts, poaPriData]: [
        fromPOAPRIlistActions.UpdatingPoaPriList, any, any, any]) => {
        let costValue = 0;
        const distributionSummary = [];
        const redistributionSummary = [];

        if (action.payload.currentView === 'distribute') {
          costValue = fromPRIlistHelpers.getCostValueOnManualDistribute(salaryCosts, action.payload, costValue);
          fromPOAHelpers.createDistributionSummary(
            poaPriData,
            distributionSummary
          );

        } else if (action.payload.currentView === 'redistribute') {
          costValue = fromPRIlistHelpers.getCostValueOnManualRedistribute(allowanceCosts, costValue);
          fromPOAHelpers.createRedistributionSummary(
            poaPriData,
            redistributionSummary
          );
        }
        const updatedPRIdata = fromPRIlistHelpers.updatePRIdata(poaPriData, action.payload, costValue);
        const poaPriAssignmentPayload = {
          priState: updatedPRIdata,
          distributionSummary: distributionSummary,
          redistributionSummary: redistributionSummary
        };
        return new UpdatePoaPriList(poaPriAssignmentPayload);
      }
    )
  );

  @Effect()
  updatingPRIfromDecidedSurplus$ = this.action$.pipe(
    ofType(PoaPriListActionTypes.UpdatingPoaPriListFromDecidedSurplus),
    withLatestFrom(this.store.select(fromPoaPriSelectors.getPoaPriEntitieState)),
    map(
      ([action, poaPriData]: [fromPOAPRIlistActions.UpdatingPoaPriListFromDecidedSurplus, any]) => {
        if (poaPriData) {
          poaPriData.forEach((priRow: any) => {
            if (priRow.id === action.payload.id) {
              priRow = action.payload;
            }
          });
        }
        return new UpdatePoaPriListFromDecidedSurplus(poaPriData);
      }
    )
  );

  @Effect()
  downloadingPRI$ = this.action$.pipe(
    ofType(PoaPriListActionTypes.DownloadPoaPriLists),
    withLatestFrom(
      this.store.select(fromPoaPriSelectors.getPoaPriEntitieState)
    ),
    map(
      ([action, poaPriData]: [
        fromPOAPRIlistActions.DownloadPoaPriLists,
        any
      ]) => {
        const tableHeader = [
          'S/N', 'PRI',
          'Cadre type',
          'Organization unit',
          'Organization Unit Hierrachy',
          'Assignment Designation',
          'New Assignment Cost',
          'Redistribution Cost (allowance)',
          'Redistributed from'
        ];
        const tableRows = fromPRIlistHelpers.drawPRIreportCSVrows(poaPriData);
        const row = [tableHeader, ...tableRows];
        let csvContent = 'data:text/csv;charset=utf-8,';
        row.forEach(function(rowArray) {
          const rowEntry = rowArray.join(',');
          csvContent += rowEntry + '\r\n';
        });
        const encodedUri = encodeURI(csvContent);
        const link = document.createElement('a');
        link.setAttribute('href', encodedUri);
        link.setAttribute('download', 'PRI Report.csv');
        link.click();

        return new CompleteDownloadPoaPriLists();
      }
    )
  );

  @Effect({dispatch: false})
  applyingAutoConstraints$ = this.action$.pipe(
    ofType(PoaPriListActionTypes.ApplyAutoConstraintsOnPRIstate),
    withLatestFrom(
      this.store.select(fromPoaPriSelectors.getPoaPriEntitieState),
      this.store.select(
        fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsMaxBudgetState
      ), this.store.select(
        fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsMaxNumberState
      ),
      this.store.select(
        fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsSalaryCostState
      )
    ),
    tap(
      ([action, poaPriData, maxBudgetConstraints, maxNumberConstraints , salaryCosts]:
         [fromPOAPRIlistActions.ApplyAutoConstraintsOnPRIstate, any, any, any, any]) => {
        const cleanMaxBudgetsCosts = maxBudgetConstraints.filter(budget => !budget.name.toLowerCase().includes('total'));
        const maximumHRH = (maxNumberConstraints || []).map(maxHRH => Object.assign({}, maxHRH));
        const budgetConstraints = cleanMaxBudgetsCosts.concat(maximumHRH);
        const totalMaxBudgetIndex = maxBudgetConstraints.findIndex(budget => budget.name.toLowerCase().indexOf('total') > -1);
        const monthsForSalaryindex = salaryCosts.findIndex(item => item.name.toLowerCase().indexOf('months') > -1);
        const monthsForSalary = salaryCosts[monthsForSalaryindex].value;
        const cleanSalaryCosts = salaryCosts.filter(item => !item.name.toLowerCase().includes('months'));
        const minimumCost = monthsForSalary * Math.min.apply(null, cleanSalaryCosts.map(item => item.value));
        const maxmumBudgetCost = maxBudgetConstraints[totalMaxBudgetIndex].value;
        if (maxmumBudgetCost > minimumCost) {
            const newPRIdata = fromPRIlistHelpers.regenratePRIbasedonAutoDistribute(poaPriData, salaryCosts, budgetConstraints,
            maxmumBudgetCost, minimumCost, monthsForSalaryindex);
            this.store.dispatch(new CompletedApplyingAutoConstraintsOnPRIstate(newPRIdata));
            this.store.dispatch(new SavingPoaPRIStateDraft());
        } else {
          alert('Total Maximum bugdet is below minimal salary cost, Hence HRH distribution cant processed');
        }
        // this.store.dispatch(new CostAssumptionsLoadMaxHRHFromPRIlist());
      }
    )
  );

  @Effect()
  applyingAutoConstraintsRedstribute$ = this.action$.pipe(
    ofType(PoaPriListActionTypes.ApplyAutoConstraintsOnPRIstateRedistribute),
    withLatestFrom(
      this.store.select(fromPoaPriSelectors.getPoaPriEntitieState),
      this.store.select(
        fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsAllowanceCostState
      ),
      this.store.select(
        fromDecidedSurplusSelectors.getDecidedSurplusSummaryStateDetailed
      )
    ),
    tap(
      ([action, poaPriData, allowanceCosts, decidedSummaryCollections]: [
        fromPOAPRIlistActions.ApplyAutoConstraintsOnPRIstateRedistribute,
        any,
        any,
        any
      ]) => {
        const updatedPRIdata = fromPRIlistHelpers.updatePRIlistOnAutoRedistribute(poaPriData,
          allowanceCosts, decidedSummaryCollections);
        this.store.dispatch(new CompletedApplyingAutoConstraintsOnPRIstateRedistribute(updatedPRIdata));
        this.store.dispatch(new SavingPoaPRIStateDraft());
      }
    )
  );

  @Effect({ dispatch: false })
  savingPRIdraft$ = this.action$.pipe(
    ofType(PoaPriListActionTypes.SavingPoaPRIStateDraft),
    withLatestFrom(
      this.store.select(fromPoaPriSelectors.getPoaPriEntitieState),
      this.store.select(getCurrentGlobalFilter)
    ),
    map(
      ([action, priState, currentGlobalFilters]:
        [fromPOAPRIlistActions.SavingPoaPRIStateDraft, any, any]) => {
      const selectedOrgunit = currentGlobalFilters.ou ? currentGlobalFilters.ou : {};
        const priStateDraft = {
          lastUpdated: fromUtilHelpers.todaysDate(),
          payload: priState
        };
        this.poaService.uploadPRIstateDraft(
          selectedOrgunit.id,
          priStateDraft
        ).subscribe(response => {

        });
      }
    )
  );

  @Effect({dispatch: false})
  fetchingPRIstateDatastore$ = this.action$.pipe(
    ofType(PoaPriListActionTypes.FetchingPoaPRIStateDraft),
    withLatestFrom(this.store.select(getCurrentGlobalFilter)),
    map(([action, currentGlobalFilters]:
      [fromPOAPRIlistActions.FetchingPoaPRIStateDraft, any]) => {
      const selectedOrgunit = currentGlobalFilters.ou ? currentGlobalFilters.ou : {};
      const period = currentGlobalFilters.pe ? currentGlobalFilters.pe : {};
      // starting fetching PRI-STATE
      this.poaService.fetchPRIstateDraft(selectedOrgunit.id).subscribe(
        (response: any) => {
          const lastDataSelection = {
            orgunit: selectedOrgunit.id,
            period: period.id
          };
          if (response.payload) {
            this.store.dispatch(
              new AddPoaPriList({
                lastDataSelection: lastDataSelection,
                poaPriList: response.payload,
                priDraftTime: response.lastUpdated
              })
            );
          } else {
            this.store.dispatch(
              new LoadPoaPriLists()
            );
          }
        },
        error1 => {
