import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { WisnAnalyticsService } from '../../services/wisn-analytics.service';
import { map, switchMap } from 'rxjs/operators';

@Component({
  selector: 'app-input-field',
  templateUrl: './input-field.component.html',
  styleUrls: ['./input-field.component.css']
})
export class InputFieldComponent implements OnInit {
  @Input()
  value: any;
  @Input() tooltip: any;
  @Input()
  dataElement: any;
  @Input()
  orgunit: any;
  @Input()
  dataset: any;
  @Input()
  period: any;
  @Input()
  inputType: any;
  @Input()
  dataRow: any;
  @Input()
  programStage: any;
  @Input()
  eventId: any;
  @Input()
  disableInput: boolean;
  @Input()
  storedBy: any;

  @Output()
  newValue = new EventEmitter();

  todayTimeDate: string =
    '' +
    new Date().getFullYear() +
    '-' +
    (1 + new Date().getMonth()) +
    '-' +
    new Date().getDate();

  constructor(private wisnAnalyticsService: WisnAnalyticsService) {
    this.storedBy = this.storedBy ? this.storedBy : 'Admin';
  }

  ngOnInit() {}

  updateDataValue(event, elementUid) {
    const param =
      'de=' +
      this.dataElement.id +
      '&co=' +
      this.dataset.co +
      '&ds=' +
      this.dataset.id +
      '&ou=' +
      this.orgunit.id +
      '&pe=' +
      this.period.id +
      '&value=' +
      this.value +
      '';
    document.getElementById(elementUid).style.backgroundColor = '#fff6b2';

    if (this.dataset.type === 'dataSets') {
      // console.log(this.value, this.newValue);
      this.wisnAnalyticsService.importSingleDataValue(param).subscribe(
        (response: any) => {
          try {
            document.getElementById(elementUid).style.backgroundColor =
              '#b9ffb9';
            document.getElementById(elementUid).style.borderColor = '#dc3545';
          } catch (e) {
            // do nothing for the mean time
          }
          const newStateValue = {
            orgunitId: this.orgunit.id,
            dataElementId: this.dataElement.id,
            value: this.value
          };
          this.newValue.emit(newStateValue);
        },
        error1 => {
          try {
            document.getElementById(elementUid).style.backgroundColor =
              '#ebd1d6';
          } catch (e) {
            // do nothing for the mean time
          }
        }
      );
    } else if (this.dataset.type === 'programs') {
      this.updateProgramEvent(elementUid);
    }
  }

  updateProgramEvent(elementUid) {
    const newStateValue = {
      orgunitId: this.orgunit.id,
      dataElementId: this.dataElement.id,
      value: this.value,
      domElementId: elementUid
    };
    this.newValue.emit(newStateValue);
  }
}
