import { Component, OnInit, ViewChild, ElementRef } from "@angular/core";
import { Observable } from "rxjs";
import { Store } from "@ngrx/store";
import * as fromPoaPriSelectors from "../../../../store/selectors/poa-pri.selectors";
import { AppState } from "../../../../store";
import {
  DownloadPoaPriLists,
  ResetPriListNotification,
  FetchingPoaPRIStateDraft,
  LoadPoaPriLists
} from "../../../../store/actions/poa-pri-list.actions";
import {
  DownLoadWisnReport,
  UploadWisnReport
} from "../../../../store/actions/wisn-report.actions";
import * as fromAnimations from "../../../../shared/animations/index";
import * as fromWisnReportSelectors from "../../../../store/selectors/wisn-report.selectors";
import { VisualizationExportService } from "../../../../shared/services/visualizationExportService.service";
import { currentUserRolePermissions } from "src/app/store/selectors/general-configuration.selector";
import { WisnPoaPageService } from "src/app/shared/services/wisn-poa-page.service";

@Component({
  selector: "app-pri-report",
  templateUrl: "./pri-report.component.html",
  styleUrls: ["./pri-report.component.css"],
  animations: [fromAnimations.fadeInOut]
})
export class PriReportComponent implements OnInit {
  @ViewChild("currentPriReport")
  table: ElementRef;
  @ViewChild("priSummaryTable")
  summaryTable: ElementRef;
  orgUnitSearchText: "";
  cadreSearchText: "";
  poaLoaded$: Observable<boolean>;
  poaLoading$: Observable<boolean>;
  poaPriData$: Observable<any>;
  poaPriNotification$: Observable<any>;
  wisnReportProgress$: Observable<any>;
  isDistributeHRHDisabled$: Observable<any>;
  poaRedistributionSummary$: Observable<any>;
  poaNewDistributionSummary$: Observable<any>;
  poaTotalSummary$: Observable<any>;
  priDraftDate$: Observable<any>;
  searchText: "";
  p = 1;
  itemsPerPage = 100;
  userRoleAction: any;

  showDownloadDropdown = false;
  dropDownMenuX: any;
  dropDownMenuY: any;

  constructor(
    private store: Store<AppState>,
    private tableExport: VisualizationExportService,
    private wisnPoaPages: WisnPoaPageService
  ) {
    this.poaLoaded$ = this.store.select(
      fromPoaPriSelectors.getPoaPriLoadedState
    );

    this.poaLoading$ = this.store.select(
      fromPoaPriSelectors.getPoaPriDataLoading
    );

    this.poaPriData$ = this.store.select(
      fromPoaPriSelectors.getPoaPriEntitieState
    );

    this.poaTotalSummary$ = this.store.select(
      fromPoaPriSelectors.getPoaTotalSummaryState
    );
    this.poaNewDistributionSummary$ = this.store.select(
      fromPoaPriSelectors.getDistributionSummaryState
    );
    this.poaRedistributionSummary$ = this.store.select(
      fromPoaPriSelectors.getredistributionSummaryState
    );
    this.isDistributeHRHDisabled$ = this.store.select(
      fromPoaPriSelectors.canDistributeHRH
    );

    this.wisnReportProgress$ = this.store.select(
      fromWisnReportSelectors.getWisnReportProgresLoaderState
    );

    this.poaPriNotification$ = this.store.select(
      fromPoaPriSelectors.getPoaPriNotificationState
    );

    this.wisnReportProgress$ = this.store.select(
      fromWisnReportSelectors.getWisnReportProgresLoaderState
    );

    this.priDraftDate$ = this.store.select(
      fromPoaPriSelectors.getPoaPriDrafttime
    );

    this.store
      .select(fromPoaPriSelectors.getPoaPriLoadedState)
      .subscribe(response => {
        this.userRoleAction = wisnPoaPages.getUserPermissionAction("prireport");
      });
  }

  ngOnInit() {}

  onResetNotification(emptyNotificationMessage) {
    this.store.dispatch(
      new ResetPriListNotification({
        notification: emptyNotificationMessage
      })
    );
  }

  onToggleDropdown(e) {
    if (e) {
      e.stopPropagation();
    }
    e.cancelBubble = true;
    this.dropDownMenuX = 0;
    this.dropDownMenuY = 49;
    this.showDownloadDropdown = !this.showDownloadDropdown;
    return false;
  }
  downloadTable(poaData, downloadMode) {
    const dateTime = new Date();
    const filename =
      "PRI Report - generated on " +
      dateTime.getFullYear() +
      (dateTime.getMonth() + 1 < 10 ? "-0" : "-") +
      (dateTime.getMonth() + 1) +
      (dateTime.getDay() < 10 ? "-0" : "-") +
      dateTime.getDay() +
      " " +
      (dateTime.getHours() < 10 ? ":0" : ":") +
      dateTime.getHours() +
      (dateTime.getMinutes() < 10 ? ":0" : ":") +
      dateTime.getMinutes() +
      "hrs.xls";

    if (downloadMode === "CURRENT_PAGE" && this.table) {
      const el = this.table.nativeElement;
      if (el) {
        this.tableExport.exportXLS(filename, el.outerHTML);
      }
    } else {
      const summaryEl = this.summaryTable.nativeElement;
      if (summaryEl) {
        this.tableExport.exportXLS(filename, summaryEl.outerHTML);
      }
      this.tableExport.exportPaginatedCSV(poaData, filename);
    }
  }

  chooseFile() {
    const element: HTMLElement = document.querySelector(
      'input[type="file"]'
    ) as HTMLElement;
    element.click();
    // alert('This upload feature is under rennovation due to upload formats of the WISN Outputs.');
  }

  receiveFile(csv: any) {
    // just capture the file and manipulate it later
    const fileReaded = csv.target.files[0];
    const csvRows = [];
    const reader: FileReader = new FileReader();
    reader.readAsText(fileReaded);
    reader.onload = e => {
      const csvData: any = reader.result;
      const allTextLines = csvData.split(/\r|\n|\r/);
      const headers = allTextLines[0].split(",");
      for (let i = 0; i < allTextLines.length; i++) {
        // split content based on comma
        const data = allTextLines[i].split(",");
        if (data.length === headers.length) {
          const tarr = [];
          for (let j = 0; j < headers.length; j++) {
            tarr.push(data[j].replace('"', "").replace('"', ""));
          }
          csvRows.push(tarr);
        }
      }
      this.store.dispatch(new UploadWisnReport(csvRows));
    };
  }

  onSearchByOrgUnit(e) {
    if (e) {
      e.stopPropagation();
    }
    this.orgUnitSearchText = e ? e.target.value.trim() : this.orgUnitSearchText;
  }

  onSearchByCadre(e) {
    if (e) {
      e.stopPropagation();
    }
    this.cadreSearchText = e ? e.target.value.trim() : this.cadreSearchText;
  }

  onUpdatePageSize(e) {
    this.itemsPerPage = e;
  }
  printReport() {
    window.print();
  }
  onCurrentPageUpdate(e) {
    this.p = e;
  }

  downloadWisnImportFormat() {
    this.store.dispatch(new DownLoadWisnReport());
  }

  fetchLastPRIdraft() {
    this.store.dispatch(new FetchingPoaPRIStateDraft());
  }

  loadPRI() {
    this.store.dispatch(new LoadPoaPriLists());
  }
}
