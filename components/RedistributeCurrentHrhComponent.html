import { Component, OnDestroy, OnInit } from "@angular/core";
import { Observable } from "rxjs";
import { Store } from "@ngrx/store";
import * as fromPoaPriSelectors from "../../../../store/selectors/poa-pri.selectors";
import { AppState } from "../../../../store";
import * as fromRouterActions from "../../../../store/actions/router.actions";
import {
  ApplyAutoConstraintsOnPRIstateRedistribute,
  DownloadPoaPriLists,
  LoadPoaPriLists,
  ResetPoaPriList,
  SavingPoaPRIStateDraft,
  UpdatingPoaPriList
} from "../../../../store/actions/poa-pri-list.actions";
import * as fromDecidedSurplusSelectors from "../../../../store/selectors/poa-decided-surplus.selectors";
import {
  UpdatePoaDecidedSurplusSummary,
  ResetDecidedSurplusNotification
} from "../../../../store/actions/poa-decided-surplus.actions";
import {
  ApplyAutoConstraintsOnPRIstate,
  ResetPriListNotification
} from "../../../../store/actions/poa-pri-list.actions";
import { style, animate, transition, trigger } from "@angular/animations";

import { currentUserRolePermissions } from "src/app/store/selectors/general-configuration.selector";
import { WisnPoaPageService } from "src/app/shared/services/wisn-poa-page.service";

@Component({
  selector: "app-redistribute-current-hrh",
  templateUrl: "./redistribute-current-hrh.component.html",

        style({ opacity: 0 }),
        animate(500, style({ opacity: 1 }))
      ]),
      transition(":leave", [
        // :leave is alias to '* => void'
        animate(500, style({ opacity: 0 }))
      ])
    ])
  ]
})
export class RedistributeCurrentHrhComponent implements OnInit, OnDestroy {
  poaLoaded$: Observable<boolean>;
  poaPriData$: Observable<any>;
  poaDecidedSuplusSummary$: Observable<any>;
  isRedistributeHRHDisabled$: Observable<any>;
  poaDecidedSuplusSummary: any;
  poaRedistributionsSummary$: Observable<any>;
  redistributedNotification$: Observable<any>;
  decidedSurplusNotification$: Observable<any>;
  wisnReportProgress$: Observable<any>;
  showDistributionSummary = false;
  priStateEdited = false;
  userRoleAction: any;
  orgUnitSearchText: "";
  cadreSearchText: "";
  searchText: "";
  p = 1;
  itemsPerPage = 100;

  constructor(
    private store: Store<AppState>,
    private wisnPoaPages: WisnPoaPageService
  ) {
    this.poaLoaded$ = this.store.select(
      fromPoaPriSelectors.getPoaPriLoadedState
    );

    this.poaPriData$ = this.store.select(
      fromPoaPriSelectors.getPoaRedistributionEntities
    );

    this.isRedistributeHRHDisabled$ = this.store.select(
      fromPoaPriSelectors.canRedistributeHRH
    );

    this.poaDecidedSuplusSummary$ = this.store.select(
      fromDecidedSurplusSelectors.getDecidedSurplusSummaryStateDetailed
    );

    this.redistributedNotification$ = this.store.select(
      fromPoaPriSelectors.getPoaPriNotificationState
    );

    this.decidedSurplusNotification$ = this.store.select(
      fromDecidedSurplusSelectors.getDecidedSurplusNotificationState
    );

    this.poaRedistributionsSummary$ = this.store.select(
      fromPoaPriSelectors.getredistributionSummaryState
    );

    this.wisnReportProgress$ = this.store.select(
      fromWisnReportSelectors.getWisnReportProgresLoaderState
    );

    this.userRoleAction = wisnPoaPages.getUserPermissionAction(
      "redistributesurplus"
    );
  }

  ngOnInit() {}
  ngOnDestroy() {
    if (this.priStateEdited) {
      this.store.dispatch(new SavingPoaPRIStateDraft());
    }
  }
  onResetNotification(emptyNotificationMessage) {
    this.store.dispatch(
      new ResetPriListNotification({
        notification: emptyNotificationMessage
      })
    );
    this.store.dispatch(
      new ResetDecidedSurplusNotification({
        notification: emptyNotificationMessage
      })
    );
  }

  onGoToDecidedSurplus() {
    console.log("HEy");
    this.store.dispatch(
      new fromRouterActions.Go({
        path: ["/poa/poasettings/decidesurplus"],
        extras: {
          preserveQueryParams: true,
          queryParamsHandling: "merge"
        }
      })
    );
  }
  onUpdatePageSize(e) {
    this.itemsPerPage = e;
  }

  onCurrentPageUpdate(e) {
    this.p = e;
  }
  onSearchReport(e) {
    if (e) {
      e.stopPropagation();
    }
    this.searchText = e ? e.target.value.trim() : this.searchText;
  }
  changeRowColor(AssignmentDesignstion) {
    let styleObject = {};
    if (AssignmentDesignstion === "Unassigned") {
      styleObject = { "background-color": "#ffffff" };
    } else if (AssignmentDesignstion === "Redistributed") {
      styleObject = { "background-color": "#f0fff0" };
    } else if (AssignmentDesignstion === "New") {
      styleObject = { "background-color": "#cdeffe" };
    }
    return styleObject;
  }

  assignmentDesignationChange(changedValue, rowData) {
    const changedData = {
      changedAssignmentDesignation: changedValue,
      rowData: rowData,
      currentView: "redistribute"
    };
    this.store.dispatch(new UpdatingPoaPriList(changedData));
    this.priStateEdited = true;
  }

  redistributedFromChange(redistId, priData) {
    this.store.dispatch(
      new UpdatePoaDecidedSurplusSummary({ id: redistId, priRowData: priData })
    );
  }

  downloadTable(format) {
    if (format === "CSV") {
      this.store.dispatch(new DownloadPoaPriLists());
    } else {
    }
  }

  applyAutoConstraints() {
    this.store.dispatch(new ApplyAutoConstraintsOnPRIstateRedistribute());
    this.priStateEdited = true;
  }

  onShowDistribution(e) {
    if (e) {
      e.stopPropagation();
    }
    this.showDistributionSummary = !this.showDistributionSummary;
  }

  resetPRIlist() {
    // this function is to reset POA PRI from wisn report data
    this.store.dispatch(new LoadPoaPriLists()); // Redistributed means views is on Redistribute
    // this.store.dispatch( new ResetPoaPriList('Redistributed')); //Redistributed means views is on Redistribute
    this.priStateEdited = false;
  }

  onSearchByOrgUnit(e) {
    if (e) {
      e.stopPropagation();
    }
    this.orgUnitSearchText = e ? e.target.value.trim() : this.orgUnitSearchText;
  }

  onSearchByCadre(e) {
    if (e) {
      e.stopPropagation();
    }
    this.cadreSearchText = e ? e.target.value.trim() : this.cadreSearchText;
  }
}
