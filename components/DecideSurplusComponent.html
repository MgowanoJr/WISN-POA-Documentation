import { Component, OnInit, ElementRef } from '@angular/core';
import { style, animate, transition, trigger } from '@angular/animations';

import { Store } from '@ngrx/store';
import {
  AppState,
  getGlobalConfigurationState
} from '../../../../store/reducers';
import { Observable } from 'rxjs';
import * as fromDecidedSurplusSelectors from '../../../../store/selectors/poa-decided-surplus.selectors';
import {
  UpdatePoaDecidedSurplus,
  ResetDecidedSurplusNotification
} from '../../../../store/actions/poa-decided-surplus.actions';
import * as fromAnimations from '../../../../shared/animations/index';

import { UpdatingWisnHRHData } from '../../../../store/actions/wisn-hrh-data.actions';
import * as fromWisnReportSelectors from '../../../../store/selectors/wisn-report.selectors';
import { currentUserRolePermissions } from 'src/app/store/selectors/general-configuration.selector';
import { WisnPoaPageService } from 'src/app/shared/services/wisn-poa-page.service';
@Component({
  selector: 'app-decide-surplus',
  templateUrl: './decide-surplus.component.html',
  styleUrls: ['./decide-surplus.component.css'],
  animations: [
    trigger('fadeInOut', [
      transition(':enter', [
        // :enter is alias to 'void => *'
        style({ opacity: 0 }),
        animate(500, style({ opacity: 1 }))
      ]),
      transition(':leave', [
        // :leave is alias to '* => void'
        animate(500, style({ opacity: 0 }))
      ])
    ])
  ]
})
export class DecideSurplusComponent implements OnInit {
  toggle: boolean = true;
  decidedSurplusData$: Observable<any>;
  decidedSurplusLoaded$: Observable<any>;
  decidedSurplusHeader$: Observable<any>;
  decidedSurplusSubTableHeader$: Observable<any>;
  decidedSurplusSummaryData$: Observable<any>;
  wisnReportProgress$: Observable<any>;
  decidedSurplusNotification$: Observable<any>;

  p = 1;
  itemsPerPage = 10;
  userRoleAction: any;
  constructor(
    private store: Store<AppState>,
    private wisnPoaPages: WisnPoaPageService
  ) {
    this.decidedSurplusLoaded$ = this.store.select(
      fromDecidedSurplusSelectors.getDecidedSurplusLoadedState
    );
    this.decidedSurplusData$ = this.store.select(
      fromDecidedSurplusSelectors.getAllPoaDecidedSurplusDataState
    );
    this.decidedSurplusNotification$ = this.store.select(
      fromDecidedSurplusSelectors.getDecidedSurplusNotificationState
    );
    this.decidedSurplusHeader$ = this.store.select(
      fromDecidedSurplusSelectors.getDecidedSurplusTableHeader
    );
    this.decidedSurplusSubTableHeader$ = this.store.select(
      fromDecidedSurplusSelectors.getSubTableHeadingsState
    );
    this.decidedSurplusSummaryData$ = this.store.select(
      fromDecidedSurplusSelectors.getDecidedSurplusSummaryState
    );
    this.wisnReportProgress$ = this.store.select(
      fromWisnReportSelectors.getWisnReportProgresLoaderState
    );

    this.userRoleAction = wisnPoaPages.getUserPermissionAction('decidesurplus');
  }

  ngOnInit() {}
  onResetNotification(emptyNotificationMessage) {
    this.store.dispatch(
      new ResetDecidedSurplusNotification({
        notification: emptyNotificationMessage
      })
    );
  }
  onToggleSubTable(e, rowToggleStatus, tableId) {
    if (this.toggle) {
      document
        .getElementById(tableId)
        .setAttribute('style', 'display: table-cell !important');
      this.toggle = false;
    } else {
      document
        .getElementById(tableId)
        .setAttribute('style', 'display:none !important');
      this.toggle = true;
    }
  }

  onUpdatePageSize(e) {
    this.itemsPerPage = e;
  }

  onCurrentPageUpdate(e) {
    this.p = e;
  }

  decidingSurplusChange(dataObject, wisnSurplusData) {
    // console.log(JSON.stringify(dataObject));
    // console.log(JSON.stringify(wisnSurplusData));
    if (dataObject.decidedRedistribution <= dataObject.wisnSurplus) {
      const payload = { rowData: wisnSurplusData[2], dataObject: dataObject };
      this.store.dispatch(new UpdatePoaDecidedSurplus(payload));
    } else {
      alert('Decided redistribution can not exceed WISN surplus');
    }
    // const data = { rowData: rowData, index: rowData.index, changedData: {
    //     orgunitId: this.orgunit.id,
    //     dataElementId: this.dataElement.id,
    //     value: this.value,
    //     domElementId: elementUid
    //   } };
    // this.store.dispatch(new UpdatingWisnHRHData(data));
  }
}
