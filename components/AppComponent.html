import { Component, OnInit } from '@angular/core';
import { Store } from '@ngrx/store';
import { AppState, getWisnReportState } from './store/reducers';
import { LoadCurrentUserAction } from './store/actions/current-user.actions';
import { Observable } from 'rxjs/Observable';

  getCurrentGlobalFilter,
  getCurrentOrgUnitFilter

  animate,

  style,
  transition,
  trigger

import { CheckSetupEnvironments } from './store/actions/setup-environment.actions';
import { getWisnHmisDataState } from './store/selectors/wisn-hmis.selectors';
@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.css'],
  animations: [
    trigger('open', [
      state(
        'in',
        style({
          opacity: 1
        })
      ),
      transition('void => *', [
        style({
          opacity: 0
        }),
        animate(700)
      ]),
      transition('* => void', [
        animate(300),
        style({
          opacity: 0
        })
      ])
    ])

export class AppComponent implements OnInit {
  globalFilter$: Observable<GlobalFilter>;
  orgUnitFilter$: Observable<any>;
  currentPage$: Observable<string>;
  appPages$: Observable<any>;
  showOrgUnitFilter: boolean;
  showPeriodFilter: boolean;
  showMobileMenu: boolean;
  showSubMenu: boolean;
  showMobileSubMenu: boolean;

  selectorTester$: Observable<any>;
  title = 'WISN + POA';
  constructor(
    private store: Store<AppState>,
    private pagesService: WisnPoaPageService
  ) {
    this.showMobileMenu = false;
    this.showSubMenu = false;
    this.showMobileSubMenu = false;
    store.dispatch(new CheckSetupEnvironments({}));
    this.selectorTester$ = store.select(getWisnHmisDataState);
    this.appPages$ = store.select(getAppPages);
    this.globalFilter$ = store.select(getCurrentGlobalFilter);
    this.orgUnitFilter$ = store.select(getCurrentOrgUnitFilter);
    this.currentPage$ = store.select(getCurrentPageState);
  }

  ngOnInit() {
    // disabled this function to until futher recheck or undefined error
    // setTimeout(e => {
    //   this.store.dispatch(new WisnDataUpdateInBackground());
    // }, 180000);
  }

  toggleOrgUnitFilter(e) {
    e.stopPropagation();
    this.showOrgUnitFilter = !this.showOrgUnitFilter;
  }

  togglePeriodFilter(e) {
    e.stopPropagation();
    this.showPeriodFilter = !this.showPeriodFilter;
  }

  updateOrgUnit(orgUnitModel) {
    this.showOrgUnitFilter = false;
    if (orgUnitModel.items && orgUnitModel.items[0]) {
      this.store.dispatch(new GlobalFilterUpdateAction(orgUnitModel));
    }
  }

  updatePeriod(periodModel) {
    this.showPeriodFilter = false;

    if (periodModel.items && periodModel.items[0]) {
      this.store.dispatch(new GlobalFilterUpdateAction(periodModel));
    }
  }

  toggleMobileMenu(e) {
    e.stopPropagation();
    this.showMobileMenu = !this.showMobileMenu;
  }

  toggleMobileSubMenu(e) {
    if (e) {
      e.stopPropagation();
    }
    this.showMobileSubMenu = !this.showMobileSubMenu;
  }
  toggleSubMenu(e) {
    if (e) {
      e.stopPropagation();
    }
    this.showSubMenu = !this.showSubMenu;
  }
  onRouteNavigation(e, pageLink) {
    e.stopPropagation();
    this.store.dispatch(new Go({ path: [pageLink] }));
    this.showMobileMenu = false;
  }
}
