import { Component, OnInit } from '@angular/core';
import { Store } from '@ngrx/store';
import { AppState } from '../../../../store/reducers/index';
import * as fromPoaCostAssumptionsSelectors from '../../../../store/selectors/poa-cost-assumptions.selectors';
import { Observable } from 'rxjs';
import { FilterByPipe, TailPipe } from 'ngx-pipes';
import * as fromGlobalConfigurationsSelectors from '../../../../store/selectors/general-configuration.selector';
import {
  UpdatingPoaCostAssumptionsProgram,
  ResetCostAssumptionsNotification,
  CostAssumptionsLoadMaxHRHFromPRIlist
} from '../../../../store/actions/poa-cost-assumption.actions';
import {
  style,
  state,
  animate,
  transition,
  trigger
} from '@angular/animations';
import * as fromPoaPriSelectors from '../../../../store/selectors/poa-pri.selectors';
@Component({
  selector: 'app-cost-assumptions',
  templateUrl: './cost-assumptions.component.html',
  styleUrls: ['./cost-assumptions.component.css'],
  animations: [
    trigger('fadeInOut', [

        // :enter is alias to 'void => *'

        // :leave is alias to '* => void'

export class CostAssumptionsComponent implements OnInit {
  poaCostAssumptionsLoaded$: Observable<boolean>;
  poaSalaryCosts$: Observable<any>;
  poaAllowanceCost$: Observable<any>;
  poaCostMaxBudget$: Observable<any>;
  poaCostMaxNumber$: Observable<any>;
  totalPoaMaxHRHNumber$: Observable<any>;
  costAssumptionsNotification$: Observable<any>;
  costAssumptionsProgress$: Observable<any>;
  isSalaryFormDisabled$: Observable<boolean>;
  isAllowanceFormDisabled$: Observable<boolean>;
  isConstraintsFormDisabled$: Observable<boolean>;
  remainingBudgetValue$: Observable<any>;
  priListLoadedState$: Observable<any>;
  initialLoadedMaxHRHbyPRI = false;

  constructor(private store: Store<AppState>) {
    this.poaCostAssumptionsLoaded$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsLoadedState
    );
    this.poaSalaryCosts$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsSalaryCostState
    );
    this.poaAllowanceCost$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsAllowanceCostState
    );

    this.poaCostMaxBudget$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsMaxBudgetState
    );

    this.poaCostMaxNumber$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsMaxNumberState
    );

    this.isSalaryFormDisabled$ = this.store.select(
      fromPoaCostAssumptionsSelectors.canEditSalaryData
    );

    this.isAllowanceFormDisabled$ = this.store.select(
      fromPoaCostAssumptionsSelectors.canEditAllowanceData
    );

    this.isConstraintsFormDisabled$ = this.store.select(
      fromPoaCostAssumptionsSelectors.canEditConstraintsData
    );
    this.costAssumptionsNotification$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsNotificationState
    );

    this.costAssumptionsProgress$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsProgresState
    );
    this.totalPoaMaxHRHNumber$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsMaxNumberValues
    );
    this.remainingBudgetValue$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getRemainingBudget);
    this.priListLoadedState$ = this.store.select(
      fromPoaPriSelectors.getPoaPriLoadedState
    );
  }

  onResetNotification(emptyNotificationMessage) {
    this.store.dispatch(
      new ResetCostAssumptionsNotification({
        notification: emptyNotificationMessage
      })
    );
  }
  ngOnInit() {}

  fetchMaxHRHBudgetFromPRIlist() {
    this.store.dispatch(new CostAssumptionsLoadMaxHRHFromPRIlist());
  }
  formatInputAsCurrency(e) {
    if (e) {
      e.stopPropagation();
    }
    const amount = e ? e.target.value.trim() : 0;
    return amount.toFixed(2).replace(/\d(?=(\d{3})+\.)/g, '$&,');
  }

  programDataValueChanges(e, rowData) {
    const data = { rowData: rowData, index: rowData.index, changedData: e };
    this.store.dispatch(new UpdatingPoaCostAssumptionsProgram(data));
  }

  enforceCostAssumptionsConsistency(value1, value2, value3) {
    // console.log(value1, value2, value3);
  }

  displayBudgetDivision(maxHRHValue, maxBudget, remainingBudgetValue) {
    const maxSalaryCost = (parseFloat(maxBudget) / parseFloat(maxHRHValue)) ? (parseFloat(maxBudget) / parseFloat(maxHRHValue)) : 0;
    // console.log(maxSalaryCost);
    const maxHRH: any = (parseFloat(remainingBudgetValue) / maxSalaryCost) ? (parseFloat(remainingBudgetValue) / maxSalaryCost) : 0;
    return parseInt(maxHRH, 10) ? parseInt(maxHRH, 10) : 0;
  }
}
