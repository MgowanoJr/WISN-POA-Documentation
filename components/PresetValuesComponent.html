import { Component, OnInit } from '@angular/core';
import { Observable } from 'rxjs';
import { Store } from '@ngrx/store';
import { AppState } from '../../../../store/reducers';
import * as _ from 'lodash';
import * as fromWisnPresetValuesSelectors from '../../../../store/selectors/preset-values.selectors';
import * as fromGlobalFiltersSelectors from '../../../../store/selectors/global-filter.selectors';
import * as fromGlobalFiltersActions from '../../../../store/actions/global-filter.actions';
import * as fromGeneralConfigurationsSelectors from '../../../../store/selectors/general-configuration.selector';
import * as fromRouterSelectors from '../../../../store/selectors/router.selectors';
import {
  LoadWisnPresetValues,
  UploadWisnPresetValue,
  UpsertWisnPresetValue,
  ResetWisnPresetValueNotification,
  ActivateOrgUnitForEntry,
  PreLoadWisnPresetValue,
  UpsertWisnPresetValues
} from '../../../../store/actions/wisn-preset-value.actions';
import {
  style,
  state,
  animate,
  transition,
  trigger
} from '@angular/animations';
import * as fromRouterActions from '../../../../store/actions/router.actions';
import * as fromUtilHelpers from '../../../../shared/helpers/util-functions';
import { WisnAnalyticsService } from 'src/app/shared/services/wisn-analytics.service';
@Component({
  selector: 'app-preset-values',
  templateUrl: './preset-values.component.html',
  styleUrls: ['./preset-values.component.css'],
  animations: [
    trigger('fadeInOut', [

export class PresetValuesComponent implements OnInit {
  wisnPresetValuesLoaded$: Observable<boolean>;
  wisnPresetValuesDataValues$: Observable<any>;
  wisnPresetValuesFormDesign$: Observable<any>;
  wisnPresetValuesSectionForm$: Observable<any>;
  wisnPresetValuesCurrentSelection$: Observable<any>;
  wisnPresetValuesPagesize$: Observable<any>;
  wisnPresetValuesPagenumber$: Observable<any>;
  wisnPresetValuesNotification$: Observable<any>;
  presetValueProgress$: Observable<any>;
  presetValueLoading$: Observable<any>;
  globalFilter$: Observable<any>;
  globalFilterLoaded$: Observable<boolean>;
  dataEntryLevel$: Observable<any>;
  isDataEntryLevel$: Observable<any>;
  getPresetFacilityTypes$: Observable<any>;
  currentFacilityType$: Observable<any>;

  p = 1;
  itemsPerPage = 10;
  searchText: '';
  selectedOrgUnitId: string;
  selectedOrgUnitName: string;
  selectedOrgUnit: any;
  loadingInformation = 'Loading preset data...';
  hideCadreFunctionInterval: any;
  constructor(
    private store: Store<AppState>,
    private wisnAnalyticsService: WisnAnalyticsService
  ) {
    this.getPresetFacilityTypes$ = this.store.select(
      fromWisnPresetValuesSelectors.getPresetFacilityTypes
    );
    this.currentFacilityType$ = this.store.select(
      fromWisnPresetValuesSelectors.getCurrentFacilityType
    );
    this.wisnPresetValuesLoaded$ = this.store.select(
      fromWisnPresetValuesSelectors.getWisnPresetValuesLoadedState
    );
    this.wisnPresetValuesDataValues$ = this.store.select(
      fromWisnPresetValuesSelectors.getAllWisnPresetValuesState
    );
    this.globalFilter$ = this.store.select(
      fromGlobalFiltersSelectors.getCurrentGlobalFilter
    );
    this.globalFilterLoaded$ = this.store.select(
      fromGlobalFiltersSelectors.getGlobalFilterLoadedState
    );

    this.dataEntryLevel$ = this.store.select(
      fromGeneralConfigurationsSelectors.getPresetValueEntryLevel
    );

    this.wisnPresetValuesFormDesign$ = this.store.select(
      fromWisnPresetValuesSelectors.getWisnPresetValuesFormDesignState
    );

    this.wisnPresetValuesSectionForm$ = this.store.select(
      fromWisnPresetValuesSelectors.getWisnPresetValuesSections
    );

    this.wisnPresetValuesCurrentSelection$ = this.store.select(
      fromWisnPresetValuesSelectors.getCurrentlySelectedFacility
    );

    this.wisnPresetValuesPagesize$ = this.store.select(
      fromWisnPresetValuesSelectors.getCurrentSelectionPagesize
    );

    this.wisnPresetValuesPagenumber$ = this.store.select(
      fromWisnPresetValuesSelectors.getCurrentSelectionPagenumber
    );

    this.wisnPresetValuesNotification$ = this.store.select(
      fromWisnPresetValuesSelectors.getPresetValuesNotificationState
    );

    this.isDataEntryLevel$ = this.store.select(
      fromWisnPresetValuesSelectors.canEditWisnPresetValues
    );
    this.presetValueProgress$ = this.store.select(
      fromWisnPresetValuesSelectors.getPresetValueProgressLoaderState
    );

    this.presetValueLoading$ = this.store.select(
      fromWisnPresetValuesSelectors.getWisnPresetValueLoadingState
    );

    this.store
      .select(fromGlobalFiltersSelectors.getCurrentGlobalFilter)
      .subscribe(globalFilter => {
        this.selectedOrgUnit = globalFilter ? globalFilter.ou : [];
      });

    this.store
      .select(fromWisnPresetValuesSelectors.getPresetFacilityTypes)
      .subscribe(preseFacilityTypes => {
        preseFacilityTypes.forEach(facility => {
          if (facility.status === 'active' && facility.hasLessCadres) {
            this.hideCadres(facility);
          }
        });
      });
  }

  ngOnInit() {}

  onSearchOrgUnit(e) {
    if (e) {
      e.stopPropagation();
    }
    this.searchText = e ? e.target.value.trim() : this.searchText;
  }

  onUpdatePageSize(e) {
    this.store.dispatch(
      new fromRouterActions.Go({
        path: [],
        query: {
          pagesize: e
        },
        extras: {
          preserveQueryParams: true,
          queryParamsHandling: 'merge'
        }
      })
    );
  }

  onCurrentPageUpdate(e) {
    this.store.dispatch(
      new fromRouterActions.Go({
        path: [],
        query: {
          page: e
        },
        extras: {
          preserveQueryParams: true,
          queryParamsHandling: 'merge'
        }
      })
    );
  }

  changesOnPresetValue(dataValueObject) {
    this.store.dispatch(new UpsertWisnPresetValues(dataValueObject));
  }

  toggleFacilityType(presetFacility) {
    clearInterval(this.hideCadreFunctionInterval);
    this.store.dispatch(new PreLoadWisnPresetValue(presetFacility));
    if (presetFacility.hasLessCadres) {
      this.hideCadres(presetFacility);
    }

    setTimeout(() => {
      this.store.dispatch(new ActivateOrgUnitForEntry(presetFacility));
    }, 1000);
  }

  onResetNotification(emptyNotificationMessage) {
    this.store.dispatch(
      new ResetWisnPresetValueNotification({
        notification: emptyNotificationMessage
      })
    );
  }

  savePresetValues() {
    this.loadingInformation = 'Uploading preset data...';
    this.store.dispatch(new UploadWisnPresetValue());
  }

  hideCadres(preseFacilityType) {
    this.hideCadreFunctionInterval = setInterval(() => {
      // handling cadre views
      // try {
      //   const elements = document.querySelectorAll(preseFacilityType.initCssClassQuery);
      // Array.prototype.forEach.call(elements, function(el) {
      //   if (preseFacilityType.hasLessCadres) { // checks whether presetFacilityType should show less cadres
      //     el.classList.add(preseFacilityType.cssClassToAdd);
      //     el.classList.remove(preseFacilityType.cssClassToRemove);
      //   } else {
      //     el.classList.remove(preseFacilityType.cssClassToRemove);
      //     el.classList.add(preseFacilityType.cssClassToAdd);
      //   }
      // });
      // } catch (e) {}
      const elements = document.querySelectorAll('.cadreViewaLL');
      Array.prototype.forEach.call(elements, function(el) {
        if (preseFacilityType.hasLessCadres) {
          // checks whether its a dispensary
          el.classList.add('cadreViewHide');
          el.classList.remove('cadreViewaLL');
        } else {
          el.classList.remove('cadreViewHide');
          el.classList.add('cadreViewaLL');
        }
      });
    }, 2000);
  }
}
