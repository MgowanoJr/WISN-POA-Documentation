import { Component, OnInit } from '@angular/core';
import { Store } from '@ngrx/store';
import { Observable } from 'rxjs';
import * as _ from 'lodash';
import { AppState } from '../../../../store/reducers';

import {
  style,
  state,
  animate,
  transition,
  trigger
} from '@angular/animations';

import * as fromWisnHrhSelectors from '../../../../store/selectors/wisn-hrh.selectors';
import * as fromGlobalConfigurationsSelectors from '../../../../store/selectors/general-configuration.selector';
import * as fromGlobalFiltersSelectors from '../../../../store/selectors/global-filter.selectors';

import * as fromGlobalFiltersActions from '../../../../store/actions/global-filter.actions';
import * as fromRouterActions from '../../../../store/actions/router.actions';

import { UpdateWisnHmisData } from '../../../../store/actions/wisn-hmis-data.actions';
import {
  UpdatingWisnHRHData,
  DownloadingWisnHrhDatas,
  UploadingWisnHrhDatas,
  ResetWisnHrhFormNotification
} from '../../../../store/actions/wisn-hrh-data.actions';

import { WisnPoaPageService } from 'src/app/shared/services/wisn-poa-page.service';
@Component({
  selector: 'app-hrh',
  templateUrl: './hrh.component.html',
  styleUrls: ['./hrh.component.css'],
  animations: [fromAnimations.fadeInOut]
})
export class HrhComponent implements OnInit {
  hrhDataLoaded$: Observable<boolean>;
  hrhData$: Observable<any>;
  hrhDataElements$: Observable<any>;
  globalFilterSelections$: Observable<any>;
  dataEntryLevel$: Observable<any>;
  isFormDisabled$: Observable<boolean>;
  hrhNotification$: Observable<any>;
  hrhProgress$: Observable<any>;

  searchText: '';
  p = 1;
  itemsPerPage = 10;
  showDownloadDropdown = false;
  dropDownMenuX: any;
  dropDownMenuY: any;
  userRoleAction: any;

  constructor(private store: Store<AppState>, private wisnPoaPages: WisnPoaPageService) {
    this.hrhDataLoaded$ = this.store.select(
      fromWisnHrhSelectors.getWisnHrhDataLoadedState
    );
    this.hrhData$ = this.store.select(
      fromWisnHrhSelectors.getWisnHrhDataEntities
    );
    this.hrhDataElements$ = this.store.select(
      fromWisnHrhSelectors.getWisnHrhDataElements
    );
    this.globalFilterSelections$ = this.store.select(
      fromGlobalFiltersSelectors.getCurrentGlobalFilter
    );
    this.dataEntryLevel$ = this.store.select(
      fromGlobalConfigurationsSelectors.getDataEntryLevelState
    );
    this.isFormDisabled$ = this.store.select(
      fromWisnHrhSelectors.canEditHrhData
    );
    this.hrhNotification$ = this.store.select(
      fromWisnHrhSelectors.getHrhNotificationState
    );
    this.hrhProgress$ = this.store.select(
      fromWisnHrhSelectors.getWisnHrhProgressLoader
    );

    this.store.select(
      fromWisnHrhSelectors.getWisnHrhDataLoadedState
    ).subscribe(response => {
      this.userRoleAction = wisnPoaPages.getUserPermissionAction('hrh');
    });
  }

  ngOnInit() {}

  onResetNotification(emptyNotificationMessage) {
    this.store.dispatch(
      new ResetWisnHrhFormNotification({
        notification: emptyNotificationMessage
      })
    );
  }

  onToggleDropdown(e) {
    if (e) {
      e.stopPropagation();
    }
    e.cancelBubble = true;
    this.dropDownMenuX = 0;
    this.dropDownMenuY = 49;
    this.showDownloadDropdown = !this.showDownloadDropdown;
    return false;
  }

  programDataValueChanges(e, rowData) {
    const data = {
      rowData: rowData,
      index: rowData.index,
      changedData: e,
      isFromHRHpageView: true
    };
    this.store.dispatch(new UpdatingWisnHRHData(data));
  }

  onUpdatePageSize(e) {
    this.itemsPerPage = e;
  }

  onCurrentPageUpdate(e) {
    this.p = e;
  }

  onSearchReport(e) {
    if (e) {
      e.stopPropagation();
    }
    this.searchText = e ? e.target.value.trim() : this.searchText;
  }

  downloadTable(downloadFormat) {
    this.store.dispatch(new DownloadingWisnHrhDatas());
  }

  chooseFile() {
    // $("input[type=file]").click();
    const element: HTMLElement = document.querySelector(
      'input[type="file"]'
    ) as HTMLElement;
    element.click();
  }

  receiveFile(csv: any) {
    // just capture the file and manipulate it later
    const fileReaded = csv.target.files[0];
    const csvRows = [];
    // new handleFileSelect(event);
    // this.choosedFileName = csv.target.files[0].name;
    // this.uploadFile();
    // make an action that will handle the import process
    const reader: FileReader = new FileReader();
    reader.readAsText(fileReaded);
    reader.onload = e => {
      const csvData: any = reader.result;
      const allTextLines = csvData.split(/\r|\n|\r/);
      const headers = allTextLines[0].split(',');
      for (let i = 0; i < allTextLines.length; i++) {
        // split content based on comma
        const data = allTextLines[i].split(',');
        if (data.length === headers.length) {
          const tarr = [];
          for (let j = 0; j < headers.length; j++) {
            tarr.push(data[j].replace('"', '').replace('"', ''));
            //  tarr.push(data[j].replace("\"",""));
          }
          csvRows.push(tarr);
        }
      }
      this.store.dispatch(new UploadingWisnHrhDatas(csvRows));
    };
  }

  onDrillDown(e, orgUnitId, parentOrgUnit) {
    e.stopPropagation();
    const orgUnit = _.find(parentOrgUnit ? parentOrgUnit.children : [], [
      'id',
      orgUnitId
    ]);
    if (orgUnit && orgUnit.children && orgUnit.children.length > 0) {
      const orgUnitModel = {
        items: [orgUnit],
        name: 'ou',
        value: orgUnit.id + ';LEVEL-' + orgUnit.level
      };
      this.store.dispatch(
        new fromGlobalFiltersActions.GlobalFilterUpdateAction(orgUnitModel)
      );
    }
    // else {
    //   this.store.dispatch(
    //     new fromRouterActions.Go({
    //       path: ['/wisn/presetvalues'],
    //       query: {
    //         selection: orgUnitId,
    //         pagesize: this.itemsPerPage,
    //         page: this.p
    //       },
    //       extras: {
    //         preserveQueryParams: true,
    //         queryParamsHandling: 'merge'
    //       }
    //     })
    //   );
    // }
  }
}
