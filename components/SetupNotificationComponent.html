import { Component, OnInit } from '@angular/core';
import { Store } from '@ngrx/store';
import { AppState } from 'src/app/store';
import { InstallSetupEnvironments } from 'src/app/store/actions/setup-environment.actions';

@Component({
  selector: 'app-setup-notification',
  templateUrl: './setup-notification.component.html',
  styleUrls: ['./setup-notification.component.css']
})
export class SetupNotificationComponent implements OnInit {
  NotificationHeading: string;
  NotificationMessage: string;
  constructor(private store: Store<AppState>) {
    this.NotificationHeading = 'Setting up WISN+POA Configurations...';
    this.NotificationMessage =
      'This DHIS2 instance doesnt have WISN+POA configurations yet, but you can set them up on a single click';

    store.dispatch(new InstallSetupEnvironments());
  }

  ngOnInit() {}
  // lgZ6HfZaj3f
}
