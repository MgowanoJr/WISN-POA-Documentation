import { Component, OnInit } from '@angular/core';
import { Store } from '@ngrx/store';
import { AppState } from '../../../../store/reducers/index';
import * as fromRouterActions from '../../../../store/actions/router.actions';
import {Observable} from 'rxjs';
import {getIsConfigured, getLoadingState} from '../../../../store/selectors/setup-environment.selectors';

@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.css']
})
export class HomeComponent implements OnInit {
  isConfigured$: Observable<any>;
  loading$: Observable<any>;

  constructor(private store: Store<AppState>) {
    this.isConfigured$ = this.store.select(getIsConfigured);
    this.loading$ = this.store.select(getLoadingState);
  }

  ngOnInit() {}

  onTogglePage(e, pageName) {
    if (e) {
      e.stopPropagation();
    }
    console.log(pageName);
    if (pageName) {
      this.store.dispatch(
        new fromRouterActions.Go({
          path: [pageName]
        })
      );
    }
  }
}
