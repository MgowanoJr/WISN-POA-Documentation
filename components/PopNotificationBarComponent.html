import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-pop-notification-bar',
  templateUrl: './pop-notification-bar.component.html',
  styleUrls: ['./pop-notification-bar.component.css']
})
export class PopNotificationBarComponent implements OnInit {
  @Input()
  notification;

  @Input()
  timeout;

  @Output()
  resetNotification: EventEmitter<any> = new EventEmitter<any>();
  constructor() {}

  ngOnInit() {
    setTimeout(() => {
      this.resetNotification.emit({
        notification: null
      });
    }, this.timeout);
  }
}
