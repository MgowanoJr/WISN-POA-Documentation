import { AppState } from '../../../../store';
import { Observable } from 'rxjs';
import { Store, select } from '@ngrx/store';
import {
  ApplyAutoConstraintsOnPRIstate,
  DownloadPoaPriLists,
  UpdatingPoaPriList,
  ResetPriListNotification,
  LoadPoaPriLists,
  SavingPoaPRIStateDraft,
  ResetPoaPriList
} from '../../../../store/actions/poa-pri-list.actions';
import * as fromAnimations from '../../../../shared/animations/index';

import { UploadWisnReport } from '../../../../store/actions/wisn-report.actions';
import * as fromPoaCostAssumptionsSelectors from '../../../../store/selectors/poa-cost-assumptions.selectors';
import * as fromWisnReportSelectors from '../../../../store/selectors/wisn-report.selectors';
import { currentUserRolePermissions } from 'src/app/store/selectors/general-configuration.selector';
import { WisnPoaPageService } from 'src/app/shared/services/wisn-poa-page.service';

@Component({
  selector: 'app-distribute-hrh',
  templateUrl: './distribute-hrh.component.html',
  styleUrls: ['./distribute-hrh.component.css'],
  animations: [fromAnimations.fadeInOut]
})
export class DistributeHrhComponent implements OnInit, OnDestroy {
  poaLoaded$: Observable<boolean>;
  poaPriData$: Observable<any>;
  isDistributeHRHDisabled$: Observable<boolean>;
  poaSalaryCosts$: Observable<any>;
  poaDistributionSummary$: Observable<any>;
  poaDistributeNotification$: Observable<any>;
  wisnReportProgress$: Observable<any>;
  showDistributionSummary = false;
  userRoleAction: any;
  priStateEdited = false;

  orgUnitSearchText: '';
  cadreSearchText: '';
  searchText: '';
  p = 1;
  itemsPerPage = 100;

  constructor(private store: Store<AppState>, private wisnPoaPages: WisnPoaPageService) {
    this.poaLoaded$ = this.store.select(
      fromPoaPriSelectors.getPoaPriLoadedState
    );

    this.isDistributeHRHDisabled$ = this.store.select(
      fromPoaPriSelectors.canDistributeHRH
    );

    this.poaPriData$ = this.store.select(
      fromPoaPriSelectors.getPoaNewAssignmentsEntities
    );

    this.poaDistributionSummary$ = this.store.select(
      fromPoaPriSelectors.getDistributionSummaryState
    );

    this.poaSalaryCosts$ = this.store.select(
      fromPoaCostAssumptionsSelectors.getPoaCostAssumptionsSalaryCostState
    );
    this.wisnReportProgress$ = this.store.select(
      fromWisnReportSelectors.getWisnReportProgresLoaderState
    );

    this.poaDistributeNotification$ = this.store.select(
      fromPoaPriSelectors.getPoaPriNotificationState
    );

    this.wisnReportProgress$ = this.store.select(
      fromWisnReportSelectors.getWisnReportProgresLoaderState
    );

    this.userRoleAction = wisnPoaPages.getUserPermissionAction('distributenew');
  }

  onResetNotification(emptyNotificationMessage) {
    this.store.dispatch(
      new ResetPriListNotification({
        notification: emptyNotificationMessage
      })
    );
  }

  applyAutoConstraints() {
    this.store.dispatch(new ApplyAutoConstraintsOnPRIstate());
    this.priStateEdited = true;
  }

  changeRowColor(AssignmentDesignstion) {
    let styleObject = {};
    if (AssignmentDesignstion === 'Unassigned') {
      styleObject = { 'background-color': '#ffffff' };
    } else if (AssignmentDesignstion === 'Redistributed') {
      styleObject = { 'background-color': '#f0fff0' };
    } else if (AssignmentDesignstion === 'New') {
      styleObject = { 'background-color': '#cdeffe' };
    }
    return styleObject;
  }

  assignmentDesignationChange(e, rowData) {
    if (e) {
      e.stopPropagation();
    }
    const changedValue = e ? e.target.value.trim() : 'Unassigned';
    const changedData = {
      changedAssignmentDesignation: changedValue,
      rowData: rowData,
      currentView: 'distribute'
    };
    this.store.dispatch(new UpdatingPoaPriList(changedData));
    this.priStateEdited = true;
  }

  downloadTable(format) {
    if (format === 'CSV') {
      this.store.dispatch(new DownloadPoaPriLists());
    } else {
    }
  }

  chooseFile() {
    // $("input[type=file]").click();
    const element: HTMLElement = document.querySelector(
      'input[type="file"]'
    ) as HTMLElement;
    element.click();
  }

  receiveFile(csv: any) {
    // just capture the file and manipulate it later
    const fileReaded = csv.target.files[0];
    const csvRows = [];
    // new handleFileSelect(event);
    // this.choosedFileName = csv.target.files[0].name;
    // this.uploadFile();
    // make an action that will handle the import process
    const reader: FileReader = new FileReader();
    reader.readAsText(fileReaded);
    reader.onload = e => {
      const csvData: any = reader.result;
      const allTextLines = csvData.split(/\r|\n|\r/);
      const headers = allTextLines[0].split(',');
      for (let i = 0; i < allTextLines.length; i++) {
        // split content based on comma
        const data = allTextLines[i].split(',');
        if (data.length === headers.length) {
          const tarr = [];
          for (let j = 0; j < headers.length; j++) {
            tarr.push(data[j].replace('"', '').replace('"', ''));
            //  tarr.push(data[j].replace("\"",""));
          }
          csvRows.push(tarr);
        }
      }
      this.store.dispatch(new UploadWisnReport(csvRows));
    };
  }

  ngOnInit() {}
  ngOnDestroy() {
    if (this.priStateEdited) {
      this.store.dispatch(new SavingPoaPRIStateDraft());
    }
  }
  onUpdatePageSize(e) {
    this.itemsPerPage = e;
  }

  onCurrentPageUpdate(e) {
    this.p = e;
  }
  onSearchReport(e) {
    if (e) {
      e.stopPropagation();
    }
    this.searchText = e ? e.target.value.trim() : this.searchText;
  }

  onShowDistribution(e) {
    if (e) {
      e.stopPropagation();
    }
    this.showDistributionSummary = !this.showDistributionSummary;
  }

  resetPRIlist() {
    // this function is to reset POA PRI from wisn report data
    this.store.dispatch(new LoadPoaPriLists()); // New means view on Distribute
    // this.store.dispatch( new ResetPoaPriList('New')); // New means view on Distribute
    this.priStateEdited = false;
  }

  onSearchByOrgUnit(e) {
    if (e) {
      e.stopPropagation();
    }
    this.orgUnitSearchText = e ? e.target.value.trim() : this.orgUnitSearchText;
  }

  onSearchByCadre(e) {
    if (e) {
      e.stopPropagation();
    }
    this.cadreSearchText = e ? e.target.value.trim() : this.cadreSearchText;
  }
}
