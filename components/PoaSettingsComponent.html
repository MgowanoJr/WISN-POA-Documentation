import { Component, OnInit } from '@angular/core';
import { AppState } from '../../../../store/reducers';
import { Store } from '@ngrx/store';
import { getCurrentPage } from '../../../../store/selectors/wisn-poa-pages.selectors';
import * as fromRouterSelectors from '../../../../store/selectors/router.selectors';
import * as fromGlobalFilterSelectors from '../../../../store/selectors/global-filter.selectors';

import { Observable } from 'rxjs';
@Component({
  selector: 'app-poa-settings',
  templateUrl: './poa-settings.component.html',
  styleUrls: ['./poa-settings.component.css']
})
export class PoaSettingsComponent implements OnInit {
  poaSettingsPageObject$: Observable<any>;
  currentPageState$: Observable<any>;
  globalFilters$: Observable<any>;
  constructor(private store: Store<AppState>) {
    this.poaSettingsPageObject$ = this.store.select(getCurrentPage);
    this.currentPageState$ = this.store.select(
      fromRouterSelectors.getCurrentPageState
    );
    this.globalFilters$ = this.store.select(
      fromGlobalFilterSelectors.getCurrentGlobalFilter
    );
  }
  ngOnInit() {}
}
