import {
  Component,
  OnInit,
  Input,
  ViewChild,
  ElementRef,
  EventEmitter,
  Output
} from "@angular/core";
import { AppState } from "../../../../store/reducers";
import { Store } from "@ngrx/store";
import * as fromWisnReportSelectors from "../../../../store/selectors/wisn-report.selectors";
import { Observable } from "rxjs";
import { UpdateWisnReportFromDataValues } from "../../../../store/actions/wisn-report.actions";
import * as fromGlobalFiltersActions from "../../../../store/actions/global-filter.actions";
import { FilterByPipe } from "ngx-pipes";

import * as _ from "lodash";
import { VisualizationExportService } from "../../../../shared/services/visualizationExportService.service";

@Component({
  selector: "app-wisn-report-table",
  templateUrl: "./wisn-report-table.component.html",
  styleUrls: ["./wisn-report-table.component.css"],
  providers: [FilterByPipe]
})
export class WisnReportTableComponent implements OnInit {
  @ViewChild("wisnReport")
  table: ElementRef;

  @Input()
  wisnReportDetails;

  @Input()
  globalFilterOU;
  @Input()
  wisnReportProgress;

  @Input()
  wisnReportData;

  @Input()
  allWisnReportData;

  @Input()
  filterCriteria;

  p: number = 1;
  itemsPerPage: number = 10;
  maxSize: number;
  pagingIntervals: any;
  searchText = "";
  orderByValue: string;
  showDownloadDropdown = false;
  dropDownMenuX: any;
  dropDownMenuY: any;
  wisnReportHeader$: Observable<any>;
  showPrintDropdown: boolean = false;
  @Output() triggerDownLoadReport: EventEmitter<any> = new EventEmitter<any>();

  constructor(
    private store: Store<AppState>,
    private tableExport: VisualizationExportService,
    private filterByPipe: FilterByPipe
  ) {
    this.wisnReportHeader$ = this.store.select(
      fromWisnReportSelectors.getWisnReportHeaderWithCadre
    );
  }

  ngOnInit() {
    this.resetSortingCriteria();
  }

  resetSortingCriteria() {
    this.orderByValue = this.allWisnReportData ? "0" : "orgUnitName";
  }

  onUpdatePageSize(e) {
    this.itemsPerPage = e;
  }

  onDrillDown(e, orgUnitName, parentOrgUnit) {
    e.stopPropagation();
    const orgUnit = _.find(parentOrgUnit ? parentOrgUnit.children : [], [
      "name",
      orgUnitName
    ]);
    if (orgUnit && orgUnit.children && orgUnit.children.length > 0) {
      const orgUnitModel = {
        items: [orgUnit],
        name: "ou",
        value: orgUnit.id + ";LEVEL-" + orgUnit.level
      };

      this.store.dispatch(
        new fromGlobalFiltersActions.GlobalFilterUpdateAction(orgUnitModel)
      );
    }
  }

  onCurrentPageUpdate(e) {
    this.p = e;
  }
  trackWisnReportTable(index, orgUnitWisnItem) {
    return orgUnitWisnItem ? orgUnitWisnItem.id : undefined;
  }

  onSetItemsPerPage(e) {
    if (e) {
      e.stopPropagation();
    }
    const dataValue = e ? e.target.value.trim() : this.itemsPerPage;
    this.itemsPerPage = dataValue;
  }

  onSearchReport(e) {
    if (e) {
      e.stopPropagation();
    }
    this.searchText = e ? e.target.value.trim() : this.searchText;
  }

  sortByHeader(e, dataSource) {
    if (e) {
      e.stopPropagation();
    }
    this.orderByValue = "orgUnitName";
    if (dataSource && dataSource !== `-${this.orderByValue}`) {
      this.orderByValue = dataSource
        ? `-${dataSource}`
        : `-${this.orderByValue}`;
    }
  }

  toggleToViewAll() {
    const reportData = this.wisnReportData ? this.wisnReportData.report : [];
    this.itemsPerPage = reportData.length;
  }

  printReport(printAll?: string) {
    const itemsPerPageBeforePrint = this.itemsPerPage;
    printAll === "allPages" ? this.toggleToViewAll() : null;
    setTimeout(() => {
      window.print();
    }, 20);
    window.onafterprint = () => {
      this.itemsPerPage = itemsPerPageBeforePrint;
    };
  }

  onToggleDropdown(e) {
    if (e) {
      e.stopPropagation();
    }
    e.cancelBubble = true;
    this.dropDownMenuX = 0;
    this.dropDownMenuY = 49;
    this.showDownloadDropdown = !this.showDownloadDropdown;
    this.showPrintDropdown = !this.showPrintDropdown;
    return false;
  }

  downloadTable(data, filterCriteria, downloadFormat) {
    const tableArray = this.filterByPipe.transform(
      this.wisnReportData.report,
      ["cadreCode"],
      filterCriteria,
      true
    );

    this.triggerDownLoadReport.emit(tableArray);
  }

  updateWisnReportFromDataValues() {
    this.store.dispatch(new UpdateWisnReportFromDataValues());
  }
}
