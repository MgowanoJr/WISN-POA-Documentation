import { Component, OnInit } from "@angular/core";
import { AppState } from "../../../../store/reducers";
import {
  style,
  state,
  animate,
  transition,
  trigger
} from "@angular/animations";
import { Store } from "@ngrx/store";
import { getCurrentPage } from "../../../../store/selectors/wisn-poa-pages.selectors";
import { getCurrentGlobalFilter } from "../../../../store/selectors/global-filter.selectors";
import { Observable } from "rxjs";
import * as fromRouterSelectors from "../../../../store/selectors/router.selectors";
import * as fromRouterActions from "../../../../store/actions/router.actions";
import * as fromWisnReportSelectors from "../../../../store/selectors/wisn-report.selectors";
import * as fromWisnReportActions from "../../../../store/actions/wisn-report.actions";
import { getGeneralConfig } from "../../../../store/selectors/general-configuration.selector";
import * as fromAnimations from "../../../../shared/animations/index";
import { DownLoadWisnReport } from "../../../../store/actions/wisn-report.actions";
import { UpdateWisnReportFromDataValues } from "../../../../store/actions/wisn-report.actions";
@Component({
  selector: "app-wisn-report",
  templateUrl: "./wisn-report.component.html",
  styleUrls: ["./wisn-report.component.css"],
  animations: [fromAnimations.listEnterAnimation, fromAnimations.fadeInOut]
})
export class WisnReportComponent implements OnInit {
  wisnReportObject$: Observable<any>;
  generalConfiguration$: Observable<any>;
  globalFilter$: Observable<any>;
  currentPageState$: Observable<any>;
  wisnReportData$: Observable<any>;
  wisnReportLoaded$: Observable<boolean>;
  wisnReportNotification$: Observable<any>;
  allWisnReportData$: Observable<any>;
  wisnReportProgress$: Observable<any>;
  users: any;
  currentLink: string;
  constructor(private store: Store<AppState>) {
    this.wisnReportObject$ = this.store.select(getCurrentPage);
    this.wisnReportData$ = this.store.select(
      fromWisnReportSelectors.getWisnReportEntities
    );
    this.wisnReportLoaded$ = this.store.select(
      fromWisnReportSelectors.getWisnReportLoadedState
    );
    this.generalConfiguration$ = this.store.select(getGeneralConfig);
    this.globalFilter$ = this.store.select(getCurrentGlobalFilter);
    this.currentPageState$ = this.store.select(
      fromRouterSelectors.getCurrentPageState
    );
    this.allWisnReportData$ = this.store.select(
      fromWisnReportSelectors.getAllDataRows
    );

    this.wisnReportNotification$ = this.store.select(
      fromWisnReportSelectors.getWisnReportNotificationState
    );

    this.wisnReportProgress$ = this.store.select(
      fromWisnReportSelectors.getWisnReportProgresLoaderState
    );
  }

  ngOnInit() {}

  onResetNotification(emptyNotificationMessage) {
    this.store.dispatch(
      new fromWisnReportActions.ResetWisnReportNotification({
        notification: emptyNotificationMessage
      })
    );
  }
  onViewAll(e, globalFilter) {
    if (e) {
      e.stopPropagation;
    }
    const viewAllRouteParam = e ? e.target.value.trim() : "";

    this.onToggleCadre(e, viewAllRouteParam, globalFilter);
  }

  onToggleCadre(e, cadrename, globalFilter) {
    if (e) {
      e.stopPropagation();
    }

    this.currentPageState$.subscribe(data => (this.currentLink = data));

    this.store.dispatch(
      new fromRouterActions.Go({
        path: [
          this.currentLink.split("/")[1],
          this.currentLink.split("/")[2],
          cadrename
        ],
        query: {
          orgUnit: globalFilter.ou.id,
          period: globalFilter.pe.id
        }
      })
    );
  }

  downloadWisnReport(tableData) {
    this.store.dispatch(new DownLoadWisnReport(tableData));
  }
}
