import { Component, OnInit, ViewChild, ElementRef } from "@angular/core";
import { Store } from "@ngrx/store";
import { AppState } from "../../../../store/reducers";
import { Observable } from "rxjs";

import { ResetPriListNotification } from "../../../../store/actions/poa-pri-list.actions";
import * as fromPoaPriSelectors from "../../../../store/selectors/poa-pri.selectors";
import * as fromWisnReportSelectors from "../../../../store/selectors/wisn-report.selectors";
import { getCurrentGlobalFilter } from "../../../../store/selectors/global-filter.selectors";
import * as fromAnimations from "../../../../shared/animations/index";
import { VisualizationExportService } from "../../../../shared/services/visualizationExportService.service";

@Component({
  selector: "app-standard-package",
  templateUrl: "./standard-package.component.html",
  styleUrls: ["./standard-package.component.css"],
  animations: [fromAnimations.fadeInOut]
})
export class StandardPackageComponent implements OnInit {
  @ViewChild("standardPackage")
  table: ElementRef;
  poaLoaded$: Observable<boolean>;
  poaPriData$: Observable<any>;
  surplusData$: Observable<any>;
  surplusDataTotals$: Observable<any>;
  globalFilter$: Observable<any>;
  wisnReportProgress$: Observable<any>;
  poaPriNotification$: Observable<any>;

  p: number = 1;
  itemsPerPage: number = 100;

  constructor(
    private store: Store<AppState>,
    private tableExport: VisualizationExportService
  ) {
    this.poaLoaded$ = this.store.select(
      fromPoaPriSelectors.getPoaPriLoadedState
    );

    this.poaPriData$ = this.store.select(
      fromPoaPriSelectors.getPoaPriEntitieState
    );

    this.globalFilter$ = this.store.select(getCurrentGlobalFilter);

    this.surplusData$ = this.store.select(
      fromWisnReportSelectors.getStandardPackageSummary
    );

    this.poaPriNotification$ = this.store.select(
      fromPoaPriSelectors.getPoaPriNotificationState
    );

    this.surplusDataTotals$ = this.store.select(
      fromWisnReportSelectors.getStandardPackageSummaryTotals
    );
    this.wisnReportProgress$ = this.store.select(
      fromWisnReportSelectors.getWisnReportProgresLoaderState
    );
  }

  ngOnInit() {}

  onUpdatePageSize(e) {
    this.itemsPerPage = e;
  }
  printReport(data) {
    window.print();
  }

  onCurrentPageUpdate(e) {
    this.p = e;
  }

  downloadSTDPackage(e, surplusData, poaData, downloadMode) {
    if (this.table) {
      const dateTime = new Date();
      const el = this.table.nativeElement;
      const filename =
        "Standard Package - generated on " +
        dateTime.getFullYear() +
        (dateTime.getMonth() + 1 < 10 ? "-0" : "-") +
        (dateTime.getMonth() + 1) +
        (dateTime.getDay() < 10 ? "-0" : "-") +
        dateTime.getDay() +
        " " +
        (dateTime.getHours() < 10 ? ":0" : ":") +
        dateTime.getHours() +
        (dateTime.getMinutes() < 10 ? ":0" : ":") +
        dateTime.getMinutes() +
        "hrs.xls";
      if (el) {
        if (downloadMode == "ALL") {
          this.tableExport.exportPaginatedCSV(poaData, filename, surplusData);
        } else {
          this.tableExport.exportXLS(filename, el.outerHTML);
        }
      }
    }
  }

  onResetNotification(emptyNotificationMessage) {
    this.store.dispatch(
      new ResetPriListNotification({
        notification: emptyNotificationMessage
      })
    );
  }
}
