import { Component, OnInit, Input, Output, EventEmitter } from "@angular/core";
import { FilterByPipe } from "ngx-pipes";
import * as fromHelpers from "../../../../shared/helpers/index";

@Component({
  selector: "app-wisn-report-pagination",
  templateUrl: "./wisn-report-pagination.component.html",
  styleUrls: ["./wisn-report-pagination.component.css"],
  providers: [FilterByPipe]
})
export class WisnReportPaginationComponent implements OnInit {
  @Input()
  wisnReportDetails;

  @Input()
  paginationControlId;

  @Input()
  wisnReportProgress;

  @Input()
  wisnReportData;

  @Input()
  allWisnReportData;

  @Input()
  filterCriteria;

  @Input()
  itemsPerPage;

  @Input()
  currentPage;

  @Input()
  pageInterval;

  @Output()
  updateItemsPerPage: EventEmitter<any> = new EventEmitter<any>();

  @Output()
  changeCurrentPage: EventEmitter<any> = new EventEmitter<any>();

  maxSize: number;
  pagingIntervals: any;

  constructor(private filterByPipe: FilterByPipe) {}

  ngOnInit() {
    if (this.wisnReportData && this.filterCriteria) {
      this.maxSize = this.filterByPipe.transform(
        this.wisnReportData.report,
        ["cadreCode"],
        this.filterCriteria,
        true
      ).length;

      this.pagingIntervals = fromHelpers.generatePaginationEntryList(
        this.maxSize,
        1
      );
    } else {
      this.populateViewAllEntriesList();
    }
  }

  populateViewAllEntriesList() {
    this.maxSize = this.allWisnReportData ? this.allWisnReportData.length : 0;
    this.pagingIntervals = fromHelpers.generatePaginationEntryList(
      this.maxSize,
      1
    );
  }
  checkIfIsViewAll() {
    // tslint:disable-next-line:no-unused-expression
    this.allWisnReportData ? true : false;
  }
  onSetItemsPerPage(e) {
    if (e) {
      e.stopPropagation();
    }
    const pageSize = e ? e.target.value.trim() : this.itemsPerPage;
    this.updateItemsPerPage.emit((this.itemsPerPage = pageSize));
    this.pageInterval = pageSize;
  }

  onChangeCurrentPage(e) {
    if (e) {
      this.changeCurrentPage.emit((this.currentPage = e));
    }
  }
}

<div class="d-flex justify-content-between">
  <div class="text-sm">
    <p *ngIf="itemsPerPage < maxSize">
      Showing {{ itemsPerPage }} out of {{ maxSize }} records
    </p>
    <p *ngIf="itemsPerPage >= maxSize">
      Showing {{ maxSize }} out of {{ maxSize }} records
    </p>
  </div>
  <div *ngIf="wisnReportDetails">
    <div class="row px-2">
      <div style="width: 100px">
        <select
          class="custom-select"
          id="paginationInput"
          (change)="onSetItemsPerPage($event)"
        >
          <option
            [selected]="first"
            *ngFor="let pageLength of pagingIntervals; let first = first"
            value="{{ pageLength }}"
          >
            {{ pageLength }}
          </option>
          <option value="{{ maxSize }}"> View All</option>
        </select>
      </div>
      <div>
        <pagination-controls
          id="{{ paginationControlId }}"
          (pageChange)="onChangeCurrentPage($event)"
        ></pagination-controls>
      </div>
    </div>
  </div>
</div>
