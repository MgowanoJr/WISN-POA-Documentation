import { EntityState, EntityAdapter, createEntityAdapter } from '@ngrx/entity';
import { WisnPresetValue } from '../models/wisn-preset-value.model';
import {
  PresetValuesDataProgressLoader,
  WisnPresetValueActions,
  WisnPresetValueActionTypes
} from '../actions/wisn-preset-value.actions';
import * as fromWisnPresetHelper from '../../shared/helpers/wisn-preset-value.helpers';

export interface State extends EntityState<WisnPresetValue> {
  // additional entities state properties
  loading: boolean;
  loaded: boolean;
  presetValueOrgunt: any;
  dataValues: any;
  notification: string;
  formDesign: any;
  sections: any;
  progressLoader: any;
  presetFacilityTypes: any;
  currentFacilityType: string;
}

export const adapter: EntityAdapter<WisnPresetValue> = createEntityAdapter<
  WisnPresetValue
>();

export const initialState: State = adapter.getInitialState({
  // additional entity state properties
  loaded: false,
  loading: false,
  presetValueOrgunt: { id: '', name: '' },
  notification: '',
  dataValues: [],
  formDesign: '',
  sections: [],
  progressLoader: 0,
  presetFacilityTypes: [],
  currentFacilityType: 'dispensary'
});

export function presetValuesReducer(
  state = initialState,
  action: WisnPresetValueActions
): State {
  switch (action.type) {
    case WisnPresetValueActionTypes.AddWisnPresetValue: {
      return adapter.addOne(action.payload.wisnPresetValue, {
        ...state,
        loaded: true
      });
    }

    case WisnPresetValueActionTypes.UpsertWisnPresetValue: {
      const presetValueState = { ...state };
      presetValueState.presetValueOrgunt = action.wisnPresetValue;
      return { ...presetValueState };
    }

    case WisnPresetValueActionTypes.ActivateOrgUnitForEntry: {
      const facilityType = action.payload.id;
      return { ...state,
        loaded: true,
        loading: false,
        currentFacilityType: facilityType };
    }

    case WisnPresetValueActionTypes.ResetWisnPresetValueNotification: {
      return { ...state, notification: null };
    }

    case WisnPresetValueActionTypes.AddWisnPresetValues: {
      return {
        ...state,
        dataValues: action.payload.dataValues,
        presetFacilityTypes: action.payload.presetFacilityTypes,
        presetValueOrgunt: action.payload.presetValueOrgunt
      };
    }

    case WisnPresetValueActionTypes.UpsertWisnPresetValues: {
      const dataValues = fromWisnPresetHelper
      .updatePresetDataValue(Object.assign({}, state.dataValues), state.currentFacilityType, action.payload);
      return { ...state, dataValues: dataValues };
    }

    case WisnPresetValueActionTypes.UploadWisnPresetValue: {
      return state;
    }

    case WisnPresetValueActionTypes.UpdateWisnPresetValue: {
      const presetValueState = {
        ...state,
        sections: action.data.sections,
        progressLoader: action.data.progressLoader
      };
      return presetValueState;
    }

    case WisnPresetValueActionTypes.UpdateWisnPresetValues: {
      return adapter.updateMany(action.payload.wisnPresetValues, state);
    }

    case WisnPresetValueActionTypes.PreLoadWisnPresetValue: {
      const facilityType = action.payload.id;
      const presetFacilityTypes = fromWisnPresetHelper
      .togglePresetFacilityTypes(state.presetFacilityTypes, facilityType);
      return {...state,
        loaded: false,
        loading: true,
        presetFacilityTypes: presetFacilityTypes
      };
    }

    case WisnPresetValueActionTypes.UpdateWisnFormDesign: {
      return { ...state,
        loaded: true,
        loading: false,
        notification: 'Wisn Preset Values Successfully Loaded',
        formDesign: action.payload.formDesign,
        progressLoader: action.payload.progressLoader};
    }

    case WisnPresetValueActionTypes.LoadWisnPresetValues: {
      return {
        ...state,
        loaded: false,
        loading: true,
        progressLoader: 0,
        notification: 'Loading Wisn Preset Values Form'
      };
    }

    case WisnPresetValueActionTypes.ClearWisnPresetValues: {
      return adapter.removeAll(state);
    }

    case WisnPresetValueActionTypes.PresetValuesDataProgressLoader: {
      return { ...state, progressLoader: action.payload };
    }

    default: {
      return state;
    }
  }
}
