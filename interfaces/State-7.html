import { EntityState, EntityAdapter, createEntityAdapter } from '@ngrx/entity';
import { WisnHrhData } from '../models/wisn-hrh-data.model';
import {
  AddWisnHrhDataElements,
  CompleteDownloadingWisnHrhDatas,
  DownloadingWisnHrhDatas,
  WisnHrhDataActions,
  WisnHrhDataActionTypes, WisnHRHErrorHandler
} from '../actions/wisn-hrh-data.actions';

export interface State extends EntityState<WisnHrhData> {
  // additional entities state properties
  ids: any;
  entities: any;
  loaded: boolean;
  loading: boolean;
  currentcadre: string;
  wisnHrhDataElements: any;
  hrhProgram: any;
  notification: any;
  lastDataSelection: any;
  progressLoader: any;
}

export const adapter: EntityAdapter<WisnHrhData> = createEntityAdapter<
  WisnHrhData
>();

export const initialState: State = adapter.getInitialState({
  // additional entity state properties
  ids: [],
  entities: [],
  loaded: false,
  loading: false,
  currentcadre: '',
  notification: '',
  wisnHrhDataElements: [],
  hrhProgram: [],
  lastDataSelection: {},
  progressLoader: 0
});

export function hrhReducer(
  state = initialState,
  action: WisnHrhDataActions
): State {
  switch (action.type) {
    case WisnHrhDataActionTypes.AddWisnHrhData: {
      const hrhDataState = {
        ...state,
        ids: action.wisnHrhData.stateEntityIds,
        entities: [...state.entities, ...action.wisnHrhData.wisnHrhData],
        loaded: true,
        loading: false,
        notification: 'HRH Data Form Successfully Loaded',
        progressLoader: action.wisnHrhData.progressLoader,
        wisnHrhDataElements: action.wisnHrhData.wisnHrhDataElements,
        lastDataSelection: action.wisnHrhData.lastDataSelection
      };
      return hrhDataState;
    }

    case WisnHrhDataActionTypes.AddWisnHrhDataElements: {
      const hrhState = { ...state };
      hrhState.wisnHrhDataElements = action.wisnHrhDataElements;
      return hrhState;
    }

    case WisnHrhDataActionTypes.UpdatingWisnHRHData: {
      return { ...state };
    }

    case WisnHrhDataActionTypes.UpsertWisnHrhData: {
      const hrhDataState = { ...state, notification: '' };
      const objectData = {};
      objectData[action.payload['changedData'].dataElementId] =
        action.payload['changedData'].value;
      const dataPayload = { ...action.payload['rowData'], ...objectData }; // re-create the row payload
      hrhDataState.entities[action.payload['index']] = dataPayload;
      return hrhDataState;
    }

    case WisnHrhDataActionTypes.AddWisnHrhDatas: {
      return adapter.addMany(action.payload.wisnHrhDatas, state);
    }

    case WisnHrhDataActionTypes.UpsertWisnHrhDatas: {
      return adapter.upsertMany(action.payload.wisnHrhDatas, state);
    }

    case WisnHrhDataActionTypes.UpdateWisnHrhData: {
      return adapter.updateOne(action.payload.wisnHrhData, state);
    }

    case WisnHrhDataActionTypes.UpdateWisnHrhDatas: {
      return adapter.updateMany(action.payload.wisnHrhDatas, state);
    }

    case WisnHrhDataActionTypes.DeleteWisnHrhData: {
      return adapter.removeOne(action.payload.id, state);
    }

    case WisnHrhDataActionTypes.DeleteWisnHrhDatas: {
      return adapter.removeMany(action.payload.ids, state);
    }

    case WisnHrhDataActionTypes.LoadWisnHrhDatas: {
      return {
        ...state,
        notification: 'Loading HRH Form',
        entities: [],
        progressLoader: 0,
        loaded: false,
        loading: true
      };
    }

    case WisnHrhDataActionTypes.ClearWisnHrhDatas: {
      return adapter.removeAll(state);
    }

    case WisnHrhDataActionTypes.LoadHRHprogramData: {
      return { ...state };
    }

    case WisnHrhDataActionTypes.AddHRHprogramDataValues: {
      return { ...state, hrhProgram: action.hrhProgram };
    }

    case WisnHrhDataActionTypes.CheckingWisnHRHState: {
      return { ...state };
    }

    case WisnHrhDataActionTypes.NoChangesOnWisnHRHState: {
      return { ...state };
    }

    case WisnHrhDataActionTypes.DownloadingWisnHrhDatas: {
      return { ...state };
    }

    case WisnHrhDataActionTypes.CompleteDownloadingWisnHrhDatas: {
      return { ...state };
    }

    case WisnHrhDataActionTypes.UploadingWisnHrhDatas: {
      return { ...state, notification: 'Uploading HRH data to server'};
    }

    case WisnHrhDataActionTypes.CompleteUploadingWisnHrhDatas: {
      return { ...state, entities: action.payload };
    }
    case WisnHrhDataActionTypes.ResetWisnHrhFormNotification: {
      return { ...state, notification: null };
    }

    case WisnHrhDataActionTypes.WisnHRHErrorHandler: {
      return { ...state, notification: action.status.message };
    }

    default: {
      return state;
    }
  }
}

export const {
  selectIds: getWisnHrhDataIdsState,

  selectAll: getAllWisnHrhDataState
} = adapter.getSelectors();

export const getWisnHrhNotificationState = (state: State) => state.notification;

  state.wisnHrhDataElements;
