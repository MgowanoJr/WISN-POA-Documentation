import { EntityState, EntityAdapter, createEntityAdapter } from '@ngrx/entity';
import { PoaCostAssumption } from '../models/poa-cost-assumption.model';
import {
  AddPoaCostAssumptionProgram, CostAssumputionsDataProgressLoader, CostAssumputionsMaxNumberHRH, CostAssumputionsSalaryCost,
  LoadPoaCostAssumptionsProgram,
  PoaCostAssumptionActions,
  PoaCostAssumptionActionTypes,
  POAcostAssumputionsErrorHandler,
  UpdatingPoaCostAssumptionsProgram
} from '../actions/poa-cost-assumption.actions';

export interface State extends EntityState<PoaCostAssumption> {
  // additional entities state properties
  id: any;
  entities: any;
  salaryCost: any;
  allowanceCost: any;
  maximumBudget: any;
  maximumNumber: any;
  loaded: boolean;
  loading: boolean;
  lastDataSelection: any;
  costAssumptionProgramDataValues: any;
  notification: any;
  progressLoader: any;
  remainingMaxBudget: number;
  maxHRHLoaded: boolean;
}

export const adapter: EntityAdapter<PoaCostAssumption> = createEntityAdapter<
  PoaCostAssumption
>();

export const initialState: State = adapter.getInitialState({
  // additional entity state properties
  id: [],
  entities: [],
  salaryCost: [],
  allowanceCost: [],
  maximumBudget: [],
  maximumNumber: [],
  costAssumptionProgramDataValues: [],
  loaded: false,
  loading: false,
  notification: '',
  lastDataSelection: {},
  progressLoader: 0,
  remainingMaxBudget: 0,
  maxHRHLoaded: false
});

export function poaCostAssumptionReducer(
  state = initialState,
  action: PoaCostAssumptionActions
): State {
  switch (action.type) {
    case PoaCostAssumptionActionTypes.AddPoaCostAssumption: {
      const costAssumptionState = {
        ...state,
        salaryCost: action.poaCostAssumption.monthlySalaryData,
        allowanceCost: action.poaCostAssumption.allowanceCostData,
        maximumBudget: action.poaCostAssumption.maximumBudget,
        maximumNumber: action.poaCostAssumption.maximumNumber,
        progressLoader: action.poaCostAssumption.progressLoader,
        remainingMaxBudget: action.poaCostAssumption.remainingMaxBudget,
        loaded: true,
        loading: false,
        notification: 'POA Cost Assumptions Loaded Successfully',
        lastDataSelection: action.poaCostAssumption.lastDataSelection
      };

      return costAssumptionState;
    }

    case PoaCostAssumptionActionTypes.UpsertPoaCostAssumption: {
      const costAssumptionDataState = { ...state };
      const objectData = { value: '' };
      objectData.value = action.payload.changedData.value;
      const dataPayload = { ...action.payload.rowData, ...objectData }; // re-create the row payload
      costAssumptionDataState[action.payload.rowData.entityName][
        action.payload.index
      ] = dataPayload;
      return costAssumptionDataState;
    }

    case PoaCostAssumptionActionTypes.AddPoaCostAssumptions: {
      return adapter.addMany(action.payload.poaCostAssumptions, state);
    }

    case PoaCostAssumptionActionTypes.UpsertPoaCostAssumptions: {
      return adapter.upsertMany(action.payload.poaCostAssumptions, state);
    }

    case PoaCostAssumptionActionTypes.UpdatePoaCostAssumption: {
      return adapter.updateOne(action.payload.poaCostAssumption, state);
    }

    case PoaCostAssumptionActionTypes.UpdatePoaCostAssumptions: {
      return adapter.updateMany(action.payload.poaCostAssumptions, state);
    }

    case PoaCostAssumptionActionTypes.DeletePoaCostAssumption: {
      return adapter.removeOne(action.payload.id, state);
    }

    case PoaCostAssumptionActionTypes.DeletePoaCostAssumptions: {
      return adapter.removeMany(action.payload.ids, state);
    }

    case PoaCostAssumptionActionTypes.LoadPoaCostAssumptions: {
      return {
        ...state,
        loaded: false,
        loading: true,
        progressLoader: 0,
        notification: 'Loading POA Cost Assumptions'
      };
    }

    case PoaCostAssumptionActionTypes.CheckingPoaCostAssumptionsState: {
      return { ...state };
    }

    case PoaCostAssumptionActionTypes.NoChangesOnPoaCostAssumptionsState: {
      return { ...state };
    }

    case PoaCostAssumptionActionTypes.LoadPoaCostAssumptionsProgram: {
      return { ...state, notification: 'Loading POA Cost Assumptions Program' };
    }

    case PoaCostAssumptionActionTypes.AddPoaCostAssumptionProgram: {
      return {
        ...state,
        costAssumptionProgramDataValues: action.poaCostAssumptionDataValues
      };
    }

    case PoaCostAssumptionActionTypes.UpdatingPoaCostAssumptionsProgram: {
      return { ...state };
    }

    case PoaCostAssumptionActionTypes.ClearPoaCostAssumptions: {
      return adapter.removeAll(state);
    }

    case PoaCostAssumptionActionTypes.ResetCostAssumptionsNotification: {
      return { ...state, notification: null };
    }

    case PoaCostAssumptionActionTypes.POAcostAssumputionsErrorHandler: {
      return { ...state, notification: action.status.message };
    }

    case PoaCostAssumptionActionTypes.CostAssumputionsDataProgressLoader: {
      return {...state, progressLoader: action.payload};
    }

    case PoaCostAssumptionActionTypes.CostAssumputionsSalaryCost: {
      return {...state,
      salaryCost: action.payload.salaryCost,
      maximumBudget: action.payload.maximumBudget,
      maximumNumber: action.payload.maximumNumber,
      remainingMaxBudget: action.payload.remainingMaxBudget,
    };
    }

    case PoaCostAssumptionActionTypes.CostAssumputionsMaxNumberHRH: {
      return {...state,
        maxHRHLoaded: true,
        maximumBudget: action.payload.maximumBudget,
        maximumNumber: action.payload.maximumNumber,
        remainingMaxBudget: action.payload.remainingMaxBudget,
      };
    }

    case PoaCostAssumptionActionTypes.CostAssumptionsLoadMaxHRHFromPRIlist: {
      return {...state, notification: 'Generating Maximum HRH based from PRI need'};
    }

    default: {
      return state;
    }
  }
}
