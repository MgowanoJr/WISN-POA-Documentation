import { ActionReducerMap, createSelector, MetaReducer } from '@ngrx/store';
import { storeFreeze } from 'ngrx-store-freeze';
import { environment } from '../../../environments/environment';
import { routerReducer, RouterReducerState } from '@ngrx/router-store';
import { currentUserReducer, CurrentUserState } from './current-user.reducer';
import * as fromGlobalFilter from './global-filter.reducer';
import * as fromWisnPoaPages from './wisn-poa-pages.reducer';
import * as fromGeneralConfigurations from './general-configuration.reducer';
import * as fromWisnReport from './wisn-report.reducer';
import * as fromWisnHmisData from './wisn-hmis-data.reducer';
import * as fromWisnHrhData from './wisn-hrh-data.reducer';
import * as fromWisnPresetValue from './wisn-preset-value.reducer';
import * as fromPoaPriList from './poa-pri-list.reducer';
import * as fromPoaCostAssumption from './poa-cost-assumption.reducer';
import * as fromPoaDecidedSurplus from './poa-decided-surplus.reducer';
import * as fromSetupEnvironment from './setup-environment.reducer';

export interface AppState {
  route: RouterReducerState;
  currentUser: CurrentUserState;
  globalFilter: fromGlobalFilter.GlobalFilterState;
  wisnPoaPages: fromWisnPoaPages.State;
  generalConfigurations: fromGeneralConfigurations.State;
  wisnReport: fromWisnReport.State;
  wisnHmisData: fromWisnHmisData.State;
  wisnHrhData: fromWisnHrhData.State;
  wisnPresetValue: fromWisnPresetValue.State;
  poaPriList: fromPoaPriList.State;
  poaCostAssumption: fromPoaCostAssumption.State;
  poaDecidedSurplus: fromPoaDecidedSurplus.State;
  setupEnvironment: fromSetupEnvironment.State;
}

export const reducers: ActionReducerMap<AppState> = {
  route: routerReducer,
  currentUser: currentUserReducer,
  globalFilter: fromGlobalFilter.globalFilterReducer,
  generalConfigurations: fromGeneralConfigurations.globalConfigurationReducer,
  wisnPoaPages: fromWisnPoaPages.pagesReducer,
  wisnReport: fromWisnReport.WisnReportReducer,
  wisnHmisData: fromWisnHmisData.hmisDataReducer,
  wisnHrhData: fromWisnHrhData.hrhReducer,
  wisnPresetValue: fromWisnPresetValue.presetValuesReducer,
  poaPriList: fromPoaPriList.poaPRIreducer,
  poaCostAssumption: fromPoaCostAssumption.poaCostAssumptionReducer,
  poaDecidedSurplus: fromPoaDecidedSurplus.poaDecidedSurplusReducer,
  setupEnvironment: fromSetupEnvironment.setupEnvironmentReducer
};

export const getRootState = (state: AppState) => state;

/**
 * Get current user state
 */
export const getCurrentUserState = createSelector(
  getRootState,
  (state: AppState) => state.currentUser
);

/**
 * Get global filter state
 * @param {AppState} state
 */
export const getGlobalFilterState = (state: AppState) => state.globalFilter;

export const getGlobalFilterEntityState = createSelector(
  getGlobalFilterState,
  state => state
);

export const {
  selectEntities: getGlobalFilterEntities
} = fromGlobalFilter.adapter.getSelectors(getGlobalFilterEntityState);

/**
 * Get router state
 * @param {AppState} state
 */
export const getRouterState = (state: AppState) => state.route;

/**
 * Get wisn poa pages state
 * @param {AppState} state
 */
export const getWisnPoaPagesState = (state: AppState) => state.wisnPoaPages;

export const getWisnPoaPagesEntityState = createSelector(
  getWisnPoaPagesState,
  state => state
);

export const {
  selectEntities: getWisnPoaPagesEntities
} = fromWisnPoaPages.adapter.getSelectors(getWisnPoaPagesEntityState);

/**
 * Get wisn report state
 * @param {AppState} state
 */
export const getWisnReportState = (state: AppState) => state.wisnReport;

/**
 * Get global configuration state
 * @param {AppState} state
 */
export const getGlobalConfigurationState = (state: AppState) => state.generalConfigurations;

export const getWisnReportEntityState = createSelector(
  getWisnReportState,
  state => state
);

export const {
  selectEntities: getWisnReportEntities
} = fromWisnReport.adapter.getSelectors(getWisnReportEntityState);

export const metaReducers: MetaReducer<AppState>[] = !environment.production
  ? []
  : [];
