import { EntityState, EntityAdapter, createEntityAdapter } from '@ngrx/entity';
import { PoaDecidedSurplus } from '../models/poa-decided-surplus.model';
import {
  CollectingSurplus,
  NoChangesOnDecidedSurplusState,
  PoaDecidedSurplusActions,
  PoaDecidedSurplusActionTypes,
  UpdateCollectingSurplusSummary,
  UpdatePoaDecidedSurplusSummary
} from '../actions/poa-decided-surplus.actions';

export interface State extends EntityState<PoaDecidedSurplus> {
  // additional entities state properties
  loaded: boolean;
  loading: boolean;
  lastDataSelection: any;
  decidedSummary: any;
  redistributionSurplusAvailable: any;
  notification: any;
}

export const adapter: EntityAdapter<PoaDecidedSurplus> = createEntityAdapter<
  PoaDecidedSurplus
>();

export const initialState: State = adapter.getInitialState({
  // additional entity state properties
  loaded: false,
  loading: false,
  lastDataSelection: {},
  decidedSummary: [],
  notification: '',
  redistributionSurplusAvailable: []
});

export function poaDecidedSurplusReducer(
  state = initialState,
  action: PoaDecidedSurplusActions
): State {
  switch (action.type) {
    case PoaDecidedSurplusActionTypes.AddPoaDecidedSurplus: {
      return adapter.addAll(action.poaDecidedSurplus.data, {
        ...state,
        lastDataSelection: action.poaDecidedSurplus['lastDataSelection'],
        decidedSummary: action.poaDecidedSurplus['decidedSummary'],
        loaded: true,
        notification: 'Surplus List Loaded Successfully',
        loading: false
      });
    }

    case PoaDecidedSurplusActionTypes.UpsertPoaDecidedSurplus: {
      return adapter.upsertOne(action.payload.poaDecidedSurplus, state);
    }

    case PoaDecidedSurplusActionTypes.AddPoaDecidedSurpluss: {
      return adapter.addMany(action.payload.poaDecidedSurpluss, state);
    }

    case PoaDecidedSurplusActionTypes.UpsertPoaDecidedSurpluss: {
      return adapter.upsertMany(action.payload.poaDecidedSurpluss, state);
    }

    case PoaDecidedSurplusActionTypes.UpdatePoaDecidedSurplus: {
      return { ...state };
    }

    case PoaDecidedSurplusActionTypes.UpdatePoaDecidedSurpluss: {
      return adapter.updateMany(action.payload['surplusState'], {
        ...state,
        decidedSummary: action.payload['surplusSummary'],
        redistributionSurplusAvailable: action.payload['summaryCollection']
      });
    }

    case PoaDecidedSurplusActionTypes.DeletePoaDecidedSurplus: {
      return adapter.removeOne(action.payload.id, state);
    }

    case PoaDecidedSurplusActionTypes.DeletePoaDecidedSurpluss: {
      return adapter.removeMany(action.payload.ids, state);
    }

    case PoaDecidedSurplusActionTypes.LoadPoaDecidedSurpluss: {
      return {
        ...state,
        loaded: false,
        loading: true,
        notification: 'Loading Surplus List'
      };
    }

    case PoaDecidedSurplusActionTypes.UpdatePoaDecidedSurplusSummary: {
      return { ...state };
    }

    case PoaDecidedSurplusActionTypes.UpdatedPoaDecidedSurplusSummary: {
      return {
        ...state,
        redistributionSurplusAvailable:
          action.payload.redistributionSurplusAvailable,
        decidedSummary: action.payload.decidedSummary
      };
    }

    case PoaDecidedSurplusActionTypes.CheckingDecidedSurplusState: {
      return { ...state };
    }

    case PoaDecidedSurplusActionTypes.NoChangesOnDecidedSurplusState: {
      return { ...state };
    }

    case PoaDecidedSurplusActionTypes.ClearPoaDecidedSurpluss: {
      return adapter.removeAll(state);
    }

    case PoaDecidedSurplusActionTypes.CollectingSurplus: {
      return state;
    }

    case PoaDecidedSurplusActionTypes.UpdateCollectingSurplusSummary: {
      return { ...state, redistributionSurplusAvailable: [] };
    }
    case PoaDecidedSurplusActionTypes.ResetDecidedSurplusNotification: {
      return { ...state, notification: null };
    }
    default: {
      return state;
    }
  }
}

export const {
  selectIds: getPoaDecidedSurplusDataIdsState,
  selectEntities: getPoaDecidedSurplusDataEntitiesState,
  selectAll: getAllPoaDecidedSurplusDataState,
  selectTotal
} = adapter.getSelectors();

export const getPoaDecidedSurplusLoadingState = (state: State) => state.loading;
export const getPoaDecidedSurplusNotificationState = (state: State) =>
  state.notification;

  state.decidedSummary;
export const getPoaDecidedSurplusSummaryAvailableState = (state: State) =>
  state.redistributionSurplusAvailable;
