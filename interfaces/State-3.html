import { EntityState, EntityAdapter, createEntityAdapter } from '@ngrx/entity';
import { PoaPriList } from '../models/poa-pri-list.model';
import {
  FetchingPoaPRIStateDraft,
  PoaPriListActions,
  PoaPriListActionTypes, ResetPoaPriList,
  SavingPoaPRIStateDraft
} from '../actions/poa-pri-list.actions';

export interface State extends EntityState<PoaPriList> {
  // additional entities state properties
  ids: any;
  entities: any;
  loaded: boolean;
  loading: boolean;
  currentcadre: string;
  lastDataSelection: any;
  redistributionSummary: any;
  distributionSummary: any;
  notification: any;
  priDraftTime: any;
}

export const adapter: EntityAdapter<PoaPriList> = createEntityAdapter<
  PoaPriList
>();

export const initialState: State = adapter.getInitialState({
  // additional entity state properties
  ids: [],
  entities: [],
  loaded: false,
  loading: false,
  currentcadre: '',
  notification: '',
  redistributionSummary: [],
  distributionSummary: [],
  lastDataSelection: {},
  priDraftTime: ''
});

export function poaPRIreducer(
  state = initialState,
  action: PoaPriListActions
): State {
  switch (action.type) {
    case PoaPriListActionTypes.AddPoaPriList: {
      const priState = {
        ...state,
        entities: [...action.poaPriList['poaPriList']],
        loaded: true,
        loading: false,
        priDraftTime: action.poaPriList.priDraftTime,
        notification: 'POA Report Successfully loaded',
        lastDataSelection: action.poaPriList['lastDataSelection']
      };
      return priState;
    }

    case PoaPriListActionTypes.UpsertPoaPriList: {
      return adapter.upsertOne(action.payload.poaPriList, state);
    }

    case PoaPriListActionTypes.AddPoaPriLists: {
      return adapter.addMany(action.payload.poaPriLists, state);
    }

    case PoaPriListActionTypes.UpsertPoaPriLists: {
      return adapter.upsertMany(action.payload.poaPriLists, state);
    }

    case PoaPriListActionTypes.UpdatePoaPriList: {
      const priState = {
        ...state,
        entities: action.poaPriList['priState'],
        distributionSummary: action.poaPriList['distributionSummary'],
        redistributionSummary: action.poaPriList['redistributionSummary']
      };
      return priState;
    }

    case PoaPriListActionTypes.UpdatePoaPriLists: {
      return adapter.updateMany(action.payload.poaPriLists, state);
    }

    case PoaPriListActionTypes.DeletePoaPriList: {
      return adapter.removeOne(action.payload.id, state);
    }

    case PoaPriListActionTypes.DeletePoaPriLists: {
      return adapter.removeMany(action.payload.ids, state);
    }

    case PoaPriListActionTypes.LoadPoaPriLists: {
      return {
        ...state,
        loaded: false,
        loading: true,
        notification: 'Loading POA Report from WISN Report'
      };
    }

    case PoaPriListActionTypes.CheckingWisnOutputs: {
      return {
        ...state,
        loaded: true,
        loading: false,
        notification: 'Checking WISN Outputs'
      };
    }

    case PoaPriListActionTypes.ClearPoaPriLists: {
      return adapter.removeAll(state);
    }

    case PoaPriListActionTypes.CheckingPoaPRIState: {
      return { ...state, notification: 'Checking POA PRI State' };
    }

    case PoaPriListActionTypes.NoChangesOnPoaPRIState: {
      return {
        ...state,
        loaded: true,
        loading: false,
      };
    }

    case PoaPriListActionTypes.DownloadPoaPriLists: {
      return state;
    }

    case PoaPriListActionTypes.CompleteDownloadPoaPriLists: {
      return state;
    }

    case PoaPriListActionTypes.UpdatingPoaPriListFromDecidedSurplus: {
      return {
        ...state,
        notification: 'Updating POA Report from decided surplus'
      };
    }

    case PoaPriListActionTypes.UpdatePoaPriListFromDecidedSurplus: {
      return { ...state, entities: action.payload };
    }

    case PoaPriListActionTypes.ApplyAutoConstraintsOnPRIstate: {
      return {
        ...state,
        notification: 'Distributing staff by supplied constraints'
      };
    }

    case PoaPriListActionTypes.CompletedApplyingAutoConstraintsOnPRIstate: {
      return {
        ...state,
        notification: 'Distribution Constraints applied successfully',
        entities: [...action.payload]
      };
    }

    case PoaPriListActionTypes.ApplyAutoConstraintsOnPRIstateRedistribute: {
      return state;
    }

    case PoaPriListActionTypes.CompletedApplyingAutoConstraintsOnPRIstateRedistribute: {
      return { ...state, entities: action.payload };
    }

    case PoaPriListActionTypes.SavingPoaPRIStateDraft: {
      return { ...state, notification: 'Saving POA Draft' };
    }
    case PoaPriListActionTypes.ResetPriListNotification: {
      return { ...state, notification: null };
    }

    case PoaPriListActionTypes.FetchingPoaPRIStateDraft: {
      return {...state, notification: 'Fetching last PRI draft'};
    }

    case PoaPriListActionTypes.ResetPoaPriList: {
      return state;
    }

    default: {
      return state;
    }
  }
}
