import { EntityState, EntityAdapter, createEntityAdapter } from '@ngrx/entity';
import { WisnHmisData } from '../models/wisn-hmis-data.model';
import {
  CompleteDownLaodWisnHmisData,
  DownLaodWisnHmisData, HmisDataProgressLoader,
  NoChangesOnWisnHmisState, ResetHmiState,
  UploadWisnHmisDatas,
  WisnHmisDataActions,
  WisnHmisDataActionTypes, WisnHMISErrorHandler
} from '../actions/wisn-hmis-data.actions';

export interface State extends EntityState<WisnHmisData> {
  // additional entities state properties
  ids: any;
  entities: any;
  loaded: boolean;
  loading: boolean;
  currentcadre: string;
  hmisDataElements: any;
  lastDataSelection: any;
  notification: string;
  progressLoader: any;
}

export const adapter: EntityAdapter<WisnHmisData> = createEntityAdapter<
  WisnHmisData
>();

export const initialState: State = adapter.getInitialState({
  // additional entity state properties
  ids: [],
  entities: { hmisData: [] },
  loaded: false,
  loading: false,
  currentcadre: '',
  hmisDataElements: [],
  lastDataSelection: {},
  notification: '',
  progressLoader: 0
});

export function hmisDataReducer(
  state = initialState,
  action: WisnHmisDataActions
): State {
  switch (action.type) {
    case WisnHmisDataActionTypes.AddWisnHmisData: {
      const hmisDataState = {
        ...state,
        ids: action.wisnHmisData.stateEntityIds,
        entities: { hmisData: [...state.entities.hmisData, ...action.wisnHmisData.wisnHmisData] },
        notification: 'Loading HMIS form',
        progressLoader: action.wisnHmisData.progressLoader,
        loaded: true,
        loading: false,
        lastDataSelection: action.wisnHmisData.lastDataSelection,
        hmisDataElements: action.wisnHmisData.hmisDataElements
      };
      return hmisDataState;
      // return adapter.addMany(action.wisnHmisData['wisnHmisData'], state);
    }

    case WisnHmisDataActionTypes.UpsertWisnHmisData: {
      return {
        ...state,
        notification: 'Adding data to HMIS Form',
        progressLoader: action.payload.progressCount,
        entities: {
          hmisData: [...state.entities['hmisData'], ...action.payload.data]
        }
      };
    }

    case WisnHmisDataActionTypes.AddWisnHmisDatas: {
      return adapter.addMany(action.payload.wisnHmisDatas, state);
    }

    case WisnHmisDataActionTypes.UpsertWisnHmisDatas: {
      return adapter.upsertMany(action.payload.wisnHmisDatas, state);
    }

    case WisnHmisDataActionTypes.ResetHmisDataNotification: {
      return { ...state, notification: null };
    }

    case WisnHmisDataActionTypes.HmisDataProgressLoader: {
      return {...state, progressLoader: action.payload};
    }

    case WisnHmisDataActionTypes.UpdateWisnHmisData: {
      const hmisDataState = {
        ...state,
        notification: 'HMIS Form has been successfully updated'
      };
      const objectData = {};
      objectData[action.payload.changedData.dataElementId] =
        action.payload.changedData.value;
      const dataPayload = { ...action.payload.rowData, ...objectData }; // re-create the row payload
      hmisDataState.entities.hmisData[action.payload.index] = dataPayload;
      return hmisDataState;
    }

    case WisnHmisDataActionTypes.UpdateWisnHmisDatas: {
      return adapter.updateMany(action.payload.wisnHmisDatas, state);
    }

    case WisnHmisDataActionTypes.DeleteWisnHmisData: {
      return adapter.removeOne(action.payload.id, state);
    }

    case WisnHmisDataActionTypes.DeleteWisnHmisDatas: {
      return adapter.removeMany(action.payload.ids, state);
    }

    case WisnHmisDataActionTypes.LoadWisnHmisDatas: {
      return {
        ...state,
        notification: 'Loading HMIS Form',
        progressLoader: 0,
        loaded: false,
        loading: true
      };
    }

    case WisnHmisDataActionTypes.ClearWisnHmisDatas: {
      return adapter.removeAll(state);
    }

    case WisnHmisDataActionTypes.CheckingWisnHmisState: {
      return { ...state };
    }

    case WisnHmisDataActionTypes.NoChangesOnWisnHmisState: {
      return { ...state };
    }

    case WisnHmisDataActionTypes.DownLaodWisnHmisData: {
      return { ...state };
    }

    case WisnHmisDataActionTypes.CompleteDownLaodWisnHmisData: {
      return { ...state };
    }

    case WisnHmisDataActionTypes.UploadWisnHmisDatas: {
      return { ...state };
    }

    case WisnHmisDataActionTypes.WisnHMISErrorHandler: {
      return { ...state, notification: action.status.message };
    }

    case WisnHmisDataActionTypes.ResetHmiState: {
      return { ...state, entities: { hmisData: []}, progressLoader: 0};
    }

    default: {
      return state;
    }
  }
}
