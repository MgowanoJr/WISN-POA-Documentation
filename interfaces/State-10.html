import { EntityState, EntityAdapter, createEntityAdapter } from '@ngrx/entity';
import { WisnReport } from '../models/wisn-report.model';
import {
  DownLoadWisnReport, UpdateWisnReportFromDataValues,
  UploadWisnReport, WisnDataUpdateInBackground, WisnErrorHandler,
  WisnReportActions,
  WisnReportActionTypes
} from '../actions/wisn-report.actions';

export interface State extends EntityState<WisnReport> {
  // additional entities state properties
  currentcadre: string;
  entities: any;
  loaded: boolean;
  loading: boolean;
  uploaded: boolean;
  notification: string;
  lastDataSelection: any;
  progressLoader: any;
  iswisnReportFetchingData: boolean;
}

export const adapter: EntityAdapter<WisnReport> = createEntityAdapter<
  WisnReport
>();

export const initialState: State = adapter.getInitialState({
  // additional entity state properties
  ids: [],
  entities: { report: [] },
  loaded: false,
  loading: false,
  uploaded: false,
  currentcadre: '',
  lastDataSelection: {},
  notification: '',
  facilitiesDataValues: [],
  progressLoader: 0,
  iswisnReportFetchingData: false
});

export function WisnReportReducer(
  state = initialState,
  action: WisnReportActions
): State {
  switch (action.type) {
    case WisnReportActionTypes.AddWisnReport: {
      const reportState = {
        ...state,
        loaded: true,
        uploaded: false,
        loading: false,
        notification: 'Wisn Report Data successfully loaded',
        progressLoader: action.wisnReport.progressLoader,
        lastDataSelection: action.wisnReport.lastDataSelection,
        iswisnReportFetchingData: action.wisnReport.iswisnReportFetchingData ?
        action.wisnReport.iswisnReportFetchingData : false
      };
      reportState.entities = { report: action.wisnReport.wisnOutputs };

      return reportState;
    }

    case WisnReportActionTypes.UpsertWisnReport: {
      return {...state,
        entities: {report: [...action.payload.data]},
        // entities: {report: [...state.entities.report, ...action.payload.data]},
        lastDataSelection: action.payload.lastDataSelection,
        progressLoader: action.payload.progressLoader,
        iswisnReportFetchingData: action.payload.iswisnReportFetchingData ?
        action.payload.iswisnReportFetchingData : false,
        loaded: true,
        uploaded: false,
        loading: false,
      };
    }

    case WisnReportActionTypes.AddWisnReports: {
      return adapter.addMany(action.payload.wisnReports, state);
    }

    case WisnReportActionTypes.WisnProgressLoaderUpdate: {
      return {...state,
        progressLoader: action.payload.progress,
        iswisnReportFetchingData: action.payload.iswisnReportFetchingData ?
        action.payload.iswisnReportFetchingData : false,
        notification: action.payload.notification
      };
    }

    case WisnReportActionTypes.UpsertWisnReports: {
      return adapter.upsertMany(action.payload.wisnReports, state);
    }

    case WisnReportActionTypes.UpdateWisnReport: {
      return adapter.updateOne(action.payload.wisnReport, state);
    }

    case WisnReportActionTypes.UpdateWisnReports: {
      return adapter.updateMany(action.payload.wisnReports, state);
    }

    case WisnReportActionTypes.DeleteWisnReport: {
      return adapter.removeOne(action.payload.id, state);
    }

    case WisnReportActionTypes.DeleteWisnReports: {
      return {...state,
        entities: {report: []},
        loaded: false,
        loading: true,
        notification: 'Loading Wisn Report Data' };
    }

    case WisnReportActionTypes.LoadWisnReports: {
      return {
        ...state,
        loaded: false,
        loading: true,
        // progressLoader: 0,
        notification: 'Loading Wisn Report Data...'
      };
    }

    case WisnReportActionTypes.ClearWisnReports: {
      return adapter.removeAll(state);
    }

    case WisnReportActionTypes.ResetWisnReportNotification: {
      return { ...state, notification: null };
    }

    case WisnReportActionTypes.CheckingWisnReport: {
      return state;
    }

    case WisnReportActionTypes.NoChangesOnWisnReport: {
      return state;
    }

    case WisnReportActionTypes.DownLoadWisnReport: {
      return { ...state, notification: 'Downloading Wisn Report Data' };
    }

    case WisnReportActionTypes.UploadWisnReport: {
      return { ...state, notification: 'Uploading Wisn Report Data' };
    }

    case WisnReportActionTypes.CompleteDownLoadWisnReport: {
      return state;
    }

    case WisnReportActionTypes.CompleteUpLoadingWisnReport: {
      const reportState = {
        ...state,
        loaded: true,
        uploaded: true,
        loading: false,
        notification: 'Wisn Report Data Successfully updated',
        lastDataSelection: action.payload['lastDataSelection']
      };
      reportState.entities = { report: action.payload['wisnOutputs'] };

      return reportState;
    }

    case WisnReportActionTypes.UpdateWisnReportFromDataValues: {
      return { ...state, loaded: false, loading: true,
        notification: 'Loading Wisn Report Data'};
    }

    case WisnReportActionTypes.WisnErrorHandler: {
      return {...state, notification: action.status.message };
    }

    case WisnReportActionTypes.WisnDataUpdateInBackground: {
      return state;
    }

    default: {
      return state;
    }
  }
}

export const getWisnReportAppState = (state: State) => state;
export const getWisnReportLoadedState = (state: State) => state.loaded;
export const getWisnReportNotificationState = (state: State) =>
  state.notification;
export const getWisnReportProgresLoader = (state: State) => state.progressLoader;

export const {
  selectIds: getWisnReportIdsState,
  selectEntities: getWisnReportEntitiesState,
  selectAll: getWisnReportState
} = adapter.getSelectors();
